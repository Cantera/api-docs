.. code-block:: Matlab

    function plotSolution(s, domain, component)
        n = s.stackIndex(domain);
        d = s.domains{n};
        z = d.gridPoints;
        x = s.getSolution(domain, component);
        plot(z, x);
        xlabel('z (m)');
        ylabel(component);
    end
