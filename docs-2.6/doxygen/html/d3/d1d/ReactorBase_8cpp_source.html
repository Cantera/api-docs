108doublereal ReactorBase::residenceTime()
109{
110 doublereal mout = 0.0;
111 for (size_t i = 0; i < m_outlet.size(); i++) {
112 mout += m_outlet[i]->massFlowRate();
113 }
114 return mass()/mout;
