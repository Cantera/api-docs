52void PDSS_IdealGas::setPressure(doublereal p)
53{

59void PDSS_IdealGas::setTemperature(doublereal temp)
60{
61 m_temp = temp;
62 m_spthermo->updatePropertiesTemp(temp, &m_cp0_R, &m_h0_RT, &m_s0_R);
63 m_g0_RT = m_h0_RT - m_s0_R;
64 m_V0 = GasConstant * m_temp / m_p0;
65 m_hss_RT = m_h0_RT;
66 m_cpss_R = m_cp0_R;
67 m_sss_R = m_s0_R - log(m_pres/m_p0);
68 m_gss_RT = m_hss_RT - m_sss_R;
69 m_Vss = GasConstant * m_temp / m_pres;
70}
