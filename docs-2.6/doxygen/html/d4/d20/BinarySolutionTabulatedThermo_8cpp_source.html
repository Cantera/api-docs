45void BinarySolutionTabulatedThermo::_updateThermo() const
46{
47 double xnow = moleFraction(m_kk_tab);
48 bool x_changed = (m_xlast != xnow);
49
50 if (x_changed) {
51 std::tie(m_h0_tab, m_s0_tab) = interpolate(xnow);
52 if (xnow == 0) {
53 m_s0_tab = -BigNumber;
54 } else if (xnow == 1) {
55 m_s0_tab = BigNumber;
56 } else {
57 m_s0_tab += GasConstant*std::log(xnow/(1.0-xnow)) +
58 GasConstant/Faraday*std::log(standardConcentration(1-m_kk_tab)
59 /standardConcentration(m_kk_tab));
60 }
61 m_xlast = xnow;
62 }
63
64 double tnow = temperature();
65 if (x_changed || m_tlast != tnow) {
66 // Update the thermodynamic functions of the reference state.
67 m_spthermo.update(tnow, m_cp0_R.data(), m_h0_RT.data(), m_s0_R.data());
68 double rrt = 1.0 / RT();
69 m_h0_RT[m_kk_tab] += m_h0_tab * rrt;
70 m_s0_R[m_kk_tab] += m_s0_tab / GasConstant;
71 for (size_t k = 0; k < m_kk; k++) {
72 m_g0_RT[k] = m_h0_RT[k] - m_s0_R[k];
73 }
74 m_tlast = tnow;
75 }
76}

78void BinarySolutionTabulatedThermo::initThermo()
79{
80 if (m_input.hasKey("tabulated-thermo")) {
81 m_kk_tab = speciesIndex(m_input["tabulated-species"].asString());
82 if (m_kk_tab == npos) {
83 throw InputFileError("BinarySolutionTabulatedThermo::initThermo",
84 m_input["tabulated-species"],
85 "Species '{}' is not in phase '{}'",
86 m_input["tabulated-species"].asString(), name());
87 }
88 const AnyMap& table = m_input["tabulated-thermo"].as<AnyMap>();
89 vector_fp x = table["mole-fractions"].asVector<double>();
90 size_t N = x.size();
91 vector_fp h = table.convertVector("enthalpy", "J/kmol", N);
92 vector_fp s = table.convertVector("entropy", "J/kmol/K", N);
93
94 // Sort the x, h, s data in the order of increasing x
95 std::vector<std::pair<double,double>> x_h(N), x_s(N);
96 for(size_t i = 0; i < N; i++){
97 x_h[i] = {x[i], h[i]};
98 x_s[i] = {x[i], s[i]};
99 }
100 std::sort(x_h.begin(), x_h.end());
101 std::sort(x_s.begin(), x_s.end());
102
103 // Store the sorted values in different arrays
104 m_molefrac_tab.resize(N);
105 m_enthalpy_tab.resize(N);
106 m_entropy_tab.resize(N);
107 for (size_t i = 0; i < N; i++) {
108 m_molefrac_tab[i] = x_h[i].first;
109 m_enthalpy_tab[i] = x_h[i].second;
110 m_entropy_tab[i] = x_s[i].second;
111 }
112 }
113 IdealSolidSolnPhase::initThermo();
114}

127void BinarySolutionTabulatedThermo::initThermoXML(XML_Node& phaseNode, const std::string& id_)
128{
129 vector_fp x, h, s;
130 std::vector<std::pair<double,double>> x_h_temp, x_s_temp;
131
132 if (id_.size() > 0) {
133 if (phaseNode.id() != id_) {
134 throw CanteraError("BinarySolutionTabulatedThermo::initThermoXML",
135 "phasenode and Id are incompatible");
136 }
137 }
138 if (nSpecies()!=2) {
139 throw CanteraError("BinarySolutionTabulatedThermo::initThermoXML",
140 "No. of species should be equal to 2!");
141 }
142 if (phaseNode.hasChild("thermo")) {
143 XML_Node& thermoNode = phaseNode.child("thermo");
144 std::string mString = thermoNode["model"];
145 if (!caseInsensitiveEquals(mString, "binarysolutiontabulatedthermo")) {
146 throw CanteraError("BinarySolutionTabulatedThermo::initThermoXML",
147 "Unknown thermo model: " + mString);
148 }
149 if (thermoNode.hasChild("tabulatedSpecies")) {
150 XML_Node& speciesNode = thermoNode.child("tabulatedSpecies");
151 std::string tabulated_species_name = speciesNode["name"];
152 m_kk_tab = speciesIndex(tabulated_species_name);
153 if (m_kk_tab == npos) {
154 throw CanteraError("BinarySolutionTabulatedThermo::initThermoXML",
155 "Species " + tabulated_species_name + " not found.");
156 }
157 }
158 if (thermoNode.hasChild("tabulatedThermo")) {
159 XML_Node& dataNode = thermoNode.child("tabulatedThermo");
160 getFloatArray(dataNode, x, true, "", "moleFraction");
161 getFloatArray(dataNode, h, true, "J/kmol", "enthalpy");
162 getFloatArray(dataNode, s, true, "J/kmol/K", "entropy");
163
164 // Check for data length consistency
165 if ((x.size() != h.size()) || (x.size() != s.size()) || (h.size() != s.size())) {
166 throw CanteraError("BinarySolutionTabulatedThermo::initThermoXML",
167 "Species tabulated thermo data has different lengths.");
168 }
169 // Sort the x, h, s data in the order of increasing x
170 for(size_t i = 0; i < x.size(); i++){
171 x_h_temp.push_back(std::make_pair(x[i],h[i]));
172 x_s_temp.push_back(std::make_pair(x[i],s[i]));
173 }
174 std::sort(x_h_temp.begin(), x_h_temp.end());
175 std::sort(x_s_temp.begin(), x_s_temp.end());
176
177 // Store the sorted values in different arrays
178 m_molefrac_tab.resize(x_h_temp.size());
179 m_enthalpy_tab.resize(x_h_temp.size());
180 m_entropy_tab.resize(x_h_temp.size());
181 for (size_t i = 0; i < x_h_temp.size(); i++) {
182 m_molefrac_tab[i] = x_h_temp[i].first;
183 m_enthalpy_tab[i] = x_h_temp[i].second;
184 m_entropy_tab[i] = x_s_temp[i].second;
185 }
186 } else {
187 throw CanteraError("BinarySolutionTabulatedThermo::initThermoXML",
188 "Unspecified tabulated species or thermo");
189 }
190 } else {
191 throw CanteraError("BinarySolutionTabulatedThermo::initThermoXML",
192 "Unspecified thermo model");
193 }
194
195 /*
196 * Form of the standard concentrations. Must have one of:
197 *
198 * <standardConc model="unity" />
199 * <standardConc model="molar_volume" />
200 * <standardConc model="solvent_volume" />
201 */
202 if (phaseNode.hasChild("standardConc")) {
203 XML_Node& scNode = phaseNode.child("standardConc");
204 setStandardConcentrationModel(scNode.attrib("model"));
205 } else {
206 throw CanteraError("BinarySolutionTabulatedThermo::initThermoXML",
207 "Unspecified standardConc model");
208 }
209
210 // Call the base initThermo, which handles setting the initial state
211 ThermoPhase::initThermoXML(phaseNode, id_);
212}

214std::pair<double, double> BinarySolutionTabulatedThermo::interpolate(double x) const
215{
216 std::pair<double, double> c;
217 // Check if x is out of bound
218 if (x > m_molefrac_tab.back()) {
219 c.first = m_enthalpy_tab.back();
220 c.second = m_entropy_tab.back();
221 return c;
222 }
223 if (x < m_molefrac_tab[0]) {
224 c.first = m_enthalpy_tab[0];
225 c.second = m_entropy_tab[0];
226 return c;
227 }
228 size_t i = std::distance(m_molefrac_tab.begin(),
229 std::lower_bound(m_molefrac_tab.begin(), m_molefrac_tab.end(), x));
230 c.first = m_enthalpy_tab[i-1] + (m_enthalpy_tab[i] - m_enthalpy_tab[i-1])
231 * (x - m_molefrac_tab[i-1])/(m_molefrac_tab[i]- m_molefrac_tab[i-1]);
232 c.second = m_entropy_tab[i-1] + (m_entropy_tab[i] - m_entropy_tab[i-1])
233 * (x - m_molefrac_tab[i-1])/(m_molefrac_tab[i]- m_molefrac_tab[i-1]);
234 return c;
235}
