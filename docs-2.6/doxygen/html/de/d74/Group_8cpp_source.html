15void Group::validate()
16{
17 // if already checked and not valid, return
18 if (m_sign == -999) {
19 return;
20 }
21
22 m_sign = 0;
23 bool ok = true;
24 for (size_t m = 0; m < m_comp.size(); m++) {
25 if (m_comp[m] != 0) {
26 if (m_sign == 0) {
27 m_sign = m_comp[m]/abs(m_comp[m]);
28 } else if (m_sign * m_comp[m] < 0) {
29 ok = false;
30 break;
31 }
32 }
33 }
34 if (!ok) {
35 m_sign = -999;
36 }
37}
