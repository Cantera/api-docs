340void Phase::setMoleFractions(const double* const x)
341{
342 // Use m_y as a temporary work vector for the non-negative mole fractions
343 double norm = 0.0;
344 // sum is calculated below as the unnormalized molecular weight
345 double sum = 0;
346 for (size_t k = 0; k < m_kk; k++) {
347 double xk = std::max(x[k], 0.0); // Ignore negative mole fractions
348 m_y[k] = xk;
349 norm += xk;
350 sum += m_molwts[k] * xk;
351 }
352
353 // Set m_ym to the normalized mole fractions divided by the normalized mean
354 // molecular weight:
355 // m_ym_k = X_k / (sum_k X_k M_k)
356 const double invSum = 1.0/sum;
357 for (size_t k=0; k < m_kk; k++) {
358 m_ym[k] = m_y[k]*invSum;
359 }
360
361 // Now set m_y to the normalized mass fractions:
362 // m_y = X_k M_k / (sum_k X_k M_k)
363 for (size_t k=0; k < m_kk; k++) {
364 m_y[k] = m_ym[k] * m_molwts[k];
365 }
366
367 // Calculate the normalized molecular weight
368 m_mmw = sum/norm;
369 compositionChanged();
370}

392void Phase::setMassFractions(const double* const y)
393{
394 for (size_t k = 0; k < m_kk; k++) {
395 m_y[k] = std::max(y[k], 0.0); // Ignore negative mass fractions
396 }
397 double norm = accumulate(m_y.begin(), m_y.end(), 0.0);
398 scale(m_y.begin(), m_y.end(), m_y.begin(), 1.0/norm);
399
400 transform(m_y.begin(), m_y.end(), m_rmolwts.begin(),
401 m_ym.begin(), multiplies<double>());
402 m_mmw = 1.0 / accumulate(m_ym.begin(), m_ym.end(), 0.0);
403 compositionChanged();
404}

602void Phase::setConcentrations(const double* const conc)
603{
604 assertCompressible("setConcentrations");
605
606 // Use m_y as temporary storage for non-negative concentrations
607 double sum = 0.0, norm = 0.0;
608 for (size_t k = 0; k != m_kk; ++k) {
609 double ck = std::max(conc[k], 0.0); // Ignore negative concentrations
610 m_y[k] = ck;
611 sum += ck * m_molwts[k];
612 norm += ck;
613 }
614 m_mmw = sum/norm;
615 setDensity(sum);
616 double rsum = 1.0/sum;
617 for (size_t k = 0; k != m_kk; ++k) {
618 m_ym[k] = m_y[k] * rsum;
619 m_y[k] = m_ym[k] * m_molwts[k]; // m_y is now the mass fraction
620 }
621 compositionChanged();
622}

643doublereal Phase::elementalMassFraction(const size_t m) const
644{
645 checkElementIndex(m);
646 doublereal Z_m = 0.0;
647 for (size_t k = 0; k != m_kk; ++k) {
648 Z_m += nAtoms(k, m) * atomicWeight(m) / molecularWeight(k)
649 * massFraction(k);
650 }
651 return Z_m;
652}

654doublereal Phase::elementalMoleFraction(const size_t m) const
655{
656 checkElementIndex(m);
657 double denom = 0;
658 for (size_t k = 0; k < m_kk; k++) {
659 double atoms = 0;
660 for (size_t j = 0; j < nElements(); j++) {
661 atoms += nAtoms(k, j);
662 }
663 denom += atoms * moleFraction(k);
664 }
665 doublereal numerator = 0.0;
666 for (size_t k = 0; k != m_kk; ++k) {
667 numerator += nAtoms(k, m) * moleFraction(k);
668 }
669 return numerator / denom;
670}

737size_t Phase::addElement(const std::string& symbol, doublereal weight,
738 int atomic_number, doublereal entropy298,
739 int elem_type)
740{
741 // Look up the atomic weight if not given
742 if (weight == 0.0) {
743 try {
744 weight = getElementWeight(symbol);
745 } catch (CanteraError&) {
746 // assume this is just a custom element with zero atomic weight
747 }
748 } else if (weight == -12345.0) {
749 weight = getElementWeight(symbol);
750 }
751
752 // Try to look up the standard entropy if not given. Fail silently.
753 if (entropy298 == ENTROPY298_UNKNOWN) {
754 try {
755 const static AnyMap db = AnyMap::fromYamlFile(
756 "element-standard-entropies.yaml");
757 const AnyMap& elem = db["elements"].getMapWhere("symbol", symbol);
758 entropy298 = elem.convert("entropy298", "J/kmol/K", ENTROPY298_UNKNOWN);
759 } catch (CanteraError&) {
760 }
761 }
762
763 // Check for duplicates
764 auto iter = find(m_elementNames.begin(), m_elementNames.end(), symbol);
765 if (iter != m_elementNames.end()) {
766 size_t m = iter - m_elementNames.begin();
767 if (m_atomicWeights[m] != weight) {
768 throw CanteraError("Phase::addElement",
769 "Duplicate elements ({}) have different weights", symbol);
770 } else {
771 // Ignore attempt to add duplicate element with the same weight
772 return m;
773 }
774 }
775
776 // Add the new element
777 m_atomicWeights.push_back(weight);
778 m_elementNames.push_back(symbol);
779 m_atomicNumbers.push_back(atomic_number);
780 m_entropy298.push_back(entropy298);
781 if (symbol == "E") {
782 m_elem_type.push_back(CT_ELEM_TYPE_ELECTRONCHARGE);
783 } else {
784 m_elem_type.push_back(elem_type);
785 }
786 m_mm++;
787
788 // Update species compositions
789 if (m_kk) {
790 vector_fp old(m_speciesComp);
791 m_speciesComp.resize(m_kk*m_mm, 0.0);
792 for (size_t k = 0; k < m_kk; k++) {
793 size_t m_old = m_mm - 1;
794 for (size_t m = 0; m < m_old; m++) {
795 m_speciesComp[k * m_mm + m] = old[k * (m_old) + m];
796 }
797 m_speciesComp[k * (m_mm) + (m_mm-1)] = 0.0;
798 }
799 }
800
801 return m_mm-1;
802}

804bool Phase::addSpecies(shared_ptr<Species> spec) {
805 if (m_species.find(spec->name) != m_species.end()) {
806 throw CanteraError("Phase::addSpecies",
807 "Phase '{}' already contains a species named '{}'.",
808 m_name, spec->name);
809 }
810 vector_fp comp(nElements());
811 for (const auto& elem : spec->composition) {
812 size_t m = elementIndex(elem.first);
813 if (m == npos) { // Element doesn't exist in this phase
814 switch (m_undefinedElementBehavior) {
815 case UndefElement::ignore:
816 return false;
817
818 case UndefElement::add:
819 addElement(elem.first);
820 comp.resize(nElements());
821 m = elementIndex(elem.first);
822 break;
823
824 case UndefElement::error:
825 default:
826 throw CanteraError("Phase::addSpecies",
827 "Species '{}' contains an undefined element '{}'.",
828 spec->name, elem.first);
829 }
830 }
831 comp[m] = elem.second;
832 }
833
834 m_speciesNames.push_back(spec->name);
835 m_species[spec->name] = spec;
836 m_speciesIndices[spec->name] = m_kk;
837 m_speciesCharge.push_back(spec->charge);
838 size_t ne = nElements();
839
840 std::string nLower = toLowerCopy(spec->name);
841 if (m_speciesLower.find(nLower) == m_speciesLower.end()) {
842 m_speciesLower[nLower] = m_kk;
843 } else {
844 m_speciesLower[nLower] = npos;
845 }
846
847 double wt = 0.0;
848 const vector_fp& aw = atomicWeights();
849 if (spec->charge != 0.0) {
850 size_t eindex = elementIndex("E");
851 if (eindex != npos) {
852 doublereal ecomp = comp[eindex];
853 if (fabs(spec->charge + ecomp) > 0.001) {
854 if (ecomp != 0.0) {
855 throw CanteraError("Phase::addSpecies",
856 "Input charge and element E compositions differ "
857 "for species " + spec->name);
858 } else {
859 // Just fix up the element E composition based on the input
860 // species charge
861 comp[eindex] = -spec->charge;
862 }
863 }
864 } else {
865 addElement("E", 0.000545, 0, 0.0, CT_ELEM_TYPE_ELECTRONCHARGE);
866 ne = nElements();
867 eindex = elementIndex("E");
868 comp.resize(ne);
869 comp[ne - 1] = - spec->charge;
870 }
871 }
872 for (size_t m = 0; m < ne; m++) {
873 m_speciesComp.push_back(comp[m]);
874 wt += comp[m] * aw[m];
875 }
876
877 // Some surface phases may define species representing empty sites
878 // that have zero molecular weight. Give them a very small molecular
879 // weight to avoid dividing by zero.
880 wt = std::max(wt, Tiny);
881 m_molwts.push_back(wt);
882 m_rmolwts.push_back(1.0/wt);
883 m_kk++;
884
885 // Ensure that the Phase has a valid mass fraction vector that sums to
886 // one. We will assume that species 0 has a mass fraction of 1.0 and mass
887 // fraction of all other species is 0.0.
888 if (m_kk == 1) {
889 m_y.push_back(1.0);
890 m_ym.push_back(m_rmolwts[0]);
891 m_mmw = 1.0 / m_ym[0];
892 } else {
893 m_y.push_back(0.0);
894 m_ym.push_back(0.0);
895 }
896 invalidateCache();
897 return true;
898}

900void Phase::modifySpecies(size_t k, shared_ptr<Species> spec)
901{
902 if (speciesName(k) != spec->name) {
903 throw CanteraError("Phase::modifySpecies",
904 "New species name '{}' does not match existing name '{}'",
905 spec->name, speciesName(k));
906 }

908 if (spec->composition != old->composition) {
909 throw CanteraError("Phase::modifySpecies",
910 "New composition for '{}' does not match existing composition",
911 spec->name);
912 }
913 m_species[spec->name] = spec;
914 invalidateCache();
915}

1023void Phase::massFractionsToMoleFractions(const double* Y, double* X) const
1024{
1025 double rmmw = 0.0;
1026 for (size_t k = 0; k != m_kk; ++k) {
1027 rmmw += Y[k]/m_molwts[k];
1028 }
1029 if (rmmw == 0.0) {
1030 throw CanteraError("Phase::massFractionsToMoleFractions",
1031 "no input composition given");
1032 }
1033 for (size_t k = 0; k != m_kk; ++k) {
1034 X[k] = Y[k]/(rmmw*m_molwts[k]);
1035 }
1036}
1037
1038void Phase::moleFractionsToMassFractions(const double* X, double* Y) const
1039{
1040 double mmw = dot(X, X+m_kk, m_molwts.data());
1041 if (mmw == 0.0) {
1042 throw CanteraError("Phase::moleFractionsToMassFractions",
1043 "no input composition given");
1044 }
1045 double rmmw = 1.0/mmw;
1046 for (size_t k = 0; k != m_kk; ++k) {
1047 Y[k] = X[k]*m_molwts[k]*rmmw;
1048 }
1049}
