99 void Sim1D::setProfile(size_t dom, size_t comp,
100  const vector<double>& pos, const vector<double>& values)
101 {
102  if (pos.front() != 0.0 || pos.back() != 1.0) {
103  throw CanteraError("Sim1D::setProfile",
104  "`pos` vector must span the range [0, 1]. Got a vector spanning "
105  "[{}, {}] instead.", pos.front(), pos.back());
106  }
107  Domain1D& d = domain(dom);
108  double z0 = d.zmin();
109  double z1 = d.zmax();
110  for (size_t n = 0; n < d.nPoints(); n++) {
111  double zpt = d.z(n);
112  double frac = (zpt - z0)/(z1 - z0);
113  double v = linearInterp(frac, pos, values);
114  setValue(dom, comp, n, v);
115  }

129 void Sim1D::save(const string& fname, const string& name, const string& desc,
130  bool overwrite, int compression, const string& basis)
131 {
132  size_t dot = fname.find_last_of(".");
133  string extension = (dot != npos) ? toLowerCopy(fname.substr(dot+1)) : "";
134  if (extension == "csv") {
135  for (auto dom : m_dom) {
136  auto arr = dom->asArray(m_state->data() + dom->loc());
137  if (dom->size() > 1) {
138  arr->writeEntry(fname, overwrite, basis);
139  break;
140  }
141  }
142  return;
143  }
144  if (basis != "") {
145  warn_user("Sim1D::save",
146  "Species basis '{}' not implemented for HDF5 or YAML output.", basis);
147  }
148  if (extension == "h5" || extension == "hdf" || extension == "hdf5") {
149  SolutionArray::writeHeader(fname, name, desc, overwrite);
150  for (auto dom : m_dom) {
151  auto arr = dom->asArray(m_state->data() + dom->loc());
152  arr->writeEntry(fname, name, dom->id(), overwrite, compression);
153  }
154  return;
155  }
156  if (extension == "yaml" || extension == "yml") {
157  // Check for an existing file and load it if present
158  AnyMap data;
159  if (std::ifstream(fname).good()) {
160  data = AnyMap::fromYamlFile(fname);
161  }
162  SolutionArray::writeHeader(data, name, desc, overwrite);
163 
164  for (auto dom : m_dom) {
165  auto arr = dom->asArray(m_state->data() + dom->loc());
166  arr->writeEntry(data, name, dom->id(), overwrite);
167  }
168 
169  // Write the output file and remove the now-outdated cached file
170  std::ofstream out(fname);
171  out << data.toYamlString();
172  AnyMap::clearCachedFile(fname);
173  return;
174  }
175  throw CanteraError("Sim1D::save", "Unsupported file format '{}'.", extension);
176 }

204 //! convert data format used by Python h5py export (Cantera < 3.0)
205 AnyMap legacyH5(shared_ptr<SolutionArray> arr, const AnyMap& header={})
206 {
207  auto meta = arr->meta();
208  AnyMap out;
209 
210  map<string, string> meta_pairs = {
211  {"type", "Domain1D_type"},
212  {"name", "name"},
213  {"emissivity-left", "emissivity_left"},
214  {"emissivity-right", "emissivity_right"},
215  };
216  for (const auto& [newName, oldName] : meta_pairs) {
217  if (meta.hasKey(oldName)) {
218  out[newName] = meta[oldName];
219  }
220  }
221 
222  map<string, string> tol_pairs = {
223  {"transient-abstol", "transient_abstol"},
224  {"steady-abstol", "steady_abstol"},
225  {"transient-reltol", "transient_reltol"},
226  {"steady-reltol", "steady_reltol"},
227  };
228  for (const auto& [newName, oldName] : tol_pairs) {
229  if (meta.hasKey(oldName)) {
230  out["tolerances"][newName] = meta[oldName];
231  }
232  }
233 
234  if (meta.hasKey("phase")) {
235  out["phase"]["name"] = meta["phase"]["name"];
236  out["phase"]["source"] = meta["phase"]["source"];
237  }
238 
239  if (arr->size() <= 1) {
240  return out;
241  }
242 
243  map<string, string> header_pairs = {
244  {"transport-model", "transport_model"},
245  {"radiation-enabled", "radiation_enabled"},
246  {"energy-enabled", "energy_enabled"},
247  {"Soret-enabled", "soret_enabled"},
248  {"species-enabled", "species_enabled"},
249  };
250  for (const auto& [newName, oldName] : header_pairs) {
251  if (header.hasKey(oldName)) {
252  out[newName] = header[oldName];
253  }
254  }
255 
256  map<string, string> refiner_pairs = {
257  {"ratio", "ratio"},
258  {"slope", "slope"},
259  {"curve", "curve"},
260  {"prune", "prune"},
261  // {"grid-min", "???"}, // missing
262  {"max-points", "max_grid_points"},
263  };
264  for (const auto& [newName, oldName] : refiner_pairs) {
265  if (header.hasKey(oldName)) {
266  out["refine-criteria"][newName] = header[oldName];
267  }
268  }
269 
270  if (header.hasKey("fixed_temperature")) {
271  double temp = header.getDouble("fixed_temperature", -1.);
272  auto profile = arr->getComponent("T").as<vector<double>>();
273  int ix = 0;
274  while (profile[ix] <= temp && ix < arr->size()) {
275  ix++;
276  }
277  if (ix != 0) {
278  auto grid = arr->getComponent("grid").as<vector<double>>();
279  out["fixed-point"]["location"] = grid[ix - 1];
280  out["fixed-point"]["temperature"] = temp;
281  }
282  }
283 
284  return out;
285 }

296 AnyMap Sim1D::restore(const string& fname, const string& name)
297 {
298  size_t dot = fname.find_last_of(".");
299  string extension = (dot != npos) ? toLowerCopy(fname.substr(dot+1)) : "";
300  if (extension == "xml") {
301  throw CanteraError("Sim1D::restore",
302  "Restoring from XML is no longer supported.");
303  }
304  AnyMap header;
305  if (extension == "h5" || extension == "hdf" || extension == "hdf5") {
306  map<string, shared_ptr<SolutionArray>> arrs;
307  header = SolutionArray::readHeader(fname, name);
308 
309  for (auto dom : m_dom) {
310  auto arr = SolutionArray::create(dom->solution());
311  arr->readEntry(fname, name, dom->id());
312  dom->resize(dom->nComponents(), arr->size());
313  if (!header.hasKey("generator")) {
314  arr->meta() = legacyH5(arr, header);
315  }
316  arrs[dom->id()] = arr;
317  }
318  resize();
319  m_xlast_ts.clear();
320  for (auto dom : m_dom) {
321  dom->fromArray(*arrs[dom->id()], m_state->data() + dom->loc());
322  }
323  finalize();
324  } else if (extension == "yaml" || extension == "yml") {
325  AnyMap root = AnyMap::fromYamlFile(fname);
326  map<string, shared_ptr<SolutionArray>> arrs;
327  header = SolutionArray::readHeader(root, name);
328 
329  for (auto dom : m_dom) {
330  auto arr = SolutionArray::create(dom->solution());
331  arr->readEntry(root, name, dom->id());
332  dom->resize(dom->nComponents(), arr->size());
333  arrs[dom->id()] = arr;
334  }
335  resize();
336  m_xlast_ts.clear();
337  for (auto dom : m_dom) {
338  dom->fromArray(*arrs[dom->id()], m_state->data() + dom->loc());
339  }
340  finalize();
341  } else {
342  throw CanteraError("Sim1D::restore",
343  "Unknown file extension '{}'; supported extensions include "
344  "'h5'/'hdf'/'hdf5' and 'yml'/'yaml'.", extension);
345  }
346  return header;
347 }

450 void Sim1D::solve(int loglevel, bool refine_grid)
451 {
452  int new_points = 1;
453  double dt = m_tstep;
454  m_nsteps = 0;
455  finalize();
456 
457  while (new_points > 0) {
458  size_t istep = 0;
459  int nsteps = m_steps[istep];
460 
461  bool ok = false;
462  if (loglevel > 0) {
463  writeline('.', 78, true, true);
464  }
465  while (!ok) {
466  // Attempt to solve the steady problem
467  setSteadyMode();
468  newton().setOptions(m_ss_jac_age);
469  debuglog("Attempt Newton solution of steady-state problem...", loglevel);
470  int status = newtonSolve(loglevel-1);
471 
472  if (status == 0) {
473  if (loglevel > 0) {
474  writelog(" success.\n\n");
475  writelog("Problem solved on [");
476  for (size_t mm = 1; mm < nDomains(); mm+=2) {
477  writelog("{}", domain(mm).nPoints());
478  if (mm + 2 < nDomains()) {
479  writelog(", ");
480  }
481  }
482  writelog("] point grid(s).\n");
483  }
484  if (m_steady_callback) {
485  m_steady_callback->eval(0);
486  }
487 
488  if (loglevel > 6) {
489  save("debug_sim1d.yaml", "debug",
490  "After successful Newton solve");
491  }
492  if (loglevel > 7) {
493  saveResidual("debug_sim1d.yaml", "residual",
494  "After successful Newton solve");
495  }
496  ok = true;
497  } else {
498  debuglog(" failure. \n", loglevel);
499  if (loglevel > 6) {
500  save("debug_sim1d.yaml", "debug",
501  "After unsuccessful Newton solve");
502  }
503  if (loglevel > 7) {
504  saveResidual("debug_sim1d.yaml", "residual",
505  "After unsuccessful Newton solve");
506  }
507  if (loglevel > 0) {
508  writelog("Take {} timesteps ", nsteps);
509  }
510  dt = timeStep(nsteps, dt, m_state->data(), m_xnew.data(), loglevel-1);
511  m_xlast_ts = *m_state;
512  if (loglevel > 6) {
513  save("debug_sim1d.yaml", "debug", "After timestepping");
514  }
515  if (loglevel > 7) {
516  saveResidual("debug_sim1d.yaml", "residual",
517  "After timestepping");
518  }
519 
520  if (loglevel == 1) {
521  writelog(" {:10.4g} {:10.4g}\n", dt,
522  log10(ssnorm(m_state->data(), m_xnew.data())));
523  }
524  istep++;
525  if (istep >= m_steps.size()) {
526  nsteps = m_steps.back();
527  } else {
528  nsteps = m_steps[istep];
529  }
530  dt = std::min(dt, m_tmax);
531  }
532  }
533  if (loglevel > 0) {
534  writeline('.', 78, true, true);
535  }
536  if (loglevel > 2) {
537  show();
538  }
539 
540  if (refine_grid) {
541  new_points = refine(loglevel);
542  if (new_points) {
543  // If the grid has changed, preemptively reduce the timestep
544  // to avoid multiple successive failed time steps.
545  dt = m_tstep;
546  }
547  if (new_points && loglevel > 6) {
548  save("debug_sim1d.yaml", "debug", "After regridding");
549  }
550  if (new_points && loglevel > 7) {
551  saveResidual("debug_sim1d.yaml", "residual",
552  "After regridding");
553  }
554  } else {
555  debuglog("grid refinement disabled.\n", loglevel);
556  new_points = 0;
557  }
558  }
559 }

561 int Sim1D::refine(int loglevel)
562 {
563  int ianalyze, np = 0;
564  vector<double> znew, xnew;
565  vector<size_t> dsize;
566 
567  m_xlast_ss = *m_state;
568  m_grid_last_ss.clear();
569 
570  for (size_t n = 0; n < nDomains(); n++) {
571  Domain1D& d = domain(n);
572  Refiner& r = d.refiner();
573 
574  // Save the old grid corresponding to the converged solution
575  m_grid_last_ss.push_back(d.grid());
576 
577  // determine where new points are needed
578  ianalyze = r.analyze(d.grid().size(), d.grid().data(),
579  m_state->data() + start(n));
580  if (ianalyze < 0) {
581  return ianalyze;
582  }
583 
584  if (loglevel > 0) {
585  r.show();
586  }
587 
588  np += r.nNewPoints();
589  size_t comp = d.nComponents();
590 
591  // loop over points in the current grid
592  size_t npnow = d.nPoints();
593  size_t nstart = znew.size();
594  for (size_t m = 0; m < npnow; m++) {
595  if (r.keepPoint(m)) {
596  // add the current grid point to the new grid
597  znew.push_back(d.grid(m));
598 
599  // do the same for the solution at this point
600  for (size_t i = 0; i < comp; i++) {
601  xnew.push_back(value(n, i, m));
602  }
603 
604  // now check whether a new point is needed in the interval to
605  // the right of point m, and if so, add entries to znew and xnew
606  // for this new point
607  if (r.newPointNeeded(m) && m + 1 < npnow) {
608  // add new point at midpoint
609  double zmid = 0.5*(d.grid(m) + d.grid(m+1));
610  znew.push_back(zmid);
611  np++;
612 
613  // for each component, linearly interpolate
614  // the solution to this point
615  for (size_t i = 0; i < comp; i++) {
616  double xmid = 0.5*(value(n, i, m) + value(n, i, m+1));
617  xnew.push_back(xmid);
618  }
619  }
620  } else {
621  if (loglevel > 0) {
622  writelog("refine: discarding point at {}\n", d.grid(m));
623  }
624  }
625  }
626  dsize.push_back(znew.size() - nstart);
627  }
628 
629  // At this point, the new grid znew and the new solution vector xnew have
630  // been constructed, but the domains themselves have not yet been modified.
631  // Now update each domain with the new grid.
632 
633  size_t gridstart = 0, gridsize;
634  for (size_t n = 0; n < nDomains(); n++) {
635  Domain1D& d = domain(n);
636  gridsize = dsize[n];
637  if (gridsize != 0) {
638  d.setupGrid(gridsize, &znew[gridstart]);
639  }
640  gridstart += gridsize;
641  }
642 
643  // Replace the current solution vector with the new one
644  *m_state = xnew;
645  resize();
646  finalize();
647  return np;
648 }

650 int Sim1D::setFixedTemperature(double t)
651 {
652  int np = 0;
653  vector<double> znew, xnew;
654  double zfixed = 0.0;
655  double z1 = 0.0, z2 = 0.0;
656  vector<size_t> dsize;
657 
658  for (size_t n = 0; n < nDomains(); n++) {
659  Domain1D& d = domain(n);
660  size_t comp = d.nComponents();
661  size_t mfixed = npos;
662 
663  // loop over current grid to determine where new point is needed
664  StFlow* d_free = dynamic_cast<StFlow*>(&domain(n));
665  size_t npnow = d.nPoints();
666  size_t nstart = znew.size();
667  if (d_free && !d_free->fixed_mdot()) {
668  for (size_t m = 0; m < npnow - 1; m++) {
669  bool fixedpt = false;
670  double t1 = value(n, c_offset_T, m);
671  double t2 = value(n, c_offset_T, m + 1);
672  // threshold to avoid adding new point too close to existing point
673  double thresh = min(1., 1.e-1 * (t2 - t1));
674  z1 = d.grid(m);
675  z2 = d.grid(m + 1);
676  if (fabs(t - t1) <= thresh) {
677  zfixed = z1;
678  fixedpt = true;
679  } else if (fabs(t2 - t) <= thresh) {
680  zfixed = z2;
681  fixedpt = true;
682  } else if ((t1 < t) && (t < t2)) {
683  mfixed = m;
684  zfixed = (z1 - z2) / (t1 - t2) * (t - t2) + z2;
685  fixedpt = true;
686  }
687 
688  if (fixedpt) {
689  d_free->m_zfixed = zfixed;
690  d_free->m_tfixed = t;
691  break;
692  }
693  }
694  }
695 
696  // copy solution domain and push back values
697  for (size_t m = 0; m < npnow; m++) {
698  // add the current grid point to the new grid
699  znew.push_back(d.grid(m));
700 
701  // do the same for the solution at this point
702  for (size_t i = 0; i < comp; i++) {
703  xnew.push_back(value(n, i, m));
704  }
705  if (m == mfixed) {
706  // add new point at zfixed (mfixed is not npos)
707  znew.push_back(zfixed);
708  np++;
709  double interp_factor = (zfixed - z2) / (z1 - z2);
710  // for each component, linearly interpolate
711  // the solution to this point
712  for (size_t i = 0; i < comp; i++) {
713  double xmid = interp_factor*(value(n, i, m) - value(n, i, m+1)) + value(n,i,m+1);
714  xnew.push_back(xmid);
715  }
716  }
717  }
718  dsize.push_back(znew.size() - nstart);
719  }
720 
721  // At this point, the new grid znew and the new solution vector xnew have
722  // been constructed, but the domains themselves have not yet been modified.
723  // Now update each domain with the new grid.
724  size_t gridstart = 0;
725  for (size_t n = 0; n < nDomains(); n++) {
726  Domain1D& d = domain(n);
727  size_t gridsize = dsize[n];
728  d.setupGrid(gridsize, &znew[gridstart]);
729  gridstart += gridsize;
730  }
731 
732  // Replace the current solution vector with the new one
733  *m_state = xnew;
734 
735  resize();
736  finalize();
737  return np;
738 }
