58doublereal PDSS_HKFT::cp_mole() const
59{
60 doublereal pbar = m_pres * 1.0E-5;
61 doublereal c1term = m_c1;
62 doublereal c2term = m_c2 / (m_temp - 228.) / (m_temp - 228.);
63 doublereal a3term = -m_a3 / (m_temp - 228.) / (m_temp - 228.) / (m_temp - 228.) * 2.0 * m_temp * (pbar - m_presR_bar);
64 doublereal a4term = -m_a4 / (m_temp - 228.) / (m_temp - 228.) / (m_temp - 228.) * 2.0 * m_temp
65 * log((2600. + pbar)/(2600. + m_presR_bar));
66
67 doublereal omega_j;
68 doublereal domega_jdT;
69 doublereal d2omega_jdT2;
70 if (m_charge_j == 0.0) {
71 omega_j = m_omega_pr_tr;
72 domega_jdT = 0.0;
73 d2omega_jdT2 = 0.0;
74 } else {
75 doublereal nu = 166027;
76 doublereal r_e_j_pr_tr = m_charge_j * m_charge_j / (m_omega_pr_tr/nu + m_charge_j/3.082);
77 doublereal gval = gstar(m_temp, m_pres, 0);
78 doublereal dgvaldT = gstar(m_temp, m_pres, 1);
79 doublereal d2gvaldT2 = gstar(m_temp, m_pres, 2);
80
81 doublereal r_e_j = r_e_j_pr_tr + fabs(m_charge_j) * gval;
82 doublereal dr_e_jdT = fabs(m_charge_j) * dgvaldT;
83 doublereal d2r_e_jdT2 = fabs(m_charge_j) * d2gvaldT2;
84 doublereal r_e_j2 = r_e_j * r_e_j;
85
86 doublereal charge2 = m_charge_j * m_charge_j;
87 doublereal r_e_H = 3.082 + gval;
88 doublereal r_e_H2 = r_e_H * r_e_H;
89 omega_j = nu * (charge2 / r_e_j - m_charge_j / r_e_H);
90 domega_jdT = nu * (-(charge2 / r_e_j2 * dr_e_jdT)
91 +(m_charge_j / r_e_H2 * dgvaldT));
92 d2omega_jdT2 = nu * (2.0*charge2*dr_e_jdT*dr_e_jdT/(r_e_j2*r_e_j) - charge2*d2r_e_jdT2/r_e_j2
93 -2.0*m_charge_j*dgvaldT*dgvaldT/(r_e_H2*r_e_H) + m_charge_j*d2gvaldT2 /r_e_H2);
94 }
95
96 doublereal relepsilon = m_waterProps->relEpsilon(m_temp, m_pres, 0);
97 doublereal drelepsilondT = m_waterProps->relEpsilon(m_temp, m_pres, 1);
98 doublereal Y = drelepsilondT / (relepsilon * relepsilon);
99 doublereal d2relepsilondT2 = m_waterProps->relEpsilon(m_temp, m_pres, 2);
100
101 doublereal X = d2relepsilondT2 / (relepsilon* relepsilon) - 2.0 * relepsilon * Y * Y;
102 doublereal Z = -1.0 / relepsilon;
103
104 doublereal yterm = 2.0 * m_temp * Y * domega_jdT;
105 doublereal xterm = omega_j * m_temp * X;
106 doublereal otterm = m_temp * d2omega_jdT2 * (Z + 1.0);
107 doublereal rterm = - m_domega_jdT_prtr * (m_Z_pr_tr + 1.0);
108
109 doublereal Cp_calgmol = c1term + c2term + a3term + a4term + yterm + xterm + otterm + rterm;
110
111 return m_units.convertTo(Cp_calgmol, "J/kmol");
112}

114doublereal PDSS_HKFT::molarVolume() const
115{
116 // Initially do all calculations in (cal/gmol/Pa)
117
118 doublereal a1term = m_a1 * 1.0E-5;
119 doublereal a2term = m_a2 / (2600.E5 + m_pres);
120 doublereal a3term = m_a3 * 1.0E-5/ (m_temp - 228.);
121 doublereal a4term = m_a4 / (m_temp - 228.) / (2600.E5 + m_pres);
122
123 doublereal omega_j;
124 doublereal domega_jdP;
125 if (m_charge_j == 0.0) {
126 omega_j = m_omega_pr_tr;
127 domega_jdP = 0.0;
128 } else {
129 doublereal nu = 166027.;
130 doublereal charge2 = m_charge_j * m_charge_j;
131 doublereal r_e_j_pr_tr = charge2 / (m_omega_pr_tr/nu + m_charge_j/3.082);
132
133 doublereal gval = gstar(m_temp, m_pres, 0);
134 doublereal dgvaldP = gstar(m_temp, m_pres, 3);
135
136 doublereal r_e_j = r_e_j_pr_tr + fabs(m_charge_j) * gval;
137 doublereal r_e_H = 3.082 + gval;
138
139 omega_j = nu * (charge2 / r_e_j - m_charge_j / r_e_H);
140 doublereal dr_e_jdP = fabs(m_charge_j) * dgvaldP;
141 domega_jdP = - nu * (charge2 / (r_e_j * r_e_j) * dr_e_jdP)
142 + nu * m_charge_j / (r_e_H * r_e_H) * dgvaldP;
143 }
144
145 doublereal drelepsilondP = m_waterProps->relEpsilon(m_temp, m_pres, 3);
146 doublereal relepsilon = m_waterProps->relEpsilon(m_temp, m_pres, 0);
147 doublereal Q = drelepsilondP / (relepsilon * relepsilon);
148 doublereal Z = -1.0 / relepsilon;
149 doublereal wterm = - domega_jdP * (Z + 1.0);
150 doublereal qterm = - omega_j * Q;
151 doublereal molVol_calgmolPascal = a1term + a2term + a3term + a4term + wterm + qterm;
152
153 // Convert to m**3 / kmol from (cal/gmol/Pa)
154 return m_units.convertTo(molVol_calgmolPascal, "J/kmol");
155}

213void PDSS_HKFT::initThermo()
214{
215 PDSS::initThermo();
216 if (m_input.hasKey("h0")) {
217 m_deltaH_formation_tr_pr = m_input.convert("h0", "cal/gmol");
218 }
219 if (m_input.hasKey("g0")) {
220 m_deltaG_formation_tr_pr = m_input.convert("g0", "cal/gmol");
221 }
222 if (m_input.hasKey("s0")) {
223 m_Entrop_tr_pr = m_input.convert("s0", "cal/gmol/K");
224 }
225 auto& units = m_input.units();
226 if (m_input.hasKey("a")) {
227 auto& a = m_input["a"].asVector<AnyValue>(4);
228 m_a1 = units.convert(a[0], "cal/gmol/bar");
229 m_a2 = units.convert(a[1], "cal/gmol");
230 m_a3 = units.convert(a[2], "cal*K/gmol/bar");
231 m_a4 = units.convert(a[3], "cal*K/gmol");
232 }
233 if (m_input.hasKey("c")) {
234 auto& c = m_input["c"].asVector<AnyValue>(2);
235 m_c1 = units.convert(c[0], "cal/gmol/K");
236 m_c2 = units.convert(c[1], "cal*K/gmol");
237 }
238 if (m_input.hasKey("omega")) {
239 m_omega_pr_tr = m_input.convert("omega", "cal/gmol");
240 }
241
242 // Ok, if we are missing one, then we construct its value from the other two.
243 // This code has been internally verified.
244 m_charge_j = m_tp->charge(m_spindex);
245 if (std::isnan(m_deltaH_formation_tr_pr)) {
246 convertDGFormation();
247 double Hcalc = m_Mu0_tr_pr + 298.15 * m_units.convertTo(m_Entrop_tr_pr, "J/kmol");
248 m_deltaH_formation_tr_pr = m_units.convertFrom(Hcalc, "J/kmol");
249 } else if (std::isnan(m_deltaG_formation_tr_pr)) {
250 double DHjmol = m_units.convertTo(m_deltaH_formation_tr_pr, "J/kmol");
251 m_Mu0_tr_pr = DHjmol - 298.15 * m_units.convertTo(m_Entrop_tr_pr, "J/kmol");
252 m_deltaG_formation_tr_pr = m_units.convertFrom(m_Mu0_tr_pr, "J/kmol");
253 double tmp = m_Mu0_tr_pr;
254 convertDGFormation();
255 double totalSum = m_Mu0_tr_pr - tmp;
256 m_Mu0_tr_pr = tmp;
257 m_deltaG_formation_tr_pr = m_units.convertFrom(m_Mu0_tr_pr - totalSum, "J/kmol");
258 } else if (std::isnan(m_Entrop_tr_pr)) {
259 convertDGFormation();
260 double DHjmol = m_units.convertTo(m_deltaH_formation_tr_pr, "J/kmol");
261 m_Entrop_tr_pr = m_units.convertFrom((DHjmol - m_Mu0_tr_pr) / 298.15, "J/kmol");
262 }
263
264 m_waterSS = &dynamic_cast<PDSS_Water&>(*m_tp->providePDSS(0));
265
266 // Section to initialize m_Z_pr_tr and m_Y_pr_tr
267 m_temp = 273.15 + 25.;
268 m_pres = OneAtm;
269 doublereal relepsilon = m_waterProps->relEpsilon(m_temp, m_pres, 0);
270 m_waterSS->setState_TP(m_temp, m_pres);
271 m_densWaterSS = m_waterSS->density();
272 m_Z_pr_tr = -1.0 / relepsilon;
273 doublereal drelepsilondT = m_waterProps->relEpsilon(m_temp, m_pres, 1);
274 m_Y_pr_tr = drelepsilondT / (relepsilon * relepsilon);
275 m_waterProps = make_unique<WaterProps>(m_waterSS);
276 m_presR_bar = OneAtm / 1.0E5;
277 m_presR_bar = 1.0;
278 convertDGFormation();
279
280 // Ok, we have mu. Let's check it against the input value
281 // of DH_F to see that we have some internal consistency
282 double Hcalc = m_Mu0_tr_pr + 298.15 * m_units.convertTo(m_Entrop_tr_pr, "J/kmol");
283 double DHjmol = m_units.convertTo(m_deltaH_formation_tr_pr, "J/kmol");
284
285 // If the discrepancy is greater than 100 cal gmol-1, print
286 // an error and exit.
287 if (fabs(Hcalc -DHjmol) > m_units.convertTo(100, "J/kmol")) {
288 std::string sname = m_tp->speciesName(m_spindex);
289 if (s_InputInconsistencyErrorExit) {
290 throw CanteraError("PDSS_HKFT::initThermo", "For {}, DHjmol is"
291 " not consistent with G and S: {} vs {} J/kmol",
292 sname, Hcalc, m_units.convertTo(m_deltaH_formation_tr_pr, "J/kmol"));
293 } else {
294 warn_user("PDSS_HKFT::initThermo",
295 "DHjmol for {} is not consistent with G and S: calculated {} "
296 "vs input {} J/kmol; continuing with consistent DHjmol = {}",
297 sname, Hcalc, m_units.convertTo(m_deltaH_formation_tr_pr, "J/kmol"),
298 Hcalc);
299 m_deltaH_formation_tr_pr = m_units.convertFrom(Hcalc, "J/kmol");
300 }
301 }
302 doublereal nu = 166027;
303 doublereal r_e_j_pr_tr;
304 if (m_charge_j != 0.0) {
305 r_e_j_pr_tr = m_charge_j * m_charge_j / (m_omega_pr_tr/nu + m_charge_j/3.082);
306 } else {
307 r_e_j_pr_tr = 0.0;
308 }
309
310 if (m_charge_j == 0.0) {
311 m_domega_jdT_prtr = 0.0;
312 } else {
313 doublereal gval = gstar(m_temp, m_pres, 0);
314 doublereal dgvaldT = gstar(m_temp, m_pres, 1);
315 doublereal r_e_j = r_e_j_pr_tr + fabs(m_charge_j) * gval;
316 doublereal dr_e_jdT = fabs(m_charge_j) * dgvaldT;
317 m_domega_jdT_prtr = - nu * (m_charge_j * m_charge_j / (r_e_j * r_e_j) * dr_e_jdT)
318 + nu * m_charge_j / (3.082 + gval) / (3.082 + gval) * dgvaldT;
319 }
320}

372doublereal PDSS_HKFT::deltaH() const
373{
374 warn_deprecated("PDSS_HKFT::deltaH", "To be removed after Cantera 3.0");
375 doublereal pbar = m_pres * 1.0E-5;
376 doublereal c1term = m_c1 * (m_temp - 298.15);
377 doublereal a1term = m_a1 * (pbar - m_presR_bar);
378 doublereal a2term = m_a2 * log((2600. + pbar)/(2600. + m_presR_bar));
379 doublereal c2term = -m_c2 * (1.0/(m_temp - 228.) - 1.0/(298.15 - 228.));
380 double a3tmp = (2.0 * m_temp - 228.)/ (m_temp - 228.) /(m_temp - 228.);
381 doublereal a3term = m_a3 * a3tmp * (pbar - m_presR_bar);
382 doublereal a4term = m_a4 * a3tmp * log((2600. + pbar)/(2600. + m_presR_bar));
383 doublereal omega_j;
384 doublereal domega_jdT;
385 if (m_charge_j == 0.0) {
386 omega_j = m_omega_pr_tr;
387 domega_jdT = 0.0;
388 } else {
389 doublereal nu = 166027;
390 doublereal r_e_j_pr_tr = m_charge_j * m_charge_j / (m_omega_pr_tr/nu + m_charge_j/3.082);
391 doublereal gval = gstar(m_temp, m_pres, 0);
392 doublereal r_e_j = r_e_j_pr_tr + fabs(m_charge_j) * gval;
393 doublereal dgvaldT = gstar(m_temp, m_pres, 1);
394 doublereal dr_e_jdT = fabs(m_charge_j) * dgvaldT;
395 omega_j = nu * (m_charge_j * m_charge_j / r_e_j - m_charge_j / (3.082 + gval));
396 domega_jdT = - nu * (m_charge_j * m_charge_j / (r_e_j * r_e_j) * dr_e_jdT)
397 + nu * m_charge_j / (3.082 + gval) / (3.082 + gval) * dgvaldT;
398 }
399
400 doublereal relepsilon = m_waterProps->relEpsilon(m_temp, m_pres, 0);
401 doublereal drelepsilondT = m_waterProps->relEpsilon(m_temp, m_pres, 1);
402
403 doublereal Y = drelepsilondT / (relepsilon * relepsilon);
404 doublereal Z = -1.0 / relepsilon;
405
406 doublereal yterm = m_temp * omega_j * Y;
407 doublereal yrterm = - 298.15 * m_omega_pr_tr * m_Y_pr_tr;
408
409 doublereal wterm = - omega_j * (Z + 1.0);
410 doublereal wrterm = + m_omega_pr_tr * (m_Z_pr_tr + 1.0);
411
412 doublereal otterm = m_temp * domega_jdT * (Z + 1.0);
413 doublereal otrterm = - m_temp * m_domega_jdT_prtr * (m_Z_pr_tr + 1.0);
414
415 doublereal deltaH_calgmol = c1term + a1term + a2term + c2term + a3term + a4term
416 + yterm + yrterm + wterm + wrterm + otterm + otrterm;
417
418 return m_units.convertTo(deltaH_calgmol, "J/kmol");
419}

421doublereal PDSS_HKFT::deltaG() const
422{
423 doublereal pbar = m_pres * 1.0E-5;
424 doublereal sterm = - m_Entrop_tr_pr * (m_temp - 298.15);
425 doublereal c1term = -m_c1 * (m_temp * log(m_temp/298.15) - (m_temp - 298.15));
426 doublereal a1term = m_a1 * (pbar - m_presR_bar);
427 doublereal a2term = m_a2 * log((2600. + pbar)/(2600. + m_presR_bar));
428 doublereal c2term = -m_c2 * ((1.0/(m_temp - 228.) - 1.0/(298.15 - 228.)) * (228. - m_temp)/228.
429 - m_temp / (228.*228.) * log((298.15*(m_temp-228.)) / (m_temp*(298.15-228.))));
430 doublereal a3term = m_a3 / (m_temp - 228.) * (pbar - m_presR_bar);
431 doublereal a4term = m_a4 / (m_temp - 228.) * log((2600. + pbar)/(2600. + m_presR_bar));
432
433 doublereal omega_j;
434 if (m_charge_j == 0.0) {
435 omega_j = m_omega_pr_tr;
436 } else {
437 doublereal nu = 166027;
438 doublereal r_e_j_pr_tr = m_charge_j * m_charge_j / (m_omega_pr_tr/nu + m_charge_j/3.082);
439 doublereal gval = gstar(m_temp, m_pres, 0);
440 doublereal r_e_j = r_e_j_pr_tr + fabs(m_charge_j) * gval;
441 omega_j = nu * (m_charge_j * m_charge_j / r_e_j - m_charge_j / (3.082 + gval));
442 }
443
444 doublereal relepsilon = m_waterProps->relEpsilon(m_temp, m_pres, 0);
445 doublereal Z = -1.0 / relepsilon;
446 doublereal wterm = - omega_j * (Z + 1.0);
447 doublereal wrterm = m_omega_pr_tr * (m_Z_pr_tr + 1.0);
448 doublereal yterm = m_omega_pr_tr * m_Y_pr_tr * (m_temp - 298.15);
449 doublereal deltaG_calgmol = sterm + c1term + a1term + a2term + c2term + a3term + a4term + wterm + wrterm + yterm;
450
451 return m_units.convertTo(deltaG_calgmol, "J/kmol");
452}

454doublereal PDSS_HKFT::deltaS() const
455{
456 doublereal pbar = m_pres * 1.0E-5;
457
458 doublereal c1term = m_c1 * log(m_temp/298.15);
459 doublereal c2term = -m_c2 / 228. * ((1.0/(m_temp - 228.) - 1.0/(298.15 - 228.))
460 + 1.0 / 228. * log((298.15*(m_temp-228.)) / (m_temp*(298.15-228.))));
461 doublereal a3term = m_a3 / (m_temp - 228.) / (m_temp - 228.) * (pbar - m_presR_bar);
462 doublereal a4term = m_a4 / (m_temp - 228.) / (m_temp - 228.) * log((2600. + pbar)/(2600. + m_presR_bar));
463
464 doublereal omega_j;
465 doublereal domega_jdT;
466 if (m_charge_j == 0.0) {
467 omega_j = m_omega_pr_tr;
468 domega_jdT = 0.0;
469 } else {
470 doublereal nu = 166027;
471 doublereal r_e_j_pr_tr = m_charge_j * m_charge_j / (m_omega_pr_tr/nu + m_charge_j/3.082);
472 doublereal gval = gstar(m_temp, m_pres, 0);
473 doublereal dgvaldT = gstar(m_temp, m_pres, 1);
474 doublereal r_e_j = r_e_j_pr_tr + fabs(m_charge_j) * gval;
475 doublereal dr_e_jdT = fabs(m_charge_j) * dgvaldT;
476 omega_j = nu * (m_charge_j * m_charge_j / r_e_j - m_charge_j / (3.082 + gval));
477 domega_jdT = - nu * (m_charge_j * m_charge_j / (r_e_j * r_e_j) * dr_e_jdT)
478 + nu * m_charge_j / (3.082 + gval) / (3.082 + gval) * dgvaldT;
479 }
480
481 doublereal relepsilon = m_waterProps->relEpsilon(m_temp, m_pres, 0);
482 doublereal drelepsilondT = m_waterProps->relEpsilon(m_temp, m_pres, 1);
483 doublereal Y = drelepsilondT / (relepsilon * relepsilon);
484 doublereal Z = -1.0 / relepsilon;
485 doublereal wterm = omega_j * Y;
486 doublereal wrterm = - m_omega_pr_tr * m_Y_pr_tr;
487 doublereal otterm = domega_jdT * (Z + 1.0);
488 doublereal otrterm = - m_domega_jdT_prtr * (m_Z_pr_tr + 1.0);
489 doublereal deltaS_calgmol = c1term + c2term + a3term + a4term + wterm + wrterm + otterm + otrterm;
490
491 return m_units.convertTo(deltaS_calgmol, "J/kmol/K");
492}

494doublereal PDSS_HKFT::ag(const doublereal temp, const int ifunc) const
495{
496 static doublereal ag_coeff[3] = { -2.037662, 5.747000E-3, -6.557892E-6};
497 if (ifunc == 0) {
498 return ag_coeff[0] + ag_coeff[1] * temp + ag_coeff[2] * temp * temp;
499 } else if (ifunc == 1) {
500 return ag_coeff[1] + ag_coeff[2] * 2.0 * temp;
501 }
502 if (ifunc != 2) {
503 return 0.0;
504 }
505 return ag_coeff[2] * 2.0;
506}

508doublereal PDSS_HKFT::bg(const doublereal temp, const int ifunc) const
509{
510 static doublereal bg_coeff[3] = { 6.107361, -1.074377E-2, 1.268348E-5};
511 if (ifunc == 0) {
512 return bg_coeff[0] + bg_coeff[1] * temp + bg_coeff[2] * temp * temp;
513 } else if (ifunc == 1) {
514 return bg_coeff[1] + bg_coeff[2] * 2.0 * temp;
515 }
516 if (ifunc != 2) {
517 return 0.0;
518 }
519 return bg_coeff[2] * 2.0;

522doublereal PDSS_HKFT::f(const doublereal temp, const doublereal pres, const int ifunc) const
523{
524 static doublereal af_coeff[3] = { 3.666666E1, -0.1504956E-9, 0.5107997E-13};
525 doublereal TC = temp - 273.15;
526 doublereal presBar = pres / 1.0E5;
527
528 if (TC < 155.0) {
529 return 0.0;
530 }
531 TC = std::min(TC, 355.0);
532 if (presBar > 1000.) {
533 return 0.0;
534 }
535
536 doublereal T1 = (TC-155.0)/300.;
537 doublereal p2 = (1000. - presBar) * (1000. - presBar);
538 doublereal p3 = (1000. - presBar) * p2;
539 doublereal p4 = p2 * p2;
540 doublereal fac2 = af_coeff[1] * p3 + af_coeff[2] * p4;
541 if (ifunc == 0) {
542 return pow(T1,4.8) + af_coeff[0] * pow(T1, 16.0) * fac2;
543 } else if (ifunc == 1) {
544 return (4.8 * pow(T1,3.8) + 16.0 * af_coeff[0] * pow(T1, 15.0)) / 300. * fac2;
545 } else if (ifunc == 2) {
546 return (4.8 * 3.8 * pow(T1,2.8) + 16.0 * 15.0 * af_coeff[0] * pow(T1, 14.0)) / (300. * 300.) * fac2;
547 } else if (ifunc == 3) {
548 double fac1 = pow(T1,4.8) + af_coeff[0] * pow(T1, 16.0);
549 fac2 = - (3.0 * af_coeff[1] * p2 + 4.0 * af_coeff[2] * p3)/ 1.0E5;
550 return fac1 * fac2;
551 } else {
552 throw NotImplementedError("PDSS_HKFT::f");
553 }
554}

556doublereal PDSS_HKFT::g(const doublereal temp, const doublereal pres, const int ifunc) const
557{
558 doublereal afunc = ag(temp, 0);
559 doublereal bfunc = bg(temp, 0);
560 m_waterSS->setState_TP(temp, pres);
561 m_densWaterSS = m_waterSS->density();
562 // density in gm cm-3
563 doublereal dens = m_densWaterSS * 1.0E-3;
564 doublereal gval = afunc * pow((1.0-dens), bfunc);
565 if (dens >= 1.0) {
566 return 0.0;
567 }
568 if (ifunc == 0) {
569 return gval;
570 } else if (ifunc == 1 || ifunc == 2) {
571 doublereal afuncdT = ag(temp, 1);
572 doublereal bfuncdT = bg(temp, 1);
573 doublereal alpha = m_waterSS->thermalExpansionCoeff();
574
575 doublereal fac1 = afuncdT * gval / afunc;
576 doublereal fac2 = bfuncdT * gval * log(1.0 - dens);
577 doublereal fac3 = gval * alpha * bfunc * dens / (1.0 - dens);
578
579 doublereal dgdt = fac1 + fac2 + fac3;
580 if (ifunc == 1) {
581 return dgdt;
582 }
583
584 doublereal afuncdT2 = ag(temp, 2);
585 doublereal bfuncdT2 = bg(temp, 2);
586 doublereal dfac1dT = dgdt * afuncdT / afunc + afuncdT2 * gval / afunc
587 - afuncdT * afuncdT * gval / (afunc * afunc);
588 doublereal ddensdT = - alpha * dens;
589 doublereal dfac2dT = bfuncdT2 * gval * log(1.0 - dens)
590 + bfuncdT * dgdt * log(1.0 - dens)
591 - bfuncdT * gval /(1.0 - dens) * ddensdT;
592 doublereal dalphadT = m_waterSS->dthermalExpansionCoeffdT();
593 doublereal dfac3dT = dgdt * alpha * bfunc * dens / (1.0 - dens)
594 + gval * dalphadT * bfunc * dens / (1.0 - dens)
595 + gval * alpha * bfuncdT * dens / (1.0 - dens)
596 + gval * alpha * bfunc * ddensdT / (1.0 - dens)
597 + gval * alpha * bfunc * dens / ((1.0 - dens) * (1.0 - dens)) * ddensdT;
598
599 return dfac1dT + dfac2dT + dfac3dT;
600 } else if (ifunc == 3) {
601 doublereal beta = m_waterSS->isothermalCompressibility();
602 return - bfunc * gval * dens * beta / (1.0 - dens);
603 } else {
604 throw NotImplementedError("PDSS_HKFT::g");
605 }
606 return 0.0;
607}

609doublereal PDSS_HKFT::gstar(const doublereal temp, const doublereal pres, const int ifunc) const
610{
611 doublereal gval = g(temp, pres, ifunc);
612 doublereal fval = f(temp, pres, ifunc);
613 double res = gval - fval;
614 return res;

618{
619 size_t iE = m_tp->elementIndex(elemName);

631void PDSS_HKFT::convertDGFormation()
632{
633 // Ok let's get the element compositions and conversion factors.
634 doublereal totalSum = 0.0;
635 for (size_t m = 0; m < m_tp->nElements(); m++) {
636 double na = m_tp->nAtoms(m_spindex, m);
637 if (na > 0.0) {
638 totalSum += na * LookupGe(m_tp->elementName(m));
639 }
640 }
641 // Add in the charge
642 if (m_charge_j != 0.0) {
643 totalSum -= m_charge_j * LookupGe("H");
644 }
645 doublereal dg = m_units.convertTo(m_deltaG_formation_tr_pr, "J/kmol");
646 //! Store the result into an internal variable.
647 m_Mu0_tr_pr = dg + totalSum;
648}
