32 void GibbsExcessVPSSTP::calcDensity()
33 {
34  const vector<double>& vbar = getStandardVolumes();
35  double vtotal = 0.0;
36  for (size_t i = 0; i < m_kk; i++) {
37  vtotal += vbar[i] * moleFractions_[i];
38  }
39  double dd = meanMolecularWeight() / vtotal;
40  Phase::assignDensity(dd);
41 }

74 void GibbsExcessVPSSTP::getActivityCoefficients(double* const ac) const
75 {
76  getLnActivityCoefficients(ac);
77  for (size_t k = 0; k < m_kk; k++) {
78  if (ac[k] > 700.) {
79  ac[k] = exp(700.0);
80  } else if (ac[k] < -700.) {
81  ac[k] = exp(-700.0);
82  } else {
83  ac[k] = exp(ac[k]);
84  }
85  }
86 }

103 double GibbsExcessVPSSTP::checkMFSum(const double* const x) const
104 {
105  warn_deprecated("GibbsExcessVPSSTP::checkMFSum",
106  "Unused. To be removed after Cantera 3.0");
