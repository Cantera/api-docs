69 Reaction::Reaction(const AnyMap& node, const Kinetics& kin)
70 {
71  string rate_type = node.getString("type", "Arrhenius");
72  if (!kin.nPhases()) {
73  throw InputFileError("Reaction", node,
74  "Cannot instantiate Reaction with empty Kinetics object.");
75  }
76 
77  setParameters(node, kin);
78  size_t nDim = kin.thermo(kin.reactionPhaseIndex()).nDim();
79  if (!valid()) {
80  // If the reaction isn't valid (for example, contains undefined species),
81  // setting up the rate constant won't work
82  return;
83  }
84  if (nDim == 3) {
85  if (ba::starts_with(rate_type, "three-body-")) {
86  AnyMap rateNode = node;
87  rateNode["type"] = rate_type.substr(11, rate_type.size() - 11);
88  setRate(newReactionRate(rateNode, calculateRateCoeffUnits(kin)));
89  } else {
90  setRate(newReactionRate(node, calculateRateCoeffUnits(kin)));
91  }
92  } else {
93  AnyMap rateNode = node;
94  if (rateNode.hasKey("rate-constant")) {
95  if (!ba::starts_with(rate_type, "interface-")) {
96  rateNode["type"] = "interface-" + rate_type;
97  }
98  } else if (node.hasKey("sticking-coefficient")) {
99  if (!ba::starts_with(rate_type, "sticking-")) {
100  rateNode["type"] = "sticking-" + rate_type;
101  }
102  } else {
103  throw InputFileError("Reaction::Reaction", input,
104  "Unable to infer interface reaction type.");
105  }
106  setRate(newReactionRate(rateNode, calculateRateCoeffUnits(kin)));
107  }
108  check();
109 }

111 void Reaction::check()
112 {
113  if (!allow_nonreactant_orders) {
114  for (const auto& [name, order] : orders) {
115  if (reactants.find(name) == reactants.end()) {
116  throw InputFileError("Reaction::validate", input,
117  "Reaction order specified for non-reactant species '{}'", name);
118  }
119  }
120  }
121 
122  if (!allow_negative_orders) {
123  for (const auto& [name, order] : orders) {
124  if (order < 0.0) {
125  throw InputFileError("Reaction::validate", input,
126  "Negative reaction order specified for species '{}'", name);
127  }
128  }
129  }
130 
131  // If reaction orders are specified, then this reaction does not follow
132  // mass-action kinetics, and is not an elementary reaction. So check that it
133  // is not reversible, since computing the reverse rate from thermochemistry
134  // only works for elementary reactions.
135  if (reversible && !orders.empty()) {
136  throw InputFileError("Reaction::validate", input,
137  "Reaction orders may only be given for irreversible reactions");
138  }
139 
140  // Check reaction rate evaluator to ensure changes introduced after object
141  // instantiation are considered.
142  if (m_rate) {
143  m_rate->check(equation());
144  }
145 }

169 void Reaction::getParameters(AnyMap& reactionNode) const
170 {
171  if (!m_rate) {
172  throw CanteraError("Reaction::getParameters",
173  "Serialization of empty Reaction object is not supported.");
174  }
175 
176  reactionNode["equation"] = equation();
177 
178  if (duplicate) {
179  reactionNode["duplicate"] = true;
180  }
181  if (orders.size()) {
182  reactionNode["orders"] = orders;
183  }
184  if (allow_negative_orders) {
185  reactionNode["negative-orders"] = true;
186  }
187  if (allow_nonreactant_orders) {
188  reactionNode["nonreactant-orders"] = true;
189  }

192 
193  // strip information not needed for reconstruction
194  string rtype = reactionNode["type"].asString();
195  if (rtype == "pressure-dependent-Arrhenius") {
196  // skip
197  } else if (m_explicit_type && ba::ends_with(rtype, "Arrhenius")) {
198  // retain type information
199  if (m_third_body) {
200  reactionNode["type"] = "three-body";
201  } else {
202  reactionNode["type"] = "elementary";
203  }
204  } else if (ba::ends_with(rtype, "Arrhenius")) {
205  reactionNode.erase("type");
206  } else if (m_explicit_type) {
207  reactionNode["type"] = type();
208  } else if (ba::ends_with(rtype, "Blowers-Masel")) {
209  reactionNode["type"] = "Blowers-Masel";
210  }
211 
212  if (m_third_body) {
213  m_third_body->getParameters(reactionNode);

255 void Reaction::setRate(shared_ptr<ReactionRate> rate)
256 {
257  if (!rate) {
258  throw InputFileError("Reaction::setRate", input,
259  "Reaction rate for reaction '{}' must not be empty.", equation());
260  }
261  m_rate = rate;
262 
263  string rate_type = m_rate->type();
264  if (m_third_body) {
265  if (rate_type == "falloff" || rate_type == "chemically-activated") {
266  if (m_third_body->mass_action && !m_from_composition) {
267  throw InputFileError("Reaction::setRate", input,
268  "Third-body collider does not use '(+{})' notation.",
269  m_third_body->name());
270  }
271  m_third_body->mass_action = false;
272  } else if (rate_type == "Chebyshev") {
273  if (m_third_body->name() == "M") {
274  warn_deprecated("Chebyshev reaction equation", input, "Specifying 'M' "
275  "in the reaction equation for Chebyshev reactions is deprecated.");
276  m_third_body.reset();
277  }
278  } else if (rate_type == "pressure-dependent-Arrhenius") {
279  if (m_third_body->name() == "M") {
280  throw InputFileError("Reaction::setRate", input,
281  "Found superfluous '{}' in pressure-dependent-Arrhenius reaction.",
282  m_third_body->name());
283  }
284  }
285  } else {
286  if (rate_type == "falloff" || rate_type == "chemically-activated") {
287  if (!m_from_composition) {
288  throw InputFileError("Reaction::setRate", input,
289  "Reaction equation for falloff reaction '{}'\n does not "
290  "contain valid pressure-dependent third body", equation());
291  }
292  m_third_body = make_shared<ThirdBody>("(+M)");
293  }
294  }

342 void Reaction::setEquation(const string& equation, const Kinetics* kin)
343 {
344  parseReactionEquation(*this, equation, input, kin);
345 
346  string rate_type = input.getString("type", "");
347  if (ba::starts_with(rate_type, "three-body")) {
348  // state type when serializing
349  m_explicit_type = true;
350  } else if (rate_type == "elementary") {
351  // user override
352  m_explicit_type = true;
353  return;
354  } else if (kin && kin->thermo(kin->reactionPhaseIndex()).nDim() != 3) {
355  // interface reactions
356  return;
357  }
358 
359  string third_body;
360  size_t count = 0;
361  size_t countM = 0;
362  for (const auto& [name, stoich] : reactants) {
363  // detect explicitly specified collision partner
364  if (products.count(name)) {
365  third_body = name;
366  size_t generic = third_body == "M"
367  || third_body == "(+M)" || third_body == "(+ M)";
368  count++;
369  countM += generic;
370  if (stoich > 1 && products[third_body] > 1) {
371  count++;
372  countM += generic;
373  }
374  }
375  }
376 
377  if (count == 0) {
378  if (ba::starts_with(rate_type, "three-body")) {
379  throw InputFileError("Reaction::setEquation", input,
380  "Reactants for reaction '{}'\n"
381  "do not contain a valid third body collider", equation);
382  }
383  return;
384 
385  } else if (countM > 1) {
386  throw InputFileError("Reaction::setEquation", input,
387  "Multiple generic third body colliders 'M' are not supported", equation);
388 
389  } else if (count > 1) {
390  // equations with more than one explicit third-body collider are handled as a
391  // regular elementary reaction unless the equation contains a generic third body
392  if (countM) {
393  // generic collider 'M' is selected as third body
394  third_body = "M";
395  } else if (m_third_body) {
396  // third body is defined as explicit object
397  auto& effs = m_third_body->efficiencies;
398  if (effs.size() != 1 || !reactants.count(effs.begin()->first)) {
399  throw InputFileError("Reaction::setEquation", input,
400  "Detected ambiguous third body colliders in reaction '{}'\n"
401  "ThirdBody object needs to specify a single species", equation);
402  }
403  third_body = effs.begin()->first;
404  m_third_body->explicit_3rd = true;
405  } else if (input.hasKey("efficiencies")) {
406  // third body is implicitly defined by efficiency
407  auto effs = input["efficiencies"].asMap<double>();
408  if (effs.size() != 1 || !reactants.count(effs.begin()->first)) {
409  throw InputFileError("Reaction::setEquation", input,
410  "Detected ambiguous third body colliders in reaction '{}'\n"
411  "Collision efficiencies need to specify single species", equation);
412  }
413  third_body = effs.begin()->first;
414  m_third_body = make_shared<ThirdBody>(third_body);
415  m_third_body->explicit_3rd = true;
416  } else if (input.hasKey("default-efficiency")) {
417  // insufficient disambiguation of third bodies
418  throw InputFileError("Reaction::setEquation", input,
419  "Detected ambiguous third body colliders in reaction '{}'\n"
420  "Third-body definition requires specification of efficiencies",
421  equation);
422  } else if (ba::starts_with(rate_type, "three-body")) {
423  // no disambiguation of third bodies
424  throw InputFileError("Reaction::setEquation", input,
425  "Detected ambiguous third body colliders in reaction '{}'\n"
426  "A valid ThirdBody or collision efficiency definition is required",
427  equation);
428  } else {
429  return;
430  }
431 
432  } else if (third_body != "M" && !ba::starts_with(rate_type, "three-body")
433  && !ba::starts_with(third_body, "(+"))
434  {
435  // check for conditions of three-body reactions:
436  // - integer stoichiometric conditions
437  // - either reactant or product side involves exactly three species
438  size_t nreac = 0;
439  size_t nprod = 0;
440 
441  // ensure that all reactants have integer stoichiometric coefficients
442  for (const auto& [name, stoich] : reactants) {
443  if (trunc(stoich) != stoich) {
444  return;
445  }
446  nreac += static_cast<size_t>(stoich);
447  }
448 
449  // ensure that all products have integer stoichiometric coefficients
450  for (const auto& [name, stoich] : products) {
451  if (trunc(stoich) != stoich) {
452  return;
453  }
454  nprod += static_cast<size_t>(stoich);
455  }
456 
457  // either reactant or product side involves exactly three species
458  if (nreac != 3 && nprod != 3) {
459  return;
460  }
461  }
462 
463  if (m_third_body) {
464  string tName = m_third_body->name();
465  if (tName != third_body && third_body != "M" && tName != "M") {
466  throw InputFileError("Reaction::setEquation", input,
467  "Detected incompatible third body colliders in reaction '{}'\n"
468  "ThirdBody definition does not match equation", equation);
469  }
470  m_third_body->setName(third_body);
471  } else {
472  m_third_body = make_shared<ThirdBody>(third_body);
473  }
474 
475  // adjust reactant coefficients
476  auto reac = reactants.find(third_body);
477  if (trunc(reac->second) != 1) {
478  reac->second -= 1.;
479  } else {
480  reactants.erase(reac);
481  }
482 
483  // adjust product coefficients
484  auto prod = products.find(third_body);
485  if (trunc(prod->second) != 1) {
486  prod->second -= 1.;
487  } else {
488  products.erase(prod);
489  }
490 }

511 UnitStack Reaction::calculateRateCoeffUnits(const Kinetics& kin)
512 {
513  if (!valid()) {
514  // If a reaction is invalid because of missing species in the Kinetics
515  // object, determining the units of the rate coefficient is impossible.
516  return UnitStack({});
517  }
518 
519  // Determine the units of the rate coefficient
520  const auto& rxn_phase = kin.thermo(kin.reactionPhaseIndex());
521  UnitStack rate_units(rxn_phase.standardConcentrationUnits());
522 
523  // Set output units to standardConcentrationUnits per second
524  rate_units.join(1.);
525  rate_units.update(Units(1.0, 0, 0, -1), 1.);
526 
527  for (const auto& [name, order] : orders) {
528  const auto& phase = kin.speciesPhase(name);
529  // Account for specified reaction orders
530  rate_units.update(phase.standardConcentrationUnits(), -order);
531  }
532  for (const auto& [name, stoich] : reactants) {
533  // Order for each reactant is the reactant stoichiometric coefficient,
534  // unless already overridden by user-specified orders
535  if (name == "M" || ba::starts_with(name, "(+")) {
536  // calculateRateCoeffUnits may be called before these pseudo-species
537  // have been stripped from the reactants
538  continue;
539  } else if (orders.find(name) == orders.end()) {
540  const auto& phase = kin.speciesPhase(name);
541  // Account for each reactant species
542  rate_units.update(phase.standardConcentrationUnits(), -stoich);
543  }
544  }
545 
546  if (m_third_body && m_third_body->mass_action) {
547  // Account for third-body collision partner as the last entry
548  rate_units.join(-1);
549  }
550 
551  Reaction::rate_units = rate_units.product();
552  return rate_units;
553 }

565 void Reaction::checkBalance(const Kinetics& kin) const
566 {
567  Composition balr, balp;
568 
569  // iterate over products and reactants
570  for (const auto& [name, stoich] : products) {
571  const ThermoPhase& ph = kin.speciesPhase(name);
572  size_t k = ph.speciesIndex(name);
573  for (size_t m = 0; m < ph.nElements(); m++) {
574  balr[ph.elementName(m)] = 0.0; // so that balr contains all species
575  balp[ph.elementName(m)] += stoich * ph.nAtoms(k, m);
576  }
577  }
578  for (const auto& [name, stoich] : reactants) {
579  const ThermoPhase& ph = kin.speciesPhase(name);
580  size_t k = ph.speciesIndex(name);
581  for (size_t m = 0; m < ph.nElements(); m++) {
582  balr[ph.elementName(m)] += stoich * ph.nAtoms(k, m);
583  }
584  }
585 
586  string msg;
587  bool ok = true;
588  for (const auto& [elem, balance] : balr) {
589  double elemsum = balr[elem] + balp[elem];
590  double elemdiff = fabs(balp[elem] - balr[elem]);
591  if (elemsum > 0.0 && elemdiff / elemsum > 1e-4) {
592  ok = false;
593  msg += fmt::format(" {} {} {}\n",
594  elem, balr[elem], balp[elem]);
595  }
596  }
597  if (!ok) {
598  throw InputFileError("Reaction::checkBalance", input,
599  "The following reaction is unbalanced: {}\n"
600  " Element Reactants Products\n{}",
601  equation(), msg);
602  }
603 
604  if (kin.thermo(kin.reactionPhaseIndex()).nDim() == 3) {
605  return;
606  }
607 
608  // Check that the number of surface sites is balanced
609  double reac_sites = 0.0;
610  double prod_sites = 0.0;
611  auto& surf = dynamic_cast<const SurfPhase&>(kin.thermo(kin.reactionPhaseIndex()));
612  for (const auto& [name, stoich] : reactants) {
613  size_t k = surf.speciesIndex(name);
614  if (k != npos) {
615  reac_sites += stoich * surf.size(k);
616  }
617  }
618  for (const auto& [name, stoich] : products) {
619  size_t k = surf.speciesIndex(name);
620  if (k != npos) {
621  prod_sites += stoich * surf.size(k);
622  }
623  }
624  if (fabs(reac_sites - prod_sites) > 1e-5 * (reac_sites + prod_sites)) {
625  throw InputFileError("Reaction::checkBalance", input,
626  "Number of surface sites not balanced in reaction {}.\n"
627  "Reactant sites: {}\nProduct sites: {}",
628  equation(), reac_sites, prod_sites);
629  }
630 }

632 bool Reaction::checkSpecies(const Kinetics& kin) const
633 {
634  // Check for undeclared species
635  vector<string> undeclared;
636  updateUndeclared(undeclared, reactants, kin);
637  updateUndeclared(undeclared, products, kin);
638  if (!undeclared.empty()) {
639  if (kin.skipUndeclaredSpecies()) {
640  return false;
641  } else {
642  throw InputFileError("Reaction::checkSpecies", input, "Reaction '{}'\n"
643  "contains undeclared species: '{}'",
644  equation(), ba::join(undeclared, "', '"));
645  }
646  }
647 
648  undeclared.clear();
649  updateUndeclared(undeclared, orders, kin);
650  if (!undeclared.empty()) {
651  if (kin.skipUndeclaredSpecies()) {
652  return false;
653  } else {
654  if (input.hasKey("orders")) {
655  throw InputFileError("Reaction::checkSpecies", input["orders"],
656  "Reaction '{}'\n"
657  "defines reaction orders for undeclared species: '{}'",
658  equation(), ba::join(undeclared, "', '"));
659  }
660  // Error for empty input AnyMap (that is, XML)
661  throw InputFileError("Reaction::checkSpecies", input, "Reaction '{}'\n"
662  "defines reaction orders for undeclared species: '{}'",
663  equation(), ba::join(undeclared, "', '"));
664  }
665  }
666 
667  if (m_third_body) {
668  return m_third_body->checkSpecies(*this, kin);
669  }
670 
671  checkBalance(kin);
672 
673  return true;
674 }

676 bool Reaction::usesElectrochemistry(const Kinetics& kin) const
677 {
678  // Check electrochemistry
679  vector<double> e_counter(kin.nPhases(), 0.0);
680 
681  // Find the number of electrons in the products for each phase
682  for (const auto& [name, stoich] : products) {
683  size_t kkin = kin.kineticsSpeciesIndex(name);
684  size_t i = kin.speciesPhaseIndex(kkin);
685  size_t kphase = kin.thermo(i).speciesIndex(name);
686  e_counter[i] += stoich * kin.thermo(i).charge(kphase);
687  }
688 
689  // Subtract the number of electrons in the reactants for each phase
690  for (const auto& [name, stoich] : reactants) {
691  size_t kkin = kin.kineticsSpeciesIndex(name);
692  size_t i = kin.speciesPhaseIndex(kkin);
693  size_t kphase = kin.thermo(i).speciesIndex(name);
694  e_counter[i] -= stoich * kin.thermo(i).charge(kphase);
695  }
696 
697  // If the electrons change phases then the reaction is electrochemical
698  for (double delta_e : e_counter) {
699  if (std::abs(delta_e) > 1e-4) {
700  return true;
701  }
702  }
703 
704  return false;
705 }

906 void parseReactionEquation(Reaction& R, const string& equation,
907  const AnyBase& reactionNode, const Kinetics* kin)
908 {
909  // Parse the reaction equation to determine participating species and
910  // stoichiometric coefficients
911  vector<string> tokens;
912  tokenizeString(equation, tokens);
913  tokens.push_back("+"); // makes parsing last species not a special case
914 
915  size_t last_used = npos; // index of last-used token
916  bool reactants = true;
917  for (size_t i = 1; i < tokens.size(); i++) {
918  if (tokens[i] == "+" || ba::starts_with(tokens[i], "(+") ||
919  tokens[i] == "<=>" || tokens[i] == "=" || tokens[i] == "=>") {
920  string species = tokens[i-1];
921 
922  double stoich = 1.0;
923  bool mass_action = true;
924  if (last_used != npos && tokens[last_used] == "(+"
925  && ba::ends_with(species, ")")) {
926  // Falloff third body with space, such as "(+ M)"
927  mass_action = false;
928  species = "(+" + species;
929  } else if (last_used == i - 1 && ba::starts_with(species, "(+")
930  && ba::ends_with(species, ")")) {
931  // Falloff 3rd body written without space, such as "(+M)"
932  mass_action = false;
933  } else if (last_used == i - 2) {
934  // Species with no stoich. coefficient
935  } else if (last_used == i - 3) {
936  // Stoich. coefficient and species
937  try {
938  stoich = fpValueCheck(tokens[i-2]);
939  } catch (CanteraError& err) {
940  throw InputFileError("parseReactionEquation", reactionNode,
941  err.getMessage());
942  }
943  } else {
944  throw InputFileError("parseReactionEquation", reactionNode,
945  "Error parsing reaction string '{}'.\n"
946  "Current token: '{}'\nlast_used: '{}'",
947  equation, tokens[i],
948  (last_used == npos) ? "n/a" : tokens[last_used]);
949  }
950  if (!kin || (kin->kineticsSpeciesIndex(species) == npos
951  && mass_action && species != "M"))
952  {
953  R.setValid(false);
954  }
955 
956  if (reactants) {
957  R.reactants[species] += stoich;
958  } else {
959  R.products[species] += stoich;
960  }
961 
962  last_used = i;
963  }
964 
965  // Tokens after this point are part of the products string
966  if (tokens[i] == "<=>" || tokens[i] == "=") {
967  R.reversible = true;
968  reactants = false;
969  } else if (tokens[i] == "=>") {
970  R.reversible = false;
971  reactants = false;
972  }
973  }
974 }
