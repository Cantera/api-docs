31/*!
32 * @defgroup inputfiles Input File Handling
33 *
34 * The properties of phases and interfaces are specified in text files. These
35 * procedures handle various aspects of reading these files.
36 *
37 * For input files not specified by an absolute pathname, %Cantera searches
38 * for input files along a path that includes platform-specific default
39 * locations, and possibly user-specified locations.
40 *
41 * The current directory (".") is always searched first. Then, on Windows, the
42 * registry is checked to find the Cantera installation directory, and the
43 * 'data' subdirectory of the installation directory will be added to the search
44 * path.
45 *
46 * On any platform, if environment variable CANTERA_DATA is set to a directory
47 * name or a list of directory names separated with the OS-dependent path
48 * separator (that is, ";" on Windows, ":" elsewhere), then these directories will
49 * be added to the search path.
50 *
51 * Finally, the location where the data files were installed when
52 * %Cantera was built is added to the search path.
53 *
54 * Additional directories may be added by calling function addDirectory.
55 *
56 * %Cantera input files are written using YAML syntax. For more information on using
57 * YAML files in Cantera, see the
58 * <a href="https://cantera.org/tutorials/yaml/defining-phases.html">YAML Users' Guide</a>
59 * or the <a href="../../../../sphinx/html/yaml/index.html">YAML Input File API Reference</a>.
60 *
61 * %Cantera provides the `ck2yaml` tool for converting Chemkin-format mechanisms to the
62 * YAML format. The scripts `cti2yaml.py` and `ctml2yaml.py` can be used to convert
63 * legacy CTI and XML input files (from Cantera 2.6 and earlier) to the YAML format.
64 *
65 * @{
66 */

148//! Write a formatted message to the screen.
149//!
150//! This function passes its arguments to the fmt library 'format' function to
151//! generate a formatted string from a Python-style (curly braces) format
152//! string. This method is used throughout Cantera to write log messages. It can
153//! also be called by user programs. The advantage of using writelog over
154//! writing directly to the standard output is that messages written with
155//! writelog will display correctly even when Cantera is used from MATLAB or
156//! other application that do not have a standard output stream.
157//! @ingroup textlogs
158template <typename... Args>
159void writelog(const std::string& fmt, const Args&... args) {
160 if (sizeof...(args) == 0) {
161 writelog_direct(fmt);
162 } else {
163 writelog_direct(fmt::format(fmt, args...));
164 }
165}

167//! Write a formatted message to the screen
168/*!
169 * Using the printf formatting of C, write a message to the screen
170 * with variable values.
171 *
172 * Here, we format an internal string with the correct values
173 * and then feed it into writelog().
174 *
175 * @param fmt c format string for the following arguments
176 * @param args arguments used to interpolate the format string
177 * @ingroup textlogs
178 */
179template <typename... Args>
180void writelogf(const char* fmt, const Args& ... args) {
181 writelog_direct(fmt::sprintf(fmt, args...));
182}
