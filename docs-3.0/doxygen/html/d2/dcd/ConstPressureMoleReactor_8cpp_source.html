17void ConstPressureMoleReactor::getState(double* y)
18{
19 if (m_thermo == 0) {
20 throw CanteraError("ConstPressureMoleReactor::getState",
21 "Error: reactor is empty.");
22 }
23 m_thermo->restoreState(m_state);
24 // set mass to be used in getMoles function
25 m_mass = m_thermo->density() * m_vol;
26 // set the first array element to enthalpy
27 y[0] = m_thermo->enthalpy_mass() * m_thermo->density() * m_vol;
28 // get moles of species in remaining state
29 getMoles(y + m_sidx);
30 // set the remaining components to the surface species moles on the walls
31 getSurfaceInitialConditions(y+m_nsp+m_sidx);
32}

40void ConstPressureMoleReactor::updateState(double* y)
41{
42 // the components of y are: [0] the enthalpy, [1...K+1) are the
43 // moles of each species, and [K+1...] are the moles of surface
44 // species on each wall.
45 setMassFromMoles(y + m_sidx);
46 m_thermo->setMolesNoTruncate(y + m_sidx);
47 if (m_energy) {
48 m_thermo->setState_HP(y[0] / m_mass, m_pressure);
49 } else {
50 m_thermo->setPressure(m_pressure);
51 }
52 m_vol = m_mass / m_thermo->density();
53 updateConnected(false);
54 updateSurfaceState(y + m_nsp + m_sidx);
55}

57void ConstPressureMoleReactor::eval(double time, double* LHS, double* RHS)
58{
59 double* dndt = RHS + m_sidx; // kmol per s
60
61 evalWalls(time);
62
63 m_thermo->restoreState(m_state);
64
65 const vector_fp& imw = m_thermo->inverseMolecularWeights();
66
67 if (m_chem) {
68 m_kin->getNetProductionRates(&m_wdot[0]); // "omega dot"
69 }
70
71 // evaluate reactor surfaces
72 evalSurfaces(LHS + m_nsp + m_sidx, RHS + m_nsp + m_sidx, m_sdot.data());
73
74 // external heat transfer
75 double dHdt = m_Qdot;
76
77 for (size_t n = 0; n < m_nsp; n++) {
78 // production in gas phase and from surfaces
79 dndt[n] = m_wdot[n] * m_vol + m_sdot[n];
80 }
81
82 // add terms for outlets
83 for (auto outlet : m_outlet) {
84 // determine enthalpy contribution
85 dHdt -= outlet->massFlowRate() * m_enthalpy;
86 // flow of species into system and dilution by other species
87 for (size_t n = 0; n < m_nsp; n++) {
88 dndt[n] -= outlet->outletSpeciesMassFlowRate(n) * imw[n];
89 }
90 }
91
92 // add terms for inlets
93 for (auto inlet : m_inlet) {
94 // enthalpy contribution from inlets
95 dHdt += inlet->enthalpy_mass() * inlet->massFlowRate();
96 // flow of species into system and dilution by other species
97 for (size_t n = 0; n < m_nsp; n++) {
98 dndt[n] += inlet->outletSpeciesMassFlowRate(n) * imw[n];
99 }
100 }
101
102 if (m_energy) {
103 RHS[0] = dHdt;
104 } else {
105 RHS[0] = 0.0;
106 }
107}

121std::string ConstPressureMoleReactor::componentName(size_t k) {
122 if (k == 0) {
123 return "enthalpy";
124 } else if (k >= m_sidx && k < neq()) {
125 k -= m_sidx;
126 if (k < m_thermo->nSpecies()) {
127 return m_thermo->speciesName(k);
128 } else {
129 k -= m_thermo->nSpecies();
130 }
131 for (auto& S : m_surfaces) {
132 ThermoPhase* th = S->thermo();
133 if (k < th->nSpecies()) {
134 return th->speciesName(k);
135 } else {
136 k -= th->nSpecies();
137 }
138 }
139 }
140 throw CanteraError("ConstPressureMoleReactor::componentName",
141 "Index is out of bounds.");
142}
