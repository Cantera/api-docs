58 void PDSS_IdealGas::setTemperature(double temp)
59 {
60  m_temp = temp;
61  m_spthermo->updatePropertiesTemp(temp, &m_cp0_R, &m_h0_RT, &m_s0_R);
62  m_g0_RT = m_h0_RT - m_s0_R;
63  m_V0 = GasConstant * m_temp / m_p0;
64  m_hss_RT = m_h0_RT;
65  m_cpss_R = m_cp0_R;
66  m_sss_R = m_s0_R - log(m_pres/m_p0);
67  m_gss_RT = m_hss_RT - m_sss_R;
68  m_Vss = GasConstant * m_temp / m_pres;
