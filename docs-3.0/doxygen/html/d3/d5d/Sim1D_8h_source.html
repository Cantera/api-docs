144 /**
145 * Save current simulation data to a container file or CSV format.
146 *
147 * In order to save the content of a Sim1D object, individual domains are
148 * converted to SolutionArray objects and saved using the SolutionArray::save()
149 * method. For HDF and YAML output, all domains are written to a single container
150 * file with shared header information. Simulation settings of individual domains
151 * are preserved as meta data of the corresponding SolutionArray objects.
152 * For CSV files, only state and auxiliary data of the main 1D domain are saved.
153 *
154 * The complete state of the current object can be restored from HDF and YAML
155 * container files using the restore() method, while individual domains can be
156 * loaded using SolutionArray::restore() for further analysis. While CSV do not
157 * contain complete information, they can still be used for setting initial states
158 * of individual simulation objects for some Cantera API's.
159 *
160 * @param fname Name of output file (CSV, YAML or HDF)
161 * @param name Identifier of storage location within the container file; this
162 * node/group contains header information and multiple subgroups holding
163 * domain-specific SolutionArray data (YAML/HDF only)
164 * @param desc Custom comment describing the dataset to be stored (YAML/HDF only)
165 * @param overwrite Force overwrite if file/name exists; optional (default=false)
166 * @param compression Compression level (0-9); optional (default=0; HDF only)
167 * @param basis Output mass ("Y"/"mass") or mole ("X"/"mole") fractions;
168 * if not specified (default=""), the native basis of the underlying
169 * ThermoPhase manager is used - @see nativeState (CSV only)
170 */
171 void save(const std::string& fname, const std::string& name,
172 const std::string& desc, bool overwrite=false, int compression=0,
173 const string& basis="");

323 //! Solve the equation \f$ J^T \lambda = b \f$.
324 /**
325 * Here, \f$ J = \partial f/\partial x \f$ is the Jacobian matrix of the
326 * system of equations \f$ f(x,p)=0 \f$. This can be used to efficiently
327 * solve for the sensitivities of a scalar objective function \f$ g(x,p) \f$
328 * to a vector of parameters \f$ p \f$ by solving:
329 * \f[ J^T \lambda = \left( \frac{\partial g}{\partial x} \right)^T \f]
330 * for \f$ \lambda \f$ and then computing:
331 * \f[
332 * \left.\frac{dg}{dp}\right|_{f=0} = \frac{\partial g}{\partial p}
333 * - \lambda^T \frac{\partial f}{\partial p}
334 * \f]
335 */
336 void solveAdjoint(const double* b, double* lambda);
