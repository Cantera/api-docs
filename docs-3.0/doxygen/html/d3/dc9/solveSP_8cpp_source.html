22solveSP::solveSP(ImplicitSurfChem* surfChemPtr, int bulkFunc) :
23 m_objects(surfChemPtr->getObjects()),
24 m_bulkFunc(bulkFunc)
25{
26 for (size_t n = 0; n < m_objects.size(); n++) {
27 InterfaceKinetics* kin = m_objects[n];
28 size_t surfPhaseIndex = kin->reactionPhaseIndex();
29 SurfPhase* sp = dynamic_cast<SurfPhase*>(&kin->thermo(surfPhaseIndex));
30 if (sp == nullptr) {
31 throw CanteraError("solveSP::solveSP",
32 "InterfaceKinetics object has no surface phase");
33 }
34
35 m_numSurfPhases++;
36 m_indexKinObjSurfPhase.push_back(n);
37 m_kinObjPhaseIDSurfPhase.push_back(surfPhaseIndex);
38
39 m_ptrsSurfPhase.push_back(sp);
40 size_t nsp = sp->nSpecies();
41 m_nSpeciesSurfPhase.push_back(nsp);
42 m_numTotSurfSpecies += nsp;
43 }
44
45 if (bulkFunc == BULK_DEPOSITION) {
46 m_neq = m_numTotSurfSpecies + m_numTotBulkSpeciesSS;
47 } else {
48 m_neq = m_numTotSurfSpecies;
49 }
50
51 for (size_t n = 0; n < m_numSurfPhases; n++) {
52 size_t tsp = m_objects[n]->nTotalSpecies();
53 m_maxTotSpecies = std::max(m_maxTotSpecies, tsp);
54 }
55 m_maxTotSpecies = std::max(m_maxTotSpecies, m_neq);
56
57 m_netProductionRatesSave.resize(m_maxTotSpecies, 0.0);
58 m_numEqn1.resize(m_maxTotSpecies, 0.0);
59 m_numEqn2.resize(m_maxTotSpecies, 0.0);
60 m_XMolKinSpecies.resize(m_maxTotSpecies, 0.0);
61 m_CSolnSave.resize(m_neq, 0.0);
62 m_spSurfLarge.resize(m_numSurfPhases, 0);
63 m_kinSpecIndex.resize(m_numTotSurfSpecies + m_numTotBulkSpeciesSS, 0);
64 m_kinObjIndex.resize(m_numTotSurfSpecies + m_numTotBulkSpeciesSS, 0);
65 m_eqnIndexStartSolnPhase.resize(m_numSurfPhases + m_numBulkPhasesSS, 0);
66
67 size_t kindexSP = 0;
68 for (size_t isp = 0; isp < m_numSurfPhases; isp++) {
69 size_t iKinObject = m_indexKinObjSurfPhase[isp];
70 InterfaceKinetics* kin = m_objects[iKinObject];
71 size_t surfPhaseIndex = m_kinObjPhaseIDSurfPhase[isp];
72 size_t kstart = kin->kineticsSpeciesIndex(0, surfPhaseIndex);
73 size_t nsp = m_nSpeciesSurfPhase[isp];
74 m_eqnIndexStartSolnPhase[isp] = kindexSP;
75 for (size_t k = 0; k < nsp; k++, kindexSP++) {
76 m_kinSpecIndex[kindexSP] = kstart + k;
77 m_kinObjIndex[kindexSP] = isp;
78 }
79 }
80
81 // Dimension solution vector
82 size_t dim1 = std::max<size_t>(1, m_neq);
83 m_CSolnSP.resize(dim1, 0.0);
84 m_CSolnSPInit.resize(dim1, 0.0);
85 m_CSolnSPOld.resize(dim1, 0.0);
86 m_wtResid.resize(dim1, 0.0);
87 m_wtSpecies.resize(dim1, 0.0);
88 m_resid.resize(dim1, 0.0);
89 m_Jac.resize(dim1, dim1, 0.0);
90}

92int solveSP::solveSurfProb(int ifunc, doublereal time_scale, doublereal TKelvin,
93 doublereal PGas, doublereal reltol, doublereal abstol)
94{
95 doublereal EXTRA_ACCURACY = 0.001;
96 if (ifunc == SFLUX_JACOBIAN) {
97 EXTRA_ACCURACY *= 0.001;
98 }
99 int label_t=-1; // Species IDs for time control
100 int label_d = -1; // Species IDs for damping control
101 int label_t_old=-1;
102 doublereal label_factor = 1.0;
103 int iter=0; // iteration number on nonlinear solver
104 int iter_max=1000; // maximum number of nonlinear iterations
105 doublereal deltaT = 1.0E-10; // Delta time step
106 doublereal damp=1.0;
107 doublereal inv_t = 0.0;
108 doublereal t_real = 0.0, update_norm = 1.0E6;
109 bool do_time = false, not_converged = true;
110 m_ioflag = std::min(m_ioflag, 1);
111
112 // Set the initial value of the do_time parameter
113 if (ifunc == SFLUX_INITIALIZE || ifunc == SFLUX_TRANSIENT) {
114 do_time = true;
115 }
116
117 // Store the initial guess for the surface problem in the soln vector,
118 // CSoln, and in an separate vector CSolnInit.
119 size_t loc = 0;
120 for (size_t n = 0; n < m_numSurfPhases; n++) {
121 m_ptrsSurfPhase[n]->getConcentrations(m_numEqn1.data());
122 for (size_t k = 0; k < m_nSpeciesSurfPhase[n]; k++) {
123 m_CSolnSP[loc] = m_numEqn1[k];
124 loc++;
125 }
126 }
127
128 m_CSolnSPInit = m_CSolnSP;
129
130 // Calculate the largest species in each phase
131 evalSurfLarge(m_CSolnSP.data());
132
133 if (m_ioflag) {
134 print_header(m_ioflag, ifunc, time_scale, true, reltol, abstol);
135 }
136
137 // Quick return when there isn't a surface problem to solve
138 if (m_neq == 0) {
139 not_converged = false;
140 update_norm = 0.0;
141 }
142
143 // Start of Newton's method
144 while (not_converged && iter < iter_max) {
145 iter++;
146 // Store previous iteration's solution in the old solution vector
147 m_CSolnSPOld = m_CSolnSP;
148
149 // Evaluate the largest surface species for each surface phase every
150 // 5 iterations.
151 if (iter%5 == 4) {
152 evalSurfLarge(m_CSolnSP.data());
153 }
154
155 // Calculate the value of the time step
156 // - heuristics to stop large oscillations in deltaT
157 if (do_time) {
158 // don't hurry increase in time step at the same time as damping
159 if (damp < 1.0) {
160 label_factor = 1.0;
161 }
162 double tmp = calc_t(m_netProductionRatesSave.data(),
163 m_XMolKinSpecies.data(),
164 &label_t, &label_t_old, &label_factor, m_ioflag);
165 if (iter < 10) {
166 inv_t = tmp;
167 } else if (tmp > 2.0*inv_t) {
168 inv_t = 2.0*inv_t;
169 } else {
170 inv_t = tmp;
171 }
172
173 // Check end condition
174 if (ifunc == SFLUX_TRANSIENT) {
175 tmp = t_real + 1.0/inv_t;
176 if (tmp > time_scale) {
177 inv_t = 1.0/(time_scale - t_real);
178 }
179 }
180 } else {
181 // make steady state calc a step of 1 million seconds to prevent
182 // singular Jacobians for some pathological cases
183 inv_t = 1.0e-6;
184 }
185 deltaT = 1.0/inv_t;
186
187 // Call the routine to numerically evaluation the Jacobian and residual
188 // for the current iteration.
189 resjac_eval(m_Jac, m_resid.data(), m_CSolnSP.data(),
190 m_CSolnSPOld.data(), do_time, deltaT);
191
192 // Calculate the weights. Make sure the calculation is carried out on
193 // the first iteration.
194 if (iter%4 == 1) {
195 calcWeights(m_wtSpecies.data(), m_wtResid.data(),
196 m_Jac, m_CSolnSP.data(), abstol, reltol);
197 }
198
199 // Find the weighted norm of the residual
200 double resid_norm = calcWeightedNorm(m_wtResid.data(), m_resid.data(), m_neq);
201
202 // Solve Linear system. The solution is in m_resid
203 solve(m_Jac, m_resid.data());
204
205 // Calculate the Damping factor needed to keep all unknowns between 0
206 // and 1, and not allow too large a change (factor of 2) in any unknown.
207 damp = calc_damping(m_CSolnSP.data(), m_resid.data(), m_neq, &label_d);
208
209 // Calculate the weighted norm of the update vector Here, resid is the
210 // delta of the solution, in concentration units.
211 update_norm = calcWeightedNorm(m_wtSpecies.data(),
212 m_resid.data(), m_neq);
213
214 // Update the solution vector and real time Crop the concentrations to
215 // zero.
216 for (size_t irow = 0; irow < m_neq; irow++) {
217 m_CSolnSP[irow] -= damp * m_resid[irow];
218 }
219 for (size_t irow = 0; irow < m_neq; irow++) {
220 m_CSolnSP[irow] = std::max(0.0, m_CSolnSP[irow]);
221 }
222 updateState(m_CSolnSP.data());
223
224 if (do_time) {
225 t_real += damp/inv_t;
226 }
227
228 if (m_ioflag) {
229 printIteration(m_ioflag, damp, label_d, label_t, inv_t, t_real, iter,
230 update_norm, resid_norm, do_time);
231 }
232
233 if (ifunc == SFLUX_TRANSIENT) {
234 not_converged = (t_real < time_scale);
235 } else {
236 if (do_time) {
237 if (t_real > time_scale ||
238 (resid_norm < 1.0e-7 &&
239 update_norm*time_scale/t_real < EXTRA_ACCURACY)) {
240 do_time = false;
241 }
242 } else {
243 not_converged = ((update_norm > EXTRA_ACCURACY) ||
244 (resid_norm > EXTRA_ACCURACY));
245 }
246 }
247 } // End of Newton's Method while statement
248
249 // End Newton's method. If not converged, print error message and
250 // recalculate sdot's at equal site fractions.
251 if (not_converged && m_ioflag) {
252 writelog("#$#$#$# Error in solveSP $#$#$#$ \n");
253 writelogf("Newton iter on surface species did not converge, "
254 "update_norm = %e \n", update_norm);
255 writelog("Continuing anyway\n");
256 }
257
258 // Decide on what to return in the solution vector. Right now, will always
259 // return the last solution no matter how bad
260 if (m_ioflag) {
261 fun_eval(m_resid.data(), m_CSolnSP.data(), m_CSolnSPOld.data(),
262 false, deltaT);
263 double resid_norm = calcWeightedNorm(m_wtResid.data(), m_resid.data(), m_neq);
264 printIteration(m_ioflag, damp, label_d, label_t, inv_t, t_real, iter,
265 update_norm, resid_norm, do_time, true);
266 }
267
268 // Return with the appropriate flag
269 if (update_norm > 1.0) {
270 return -1;
271 }
272 return 1;
273}

275void solveSP::updateState(const doublereal* CSolnSP)
276{
277 vector_fp X;
278 size_t loc = 0;
279 for (size_t n = 0; n < m_numSurfPhases; n++) {
280 X.resize(m_nSpeciesSurfPhase[n]);
281 for (size_t k = 0; k < X.size(); k++) {
282 X[k] = CSolnSP[loc + k] / m_ptrsSurfPhase[n]->siteDensity();
283 }
284 m_ptrsSurfPhase[n]->setMoleFractions_NoNorm(X.data());
285 loc += m_nSpeciesSurfPhase[n];
286 }
287}

306void solveSP::evalSurfLarge(const doublereal* CSolnSP)
307{
308 size_t kindexSP = 0;
309 for (size_t isp = 0; isp < m_numSurfPhases; isp++) {
310 doublereal Clarge = CSolnSP[kindexSP];

312 kindexSP++;

322void solveSP::fun_eval(doublereal* resid, const doublereal* CSoln,
323 const doublereal* CSolnOld, const bool do_time,
324 const doublereal deltaT)
325{
326 size_t k;
327 doublereal lenScale = 1.0E-9;
328 if (m_numSurfPhases > 0) {
329 // update the surface concentrations with the input surface
330 // concentration vector
331 updateState(CSoln);
332
333 // Get the net production rates of all of the species in the
334 // surface kinetics mechanism
335 //
336 // HKM Should do it here for all kinetics objects so that
337 // bulk will eventually work.
338 if (do_time) {
339 size_t kindexSP = 0;
340 for (size_t isp = 0; isp < m_numSurfPhases; isp++) {
341 size_t nsp = m_nSpeciesSurfPhase[isp];
342 InterfaceKinetics* kinPtr = m_objects[isp];
343 size_t surfIndex = kinPtr->reactionPhaseIndex();
344 size_t kstart = kinPtr->kineticsSpeciesIndex(0, surfIndex);
345 size_t kins = kindexSP;
346 kinPtr->getNetProductionRates(m_netProductionRatesSave.data());
347 for (k = 0; k < nsp; k++, kindexSP++) {
348 resid[kindexSP] =
349 (CSoln[kindexSP] - CSolnOld[kindexSP]) / deltaT
350 - m_netProductionRatesSave[kstart + k];
351 }
352
353 size_t kspecial = kins + m_spSurfLarge[isp];
354 double sd = m_ptrsSurfPhase[isp]->siteDensity();
355 resid[kspecial] = sd;
356 for (k = 0; k < nsp; k++) {
357 resid[kspecial] -= CSoln[kins + k];
358 }
359 }
360 } else {
361 size_t kindexSP = 0;
362 for (size_t isp = 0; isp < m_numSurfPhases; isp++) {
363 size_t nsp = m_nSpeciesSurfPhase[isp];
364 InterfaceKinetics* kinPtr = m_objects[isp];
365 size_t surfIndex = kinPtr->reactionPhaseIndex();
366 size_t kstart = kinPtr->kineticsSpeciesIndex(0, surfIndex);
367 size_t kins = kindexSP;
368 kinPtr->getNetProductionRates(m_netProductionRatesSave.data());
369 for (k = 0; k < nsp; k++, kindexSP++) {
370 resid[kindexSP] = - m_netProductionRatesSave[kstart + k];
371 }
372 size_t kspecial = kins + m_spSurfLarge[isp];
373 double sd = m_ptrsSurfPhase[isp]->siteDensity();
374 resid[kspecial] = sd;
375 for (k = 0; k < nsp; k++) {
376 resid[kspecial] -= CSoln[kins + k];
377 }
378 }
379 }
380
381 if (m_bulkFunc == BULK_DEPOSITION) {
382 size_t kindexSP = m_numTotSurfSpecies;
383 for (size_t isp = 0; isp < m_numBulkPhasesSS; isp++) {
384 doublereal* XBlk = m_numEqn1.data();
385 size_t nsp = m_nSpeciesSurfPhase[isp];
386 size_t surfPhaseIndex = m_indexKinObjSurfPhase[isp];
387 InterfaceKinetics* kin = m_objects[isp];
388 double grRate = 0.0;
389 size_t kstart = kin->kineticsSpeciesIndex(0, surfPhaseIndex);
390 for (k = 0; k < nsp; k++) {
391 if (m_netProductionRatesSave[kstart + k] > 0.0) {
392 grRate += m_netProductionRatesSave[kstart + k];
393 }
394 }
395 resid[kindexSP] = m_bulkPhasePtrs[isp]->molarDensity();
396 for (k = 0; k < nsp; k++) {
397 resid[kindexSP] -= CSoln[kindexSP + k];
398 }
399 if (grRate > 0.0) {
400 for (k = 1; k < nsp; k++) {
401 if (m_netProductionRatesSave[kstart + k] > 0.0) {
402 resid[kindexSP + k] = XBlk[k] * grRate
403 - m_netProductionRatesSave[kstart + k];
404 } else {
405 resid[kindexSP + k] = XBlk[k] * grRate;
406 }
407 }
408 } else {
409 grRate = 1.0E-6;
410 //! @todo the appearance of k in this formula is suspicious
411 grRate += fabs(m_netProductionRatesSave[kstart + k]);
412 for (k = 1; k < nsp; k++) {
413 resid[kindexSP + k] = grRate * (XBlk[k] - 1.0/nsp);
414 }
415 }
416 if (do_time) {
417 for (k = 1; k < nsp; k++) {
418 resid[kindexSP + k] +=
419 lenScale / deltaT *
420 (CSoln[kindexSP + k]- CSolnOld[kindexSP + k]);
421 }
422 }
423 kindexSP += nsp;
424 }
425 }
426 }
427}

429void solveSP::resjac_eval(DenseMatrix& jac,
430 doublereal resid[], doublereal CSoln[],
431 const doublereal CSolnOld[], const bool do_time,
432 const doublereal deltaT)
433{
434 size_t kColIndex = 0;
435 // Calculate the residual
436 fun_eval(resid, CSoln, CSolnOld, do_time, deltaT);
437 // Now we will look over the columns perturbing each unknown.
438 for (size_t jsp = 0; jsp < m_numSurfPhases; jsp++) {
439 size_t nsp = m_nSpeciesSurfPhase[jsp];
440 double sd = m_ptrsSurfPhase[jsp]->siteDensity();
441 for (size_t kCol = 0; kCol < nsp; kCol++) {
442 double cSave = CSoln[kColIndex];
443 double dc = std::max(1.0E-10 * sd, fabs(cSave) * 1.0E-7);
444 CSoln[kColIndex] += dc;
445 fun_eval(m_numEqn2.data(), CSoln, CSolnOld, do_time, deltaT);
446 for (size_t i = 0; i < m_neq; i++) {
447 jac(i, kColIndex) = (m_numEqn2[i] - resid[i])/dc;
448 }
449 CSoln[kColIndex] = cSave;
450 kColIndex++;
451 }
452 }
453
454 if (m_bulkFunc == BULK_DEPOSITION) {
455 for (size_t jsp = 0; jsp < m_numBulkPhasesSS; jsp++) {
456 size_t nsp = m_numBulkSpecies[jsp];
457 double sd = m_bulkPhasePtrs[jsp]->molarDensity();
458 for (size_t kCol = 0; kCol < nsp; kCol++) {
459 double cSave = CSoln[kColIndex];
460 double dc = std::max(1.0E-10 * sd, fabs(cSave) * 1.0E-7);
461 CSoln[kColIndex] += dc;
462 fun_eval(m_numEqn2.data(), CSoln, CSolnOld, do_time, deltaT);
463 for (size_t i = 0; i < m_neq; i++) {
464 jac(i, kColIndex) = (m_numEqn2[i] - resid[i])/dc;
465 }
466 CSoln[kColIndex] = cSave;
467 kColIndex++;
468 }
469 }
470 }
471}

473/*!
474 * This function calculates a damping factor for the Newton iteration update
475 * vector, dxneg, to insure that all site and bulk fractions, x, remain
476 * bounded between zero and one.
477 *
478 * dxneg[] = negative of the update vector.
479 *
480 * The constant "APPROACH" sets the fraction of the distance to the boundary
481 * that the step can take. If the full step would not force any fraction
482 * outside of 0-1, then Newton's method is allowed to operate normally.
483 */
484static doublereal calc_damping(doublereal x[], doublereal dxneg[], size_t dim, int* label)
485{
486 const doublereal APPROACH = 0.80;
487 doublereal damp = 1.0;
488 static doublereal damp_old = 1.0; //! @todo this variable breaks thread safety
489 *label = -1;
490
491 for (size_t i = 0; i < dim; i++) {
492 // Calculate the new suggested new value of x[i]
493 double xnew = x[i] - damp * dxneg[i];
494
495 // Calculate the allowed maximum and minimum values of x[i]
496 // - Only going to allow x[i] to converge to zero by a
497 // single order of magnitude at a time
498 double xtop = 1.0 - 0.1*fabs(1.0-x[i]);
499 double xbot = fabs(x[i]*0.1) - 1.0e-16;
500 if (xnew > xtop) {
501 damp = - APPROACH * (1.0 - x[i]) / dxneg[i];
502 *label = int(i);
503 } else if (xnew < xbot) {
504 damp = APPROACH * x[i] / dxneg[i];
505 *label = int(i);
506 } else if (xnew > 3.0*std::max(x[i], 1.0E-10)) {
507 damp = - 2.0 * std::max(x[i], 1.0E-10) / dxneg[i];
508 *label = int(i);
509 }
510 }
511 damp = std::max(damp, 1e-2);
512
513 // Only allow the damping parameter to increase by a factor of three each
514 // iteration. Heuristic to avoid oscillations in the value of damp
515 if (damp > damp_old*3) {
516 damp = damp_old*3;
517 *label = -1;
518 }
519
520 // Save old value of the damping parameter for use in subsequent calls.
521 damp_old = damp;
522 return damp;
523
524} /* calc_damping */

526/*
527 * This function calculates the norm of an update, dx[], based on the
528 * weighted values of x.
529 */
530static doublereal calcWeightedNorm(const doublereal wtX[], const doublereal dx[], size_t dim)
531{
532 doublereal norm = 0.0;
533 if (dim == 0) {
534 return 0.0;
535 }
536 for (size_t i = 0; i < dim; i++) {
537 norm += pow(dx[i] / wtX[i], 2);
538 }
539 return sqrt(norm/dim);
540}

542void solveSP::calcWeights(doublereal wtSpecies[], doublereal wtResid[],
543 const Array2D& Jac, const doublereal CSoln[],
544 const doublereal abstol, const doublereal reltol)
545{
546 // First calculate the weighting factor for the concentrations of the
547 // surface species and bulk species.
548 size_t kindex = 0;
549 for (size_t isp = 0; isp < m_numSurfPhases; isp++) {
550 double sd = m_ptrsSurfPhase[isp]->siteDensity();
551 for (size_t k = 0; k < m_nSpeciesSurfPhase[isp]; k++, kindex++) {
552 wtSpecies[kindex] = abstol * sd + reltol * fabs(CSoln[kindex]);
553 }
554 }
555 if (m_bulkFunc == BULK_DEPOSITION) {
556 for (size_t isp = 0; isp < m_numBulkPhasesSS; isp++) {
557 double sd = m_bulkPhasePtrs[isp]->molarDensity();
558 for (size_t k = 0; k < m_numBulkSpecies[isp]; k++, kindex++) {
559 wtSpecies[kindex] = abstol * sd + reltol * fabs(CSoln[kindex]);
560 }
561 }
562 }
563
564 // Now do the residual Weights. Since we have the Jacobian, we will use it
565 // to generate a number based on the what a significant change in a solution
566 // variable does to each residual. This is a row sum scale operation.
567 for (size_t k = 0; k < m_neq; k++) {
568 wtResid[k] = 0.0;
569 for (size_t jcol = 0; jcol < m_neq; jcol++) {
570 wtResid[k] += fabs(Jac(k,jcol) * wtSpecies[jcol]);
571 }
572 }
573}

575doublereal solveSP::calc_t(doublereal netProdRateSolnSP[],
576 doublereal XMolSolnSP[],
577 int* label, int* label_old,
578 doublereal* label_factor, int ioflag)
579{
580 doublereal inv_timeScale = 1.0E-10;
581 size_t kindexSP = 0;
582 *label = 0;
583 updateMFSolnSP(XMolSolnSP);
584 for (size_t isp = 0; isp < m_numSurfPhases; isp++) {
585 // Get the interface kinetics associated with this surface
586 InterfaceKinetics* kin = m_objects[isp];
587
588 // Calculate the start of the species index for surfaces within
589 // the InterfaceKinetics object
590 size_t surfIndex = kin->reactionPhaseIndex();
591 size_t kstart = kin->kineticsSpeciesIndex(0, surfIndex);
592 kin->getNetProductionRates(m_numEqn1.data());
593 double sden = kin->thermo(surfIndex).molarDensity();
594 for (size_t k = 0; k < m_nSpeciesSurfPhase[isp]; k++, kindexSP++) {
595 size_t kspindex = kstart + k;
596 netProdRateSolnSP[kindexSP] = m_numEqn1[kspindex];
597 double tmp = std::max(XMolSolnSP[kindexSP], 1.0e-10);
598 tmp *= sden;
599 tmp = fabs(netProdRateSolnSP[kindexSP]/ tmp);
600 if (netProdRateSolnSP[kindexSP]> 0.0) {
601 tmp /= 100.;
602 }
603 if (tmp > inv_timeScale) {
604 inv_timeScale = tmp;
605 *label = int(kindexSP);
606 }
607 }
608 }
609
610 // Increase time step exponentially as same species repeatedly controls time
611 // step
612 if (*label == *label_old) {
613 *label_factor *= 1.5;
614 } else {
615 *label_old = *label;
616 *label_factor = 1.0;
617 }
618 return inv_timeScale / *label_factor;
619} // calc_t
