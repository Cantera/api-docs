77shared_ptr<Kinetics> newKinetics(const vector<shared_ptr<ThermoPhase>>& phases,
78 const AnyMap& phaseNode,
79 const AnyMap& rootNode,
80 shared_ptr<Solution> soln)
81{
82 std::string kinType = phaseNode.getString("kinetics", "none");
83 kinType = KineticsFactory::factory()->canonicalize(kinType);
84 if (kinType == "none") {
85 // determine phase with minimum number of dimensions
86 size_t nDim = 3;
87 for (auto& phase : phases) {
88 nDim = std::min(phase->nDim(), nDim);
89 }
90 // change kinetics type as necessary
91 if (nDim == 2) {
92 kinType = "surface";
93 } else if (nDim == 1) {
94 kinType = "edge";
95 }
96 }
97
98 shared_ptr<Kinetics> kin(KineticsFactory::factory()->newKinetics(kinType));
99 if (soln) {
100 soln->setKinetics(kin);
101 }
102 for (auto& phase : phases) {
103 kin->addThermo(phase);
104 }
105 kin->init();
106 addReactions(*kin, phaseNode, rootNode);
107 return kin;
108}

169void addReactions(Kinetics& kin, const AnyMap& phaseNode, const AnyMap& rootNode)
170{
171 kin.skipUndeclaredThirdBodies(
172 phaseNode.getBool("skip-undeclared-third-bodies", false));
173
174 loadExtensions(rootNode);
175
176 // Find sections containing reactions to add
177 vector<string> sections, rules;
178
179 if (phaseNode.hasKey("reactions")) {
180 const auto& reactionsNode = phaseNode.at("reactions");
181 if (reactionsNode.is<string>()) {
182 if (rootNode.hasKey("reactions")) {
183 // Specification of the rule for adding species from the default
184 // 'reactions' section, if it exists
185 sections.push_back("reactions");
186 rules.push_back(reactionsNode.asString());
187 } else if (reactionsNode.asString() != "none") {
188 throw InputFileError("addReactions", reactionsNode,
189 "Phase entry implies existence of 'reactions' section "
190 "which does not exist in the current input file.");
191 }
192 } else if (reactionsNode.is<vector<string>>()) {
193 // List of sections from which all species should be added
194 for (const auto& item : reactionsNode.as<vector<string>>()) {
195 sections.push_back(item);
196 rules.push_back("all");
197 }
198 } else if (reactionsNode.is<vector<AnyMap>>()) {
199 // Mapping of rules to apply for each specified section containing
200 // reactions
201 for (const auto& item : reactionsNode.as<vector<AnyMap>>()) {
202 sections.push_back(item.begin()->first);
203 rules.push_back(item.begin()->second.asString());
204 }
205 }
206 } else if (kin.kineticsType() != "none") {
207 if (!phaseNode.hasKey("kinetics")) {
208 // Do nothing - default surface or edge kinetics require separate detection
209 // while not adding reactions
210 } else if (rootNode.hasKey("reactions")) {
211 // Default behavior is to add all reactions from the 'reactions'
212 // section, if a 'kinetics' model has been specified
213 sections.push_back("reactions");
214 rules.push_back("all");
215 } else {
216 throw InputFileError("addReactions", phaseNode,
217 "Phase entry implies existence of 'reactions' section which "
218 "does not exist in the current input file. Add the field "
219 "'reactions: none' to the phase entry to specify a kinetics "
220 "model with no reactions.");
221 }
222 }
223
224 // Add reactions from each section
225 fmt::memory_buffer add_rxn_err;
226 for (size_t i = 0; i < sections.size(); i++) {
227 if (rules[i] == "all") {
228 kin.skipUndeclaredSpecies(false);
229 } else if (rules[i] == "declared-species") {
230 kin.skipUndeclaredSpecies(true);
231 } else if (rules[i] == "none") {
232 continue;
233 } else {
234 throw InputFileError("addReactions", phaseNode.at("reactions"),
235 "Unknown rule '{}' for adding species from the '{}' section.",
236 rules[i], sections[i]);
237 }
238 const auto& slash = boost::ifind_last(sections[i], "/");
239 if (slash) {
240 // specified section is in a different file
241 string fileName (sections[i].begin(), slash.begin());
242 string node(slash.end(), sections[i].end());
243 AnyMap reactions = AnyMap::fromYamlFile(fileName,
244 rootNode.getString("__file__", ""));
245 loadExtensions(reactions);
246 for (const auto& R : reactions[node].asVector<AnyMap>()) {
247 #ifdef NDEBUG
248 try {
249 kin.addReaction(newReaction(R, kin), false);
250 } catch (CanteraError& err) {
251 fmt_append(add_rxn_err, "{}", err.what());
252 }
253 #else
254 kin.addReaction(newReaction(R, kin), false);
255 #endif
256 }
257 } else {
258 // specified section is in the current file
259 for (const auto& R : rootNode.at(sections[i]).asVector<AnyMap>()) {
260 #ifdef NDEBUG
261 try {
262 kin.addReaction(newReaction(R, kin), false);
263 } catch (CanteraError& err) {
264 fmt_append(add_rxn_err, "{}", err.what());
265 }
266 #else
267 kin.addReaction(newReaction(R, kin), false);
268 #endif
269 }
270 }
271 }
272
273 if (add_rxn_err.size()) {
274 throw CanteraError("addReactions", to_string(add_rxn_err));
275 }
276 kin.checkDuplicates();
277 kin.resizeReactions();
278}
