20//! Critical Temperature value (kelvin)
21const doublereal T_c = 647.096;
22//! Critical Pressure (Pascals)
23static const doublereal P_c = 22.064E6;
24//! Value of the Density at the critical point (kg m-3)
25const doublereal Rho_c = 322.;
26//! Molecular Weight of water that is consistent with the paper (kg kmol-1)
27static const doublereal M_water = 18.015268;
28
29static const double R_water = 461.51805; // J/kg/K (Eq. 6.3)
30
31//! Gas constant that is quoted in the paper
32/*
33 * Note, this is the Rgas value quoted in the paper. For consistency
34 * we have to use that value and not the updated value
35 *
36 * The Ratio of R/M = 0.46151805 kJ kg-1 K-1 , which is Eqn. (6.3) in the paper.
37 */
38static const doublereal Rgas = 8.314371E3; // Joules kmol-1 K-1

40void WaterPropsIAPWS::calcDim(doublereal temperature, doublereal rho)
41{
42 tau = T_c / temperature;
43 delta = rho / Rho_c;
44
45 // Determine the internal state
46 if (temperature > T_c) {
47 iState = WATER_SUPERCRIT;
48 } else {
49 if (delta < 1.0) {
50 iState = WATER_GAS;
51 } else {
52 iState = WATER_LIQUID;
53 }
54 }

75doublereal WaterPropsIAPWS::density(doublereal temperature, doublereal pressure,
76 int phase, doublereal rhoguess)
77{
78 if (fabs(pressure - P_c) / P_c < 1.e-8 &&
79 fabs(temperature - T_c) / T_c < 1.e-8) {
80 // Catch critical point, as no solution is found otherwise
81 setState_TD(temperature, Rho_c);
82 return Rho_c;
83 }
84 doublereal deltaGuess = 0.0;
85 if (rhoguess == -1.0) {
86 if (phase != -1) {
87 if (temperature > T_c) {
88 rhoguess = pressure / (R_water * temperature);
89 } else {
90 if (phase == WATER_GAS || phase == WATER_SUPERCRIT) {
91 rhoguess = pressure / (R_water * temperature);
92 } else if (phase == WATER_LIQUID) {
93 // Provide a guess about the liquid density that is
94 // relatively high -> convergence from above seems robust.
95 rhoguess = 1000.;
96 } else if (phase == WATER_UNSTABLELIQUID || phase == WATER_UNSTABLEGAS) {
97 throw CanteraError("WaterPropsIAPWS::density",
98 "Unstable Branch finder is untested");
99 } else {
100 throw CanteraError("WaterPropsIAPWS::density",
101 "unknown state: {}", phase);
102 }
103 }
104 } else {
105 // Assume the Gas phase initial guess, if nothing is specified to
106 // the routine
107 rhoguess = pressure / (R_water * temperature);
108 }
109 }
110 double p_red = pressure / (R_water * temperature * Rho_c);
111 deltaGuess = rhoguess / Rho_c;
112 setState_TD(temperature, rhoguess);
113 doublereal delta_retn = m_phi.dfind(p_red, tau, deltaGuess);
114 if (delta_retn <= 0) {
115 // No solution found for first initial guess; perturb initial guess once
116 // to avoid spurious failures (band-aid fix)
117 delta_retn = m_phi.dfind(p_red, tau, 0.9 * deltaGuess);
118 }
119 doublereal density_retn;
120 if (delta_retn > 0.0) {
121 delta = delta_retn;
122
123 // Dimensionalize the density before returning
124 density_retn = delta_retn * Rho_c;
125
126 // Set the internal state -> this may be a duplication. However, let's
127 // just be sure.
128 setState_TD(temperature, density_retn);
129 } else {
130 density_retn = -1.0;
131 }
132 return density_retn;
133}

137{

139 doublereal deltaGuess = 0.0;
140 doublereal deltaSave = delta;
141 if (rhoguess == -1.0) {
142 if (phase != -1) {
143 if (temperature > T_c) {
144 rhoguess = pressure / (R_water * temperature);
145 } else {
146 if (phase == WATER_GAS || phase == WATER_SUPERCRIT) {
147 rhoguess = pressure / (R_water * temperature);
148 } else if (phase == WATER_LIQUID) {
149 // Provide a guess about the liquid density that is
150 // relatively high -> convergence from above seems robust.
151 rhoguess = 1000.;
152 } else if (phase == WATER_UNSTABLELIQUID || phase == WATER_UNSTABLEGAS) {
153 throw CanteraError("WaterPropsIAPWS::density_const",
154 "Unstable Branch finder is untested");
155 } else {
156 throw CanteraError("WaterPropsIAPWS::density_const",
157 "unknown state: {}", phase);
158 }
159 }
160 } else {
161 // Assume the Gas phase initial guess, if nothing is specified to
162 // the routine
163 rhoguess = pressure / (R_water * temperature);
164 }
165 }

168

181 density_retn = -1.0;
182 }
183

185 m_phi.tdpolycalc(tau, delta);
186 return density_retn;

199doublereal WaterPropsIAPWS::psat_est(doublereal temperature) const
200{
201 // Formula and constants from: "NBS/NRC Steam Tables: Thermodynamic and
202 // Transport Properties and Computer Programs for Vapor and Liquid States of
203 // Water in SI Units". L. Haar, J. S. Gallagher, G. S. Kell. Hemisphere
204 // Publishing. 1984.
205 static const doublereal A[8] = {
206 -7.8889166E0,
207 2.5514255E0,
208 -6.716169E0,
209 33.2239495E0,
210 -105.38479E0,
211 174.35319E0,
212 -148.39348E0,
213 48.631602E0
214 };
215 doublereal ps;
216 if (temperature < 314.) {
217 doublereal pl = 6.3573118E0 - 8858.843E0 / temperature
218 + 607.56335E0 * pow(temperature, -0.6);
219 ps = 0.1 * exp(pl);
220 } else {
221 doublereal v = temperature / 647.25;
222 doublereal w = fabs(1.0-v);
223 doublereal b = 0.0;
224 for (int i = 0; i < 8; i++) {
225 doublereal z = i + 1;
226 b += A[i] * pow(w, ((z+1.0)/2.0));
227 }
228 doublereal q = b / v;
229 ps = 22.093*exp(q);
230 }
231
232 // Original correlation was in cgs. Convert to mks
233 ps *= 1.0E6;
234 return ps;
235}

273void WaterPropsIAPWS::corr(doublereal temperature, doublereal pressure,
274 doublereal& densLiq, doublereal& densGas, doublereal& delGRT)
275{
276 densLiq = density(temperature, pressure, WATER_LIQUID, densLiq);
277 if (densLiq <= 0.0) {
278 throw CanteraError("WaterPropsIAPWS::corr",
279 "Error occurred trying to find liquid density at (T,P) = {} {}",
280 temperature, pressure);
281 }
282 setState_TD(temperature, densLiq);
283 doublereal gibbsLiqRT = m_phi.gibbs_RT();
284
285 densGas = density(temperature, pressure, WATER_GAS, densGas);
286 if (densGas <= 0.0) {
287 throw CanteraError("WaterPropsIAPWS::corr",
288 "Error occurred trying to find gas density at (T,P) = {} {}",
289 temperature, pressure);
290 }
291 setState_TD(temperature, densGas);
292 doublereal gibbsGasRT = m_phi.gibbs_RT();
293
294 delGRT = gibbsLiqRT - gibbsGasRT;
295}

297void WaterPropsIAPWS::corr1(doublereal temperature, doublereal pressure,
298 doublereal& densLiq, doublereal& densGas, doublereal& pcorr)
299{
300 densLiq = density(temperature, pressure, WATER_LIQUID, densLiq);
301 if (densLiq <= 0.0) {
302 throw CanteraError("WaterPropsIAPWS::corr1",
303 "Error occurred trying to find liquid density at (T,P) = {} {}",
304 temperature, pressure);
305 }
306 setState_TD(temperature, densLiq);
307 doublereal prL = m_phi.phiR();
308
309 densGas = density(temperature, pressure, WATER_GAS, densGas);
310 if (densGas <= 0.0) {
311 throw CanteraError("WaterPropsIAPWS::corr1",
312 "Error occurred trying to find gas density at (T,P) = {} {}",
313 temperature, pressure);
314 }
315 setState_TD(temperature, densGas);
316 doublereal prG = m_phi.phiR();
317 doublereal rhs = (prL - prG) + log(densLiq/densGas);
318 rhs /= (1.0/densGas - 1.0/densLiq);
319 pcorr = rhs * R_water * temperature;
320}

322doublereal WaterPropsIAPWS::psat(doublereal temperature, int waterState)
323{
324 static int method = 1;
325 double densLiq = -1.0, densGas = -1.0, delGRT = 0.0;
326 double dp, pcorr;
327 if (temperature >= T_c) {
328 densGas = density(temperature, P_c, WATER_SUPERCRIT);
329 setState_TD(temperature, densGas);
330 return P_c;
331 }
332 double p = psat_est(temperature);
333 for (int i = 0; i < 30; i++) {
334 if (method == 1) {
335 corr(temperature, p, densLiq, densGas, delGRT);
336 double delV = 1.0/densLiq - 1.0/densGas;
337 dp = - delGRT * R_water * temperature / delV;
338 } else {
339 corr1(temperature, p, densLiq, densGas, pcorr);
340 dp = pcorr - p;
341 }
342 p += dp;
343
344 if ((method == 1) && delGRT < 1.0E-8) {
345 break;
346 } else {
347 if (fabs(dp/p) < 1.0E-9) {
348 break;
349 }
350 }
351 }
352 // Put the fluid in the desired end condition
353 if (waterState == WATER_LIQUID) {
354 setState_TD(temperature, densLiq);
355 } else if (waterState == WATER_GAS) {
356 setState_TD(temperature, densGas);
357 } else {
358 throw CanteraError("WaterPropsIAPWS::psat",
359 "unknown water state input: {}", waterState);
360 }
361 return p;
362}

364int WaterPropsIAPWS::phaseState(bool checkState) const
365{
366 if (checkState) {
367 if (tau <= 1.0) {
368 iState = WATER_SUPERCRIT;
369 } else {
370 double T = T_c / tau;
371 double rho = delta * Rho_c;
372 double rhoMidAtm = 0.5 * (OneAtm / (R_water * 373.15) + 1.0E3);
373 double rhoMid = Rho_c + (T - T_c) * (Rho_c - rhoMidAtm) / (T_c - 373.15);
374 int iStateGuess = WATER_LIQUID;
375 if (rho < rhoMid) {
376 iStateGuess = WATER_GAS;
377 }
378 doublereal kappa = isothermalCompressibility();
379 if (kappa >= 0.0) {
380 iState = iStateGuess;
381 } else {
382 // When we are here we are between the spinodal curves
383 doublereal rhoDel = rho * 1.000001;
384 doublereal deltaSave = delta;
385 doublereal deltaDel = rhoDel / Rho_c;
386 delta = deltaDel;
387 m_phi.tdpolycalc(tau, deltaDel);
388
389 doublereal kappaDel = isothermalCompressibility();
390 doublereal d2rhodp2 = (rhoDel * kappaDel - rho * kappa) / (rhoDel - rho);
391 if (d2rhodp2 > 0.0) {
392 iState = WATER_UNSTABLELIQUID;
393 } else {
394 iState = WATER_UNSTABLEGAS;
395 }
396 delta = deltaSave;
397 m_phi.tdpolycalc(tau, delta);
398 }
399 }
400 }
401 return iState;
402}

404doublereal WaterPropsIAPWS::densSpinodalWater() const
405{
406 doublereal temperature = T_c/tau;
407 doublereal delta_save = delta;
408 // return the critical density if we are above or even just a little below
409 // the critical temperature. We just don't want to worry about the critical
410 // point at this juncture.
411 if (temperature >= T_c - 0.001) {
412 return Rho_c;
413 }
414 doublereal p = psat_est(temperature);
415 doublereal rho_low = 0.0;
416 doublereal rho_high = 1000;
417 doublereal densSatLiq = density_const(p, WATER_LIQUID);
418 doublereal dens_old = densSatLiq;
419 delta = dens_old / Rho_c;
420 m_phi.tdpolycalc(tau, delta);
421 doublereal dpdrho_old = dpdrho();
422 if (dpdrho_old > 0.0) {
423 rho_high = std::min(dens_old, rho_high);
424 } else {
425 rho_low = std::max(rho_low, dens_old);
426 }
427 doublereal dens_new = densSatLiq* (1.0001);
428 delta = dens_new / Rho_c;
429 m_phi.tdpolycalc(tau, delta);
430 doublereal dpdrho_new = dpdrho();
431 if (dpdrho_new > 0.0) {
432 rho_high = std::min(dens_new, rho_high);
433 } else {
434 rho_low = std::max(rho_low, dens_new);
435 }
436 bool conv = false;
437
438 for (int it = 0; it < 50; it++) {
439 doublereal slope = (dpdrho_new - dpdrho_old)/(dens_new - dens_old);
440 if (slope >= 0.0) {
441 slope = std::max(slope, dpdrho_new *5.0/ dens_new);
442 } else {
443 slope = -dpdrho_new;
444 // shouldn't be here for liquid spinodal
445 }
446 doublereal delta_rho = - dpdrho_new / slope;
447 if (delta_rho > 0.0) {
448 delta_rho = std::min(delta_rho, dens_new * 0.1);
449 } else {
450 delta_rho = std::max(delta_rho, - dens_new * 0.1);
451 }
452 doublereal dens_est = dens_new + delta_rho;
453 if (dens_est < rho_low) {
454 dens_est = 0.5 * (rho_low + dens_new);
455 }
456 if (dens_est > rho_high) {
457 dens_est = 0.5 * (rho_high + dens_new);
458 }
459
460 dens_old = dens_new;
461 dpdrho_old = dpdrho_new;
462 dens_new = dens_est;
463
464 delta = dens_new / Rho_c;
465 m_phi.tdpolycalc(tau, delta);
466 dpdrho_new = dpdrho();
467 if (dpdrho_new > 0.0) {
468 rho_high = std::min(dens_new, rho_high);
469 } else if (dpdrho_new < 0.0) {
470 rho_low = std::max(rho_low, dens_new);
471 } else {
472 conv = true;
473 break;
474 }
475
476 if (fabs(dpdrho_new) < 1.0E-5) {
477 conv = true;
478 break;
479 }
480 }
481
482 if (!conv) {
483 throw CanteraError("WaterPropsIAPWS::densSpinodalWater",
484 "convergence failure");
485 }
486 // Restore the original delta
487 delta = delta_save;
488 m_phi.tdpolycalc(tau, delta);
489 return dens_new;
490}

493{

522 rho_low = std::max(rho_low, dens_new);

527 if (slope >= 0.0) {

529 // shouldn't be here for gas spinodal

531 slope = std::min(slope, dpdrho_new *5.0 / dens_new);

564 conv = true;
565 break;

567 }
568

572 }
