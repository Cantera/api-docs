20 bool BlowersMaselData::update(const ThermoPhase& phase, const Kinetics& kin)
21 {
22  double rho = phase.density();
23  int mf = phase.stateMFNumber();
24  double T = phase.temperature();
25  bool changed = false;
26  if (T != temperature) {
27  ReactionData::update(T);
28  changed = true;
29  }
30  if (changed || rho != density || mf != m_state_mf_number) {
31  density = rho;
32  m_state_mf_number = mf;
33  phase.getPartialMolarEnthalpies(partialMolarEnthalpies.data());
34  changed = true;
35  }
36  return changed;
37 }

59 double BlowersMaselRate::ddTScaledFromStruct(const BlowersMaselData& shared_data) const
60 {
