51doublereal SingleSpeciesTP::cp_mole() const
52{
53 double cpbar;
54
55 // Really should have a partial molar heat capacity function in ThermoPhase.
56 // However, the standard state heat capacity will do fine here for now.
57 getCp_R(&cpbar);
58 cpbar *= GasConstant;
59 return cpbar;
60}

62doublereal SingleSpeciesTP::cv_mole() const
63{
64 // For single species, we go directory to the general Cp - Cv relation
65 //
66 // Cp = Cv + alpha**2 * V * T / beta
67 //
68 // where
69 // alpha = volume thermal expansion coefficient
70 // beta = isothermal compressibility
71 doublereal cvbar = cp_mole();
72 doublereal alpha = thermalExpansionCoeff();
73 doublereal beta = isothermalCompressibility();
74 doublereal V = molecularWeight(0)/density();
75 doublereal T = temperature();
76 if (beta != 0.0) {
77 cvbar -= alpha * alpha * V * T / beta;
78 }
79 return cvbar;
80}

173void SingleSpeciesTP::setState_HP(doublereal h, doublereal p,
174 doublereal tol)
175{
176 doublereal dt;
177 setPressure(p);
178 for (int n = 0; n < 50; n++) {
179 dt = clip((h - enthalpy_mass())/cp_mass(), -100.0, 100.0);
180 setState_TP(temperature() + dt, p);
181 if (fabs(dt / temperature()) < tol) {
182 return;
183 }
184 }
185 throw CanteraError("SingleSpeciesTP::setState_HP",
186 "no convergence. dt = {}", dt);
187}

189void SingleSpeciesTP::setState_UV(doublereal u, doublereal v,
190 doublereal tol)
191{
192 doublereal dt;
193 if (v == 0.0) {
194 setDensity(1.0E100);
195 } else {
196 setDensity(1.0/v);
197 }
198 for (int n = 0; n < 50; n++) {
199 dt = clip((u - intEnergy_mass())/cv_mass(), -100.0, 100.0);
200 setTemperature(temperature() + dt);
201 if (fabs(dt / temperature()) < tol) {
202 return;
203 }
204 }
205 throw CanteraError("SingleSpeciesTP::setState_UV",
206 "no convergence. dt = {}\nu = {} v = {}", dt, u, v);
207}

249 "Stoichiometric substances may only contain one species.");

254void SingleSpeciesTP::_updateThermo() const
255{
256 doublereal tnow = temperature();
257 if (m_tlast != tnow) {
258 m_spthermo.update(tnow, &m_cp0_R, &m_h0_RT, &m_s0_R);
259 m_tlast = tnow;
260 }
261}
