28 public:
29  virtual string type() override {
30  if (!m_rxn_rates.size()) {
31  throw CanteraError("MultiRate::type",

34  return m_rxn_rates.at(0).second.type();

38  m_indices[rxn_index] = m_rxn_rates.size();
39  m_rxn_rates.emplace_back(rxn_index, dynamic_cast<RateType&>(rate));

43  virtual bool replace(size_t rxn_index, ReactionRate& rate) override {
44  if (!m_rxn_rates.size()) {
45  throw CanteraError("MultiRate::replace",
46  "Invalid operation: cannot replace rate object "
47  "in empty rate handler.");
48  }
49  if (rate.type() != type()) {
50  throw CanteraError("MultiRate::replace",
51  "Invalid operation: cannot replace rate object of type '{}' "
52  "with a new rate of type '{}'.", type(), rate.type());
53  }
54  m_shared.invalidateCache();
55  if (m_indices.find(rxn_index) != m_indices.end()) {
56  size_t j = m_indices[rxn_index];
57  m_rxn_rates.at(j).second = dynamic_cast<RateType&>(rate);
58  return true;
59  }
60  return false;
61  }

74  virtual void processRateConstants_ddT(double* rop,
75  const double* kf,
76  double deltaT) override
77  {
78  if constexpr (has_ddT<RateType>::value) {
79  for (const auto& [iRxn, rate] : m_rxn_rates) {
80  rop[iRxn] *= rate.ddTScaledFromStruct(m_shared);
81  }
82  } else {
83  // perturb conditions
84  double dTinv = 1. / (m_shared.temperature * deltaT);
85  m_shared.perturbTemperature(deltaT);
86  _update();
87 
88  // apply numerical derivative
89  for (auto& [iRxn, rate] : m_rxn_rates) {
90  if (kf[iRxn] != 0.) {
91  double k1 = rate.evalFromStruct(m_shared);
92  rop[iRxn] *= dTinv * (k1 / kf[iRxn] - 1.);
93  } // else not needed: derivative is already zero
94  }
95 
96  // revert changes
97  m_shared.restore();
98  _update();
99  }
100  }
101 
102  virtual void processRateConstants_ddP(double* rop,
103  const double* kf,
104  double deltaP) override
105  {
106  if constexpr (has_ddP<DataType>::value) {
107  double dPinv = 1. / (m_shared.pressure * deltaP);
108  m_shared.perturbPressure(deltaP);
109  _update();
110 
111  for (auto& [iRxn, rate] : m_rxn_rates) {
112  if (kf[iRxn] != 0.) {
113  double k1 = rate.evalFromStruct(m_shared);
114  rop[iRxn] *= dPinv * (k1 / kf[iRxn] - 1.);
115  } // else not needed: derivative is already zero
116  }
117 
118  // revert changes
119  m_shared.restore();
120  _update();
121  } else {
122  for (const auto& [iRxn, rate] : m_rxn_rates) {
123  rop[iRxn] = 0.;
124  }
125  }
126  }
127 
128  virtual void processRateConstants_ddM(double* rop,
129  const double* kf,
130  double deltaM,
131  bool overwrite=true) override
132  {
133  if constexpr (has_ddM<DataType>::value) {
134  double dMinv = 1. / deltaM;
135  m_shared.perturbThirdBodies(deltaM);
136  _update();
137 
138  for (auto& [iRxn, rate] : m_rxn_rates) {
139  if (kf[iRxn] != 0. && m_shared.conc_3b[iRxn] > 0.) {
140  double k1 = rate.evalFromStruct(m_shared);
141  rop[iRxn] *= dMinv * (k1 / kf[iRxn] - 1.);
142  rop[iRxn] /= m_shared.conc_3b[iRxn];
143  } else {
144  rop[iRxn] = 0.;
145  }
146  }
147 
148  // revert changes
149  m_shared.restore();
150  _update();
151  } else {
152  if (!overwrite) {
153  // do not overwrite existing entries
154  return;
155  }
156  for (const auto& [iRxn, rate] : m_rxn_rates) {
157  rop[iRxn] = 0.;
158  }
159  }
160  }
