168//! Arrhenius reaction rate type depends only on temperature
169/*!
170 * A reaction rate coefficient of the following form.
171 *
172 * \f[
173 * k_f = A T^b \exp (-Ea/RT)
174 * \f]
175 *
176 * @ingroup arrheniusGroup
177 */
178class ArrheniusRate : public ArrheniusBase
179{
180public:
181 using ArrheniusBase::ArrheniusBase; // inherit constructors
182
183 unique_ptr<MultiRateBase> newMultiRate() const override {
184 return make_unique<MultiRate<ArrheniusRate, ArrheniusData>>();
185 }
186
187 virtual const std::string type() const override {
188 return "Arrhenius";
189 }
190
191 //! Evaluate reaction rate
192 double evalRate(double logT, double recipT) const {
193 return m_A * std::exp(m_b * logT - m_Ea_R * recipT);
194 }
195
196 //! Evaluate natural logarithm of the rate constant.
197 double evalLog(double logT, double recipT) const {
198 return m_logA + m_b * logT - m_Ea_R * recipT;
199 }
200
201 //! Evaluate reaction rate
202 /*!
203 * @param shared_data data shared by all reactions of a given type
204 */
205 double evalFromStruct(const ArrheniusData& shared_data) const {
206 return m_A * std::exp(m_b * shared_data.logT - m_Ea_R * shared_data.recipT);
207 }
208
209 //! Evaluate derivative of reaction rate with respect to temperature
210 //! divided by reaction rate
211 /*!
212 * @param shared_data data shared by all reactions of a given type
213 */
214 double ddTScaledFromStruct(const ArrheniusData& shared_data) const {
215 return (m_Ea_R * shared_data.recipT + m_b) * shared_data.recipT;
216 }
217};
