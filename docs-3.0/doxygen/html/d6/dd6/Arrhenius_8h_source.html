160 //! Arrhenius reaction rate type depends only on temperature
161 /*!
162  * A reaction rate coefficient of the following form.
163  *
164  * @f[
165  * k_f = A T^b \exp (-Ea/RT)
166  * @f]
167  *
168  * @ingroup arrheniusGroup
169  */
170 class ArrheniusRate : public ArrheniusBase
171 {
172 public:
173  using ArrheniusBase::ArrheniusBase; // inherit constructors
174 
175  unique_ptr<MultiRateBase> newMultiRate() const override {
176  return make_unique<MultiRate<ArrheniusRate, ArrheniusData>>();
177  }
178 
179  virtual const string type() const override {
180  return "Arrhenius";
181  }
182 
183  //! Evaluate reaction rate
184  double evalRate(double logT, double recipT) const {
185  return m_A * std::exp(m_b * logT - m_Ea_R * recipT);
186  }
187 
188  //! Evaluate natural logarithm of the rate constant.
189  double evalLog(double logT, double recipT) const {
190  return m_logA + m_b * logT - m_Ea_R * recipT;
191  }
192 
193  //! Evaluate reaction rate
194  /*!
195  * @param shared_data data shared by all reactions of a given type
196  */
197  double evalFromStruct(const ArrheniusData& shared_data) const {
198  return m_A * std::exp(m_b * shared_data.logT - m_Ea_R * shared_data.recipT);
199  }
200 
201  //! Evaluate derivative of reaction rate with respect to temperature
202  //! divided by reaction rate
203  /*!
204  * @param shared_data data shared by all reactions of a given type
205  */
206  double ddTScaledFromStruct(const ArrheniusData& shared_data) const {
207  return (m_Ea_R * shared_data.recipT + m_b) * shared_data.recipT;
208  }
209 };
