353  /**
354  * Update the thermodynamic properties from point j0 to point j1
355  * (inclusive), based on solution x.
356  */
357  void updateThermo(const double* x, size_t j0, size_t j1) {
358  for (size_t j = j0; j <= j1; j++) {
359  setGas(x,j);
360  m_rho[j] = m_thermo->density();
361  m_wtm[j] = m_thermo->meanMolecularWeight();
362  m_cp[j] = m_thermo->cp_mass();
363  m_thermo->getPartialMolarEnthalpies(&m_hk(0, j));
364  }
365  }
