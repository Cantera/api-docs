35void Kinetics::resizeReactions()
36{
37 size_t nRxn = nReactions();
38
39 // Stoichiometry managers
40 m_reactantStoich.resizeCoeffs(m_kk, nRxn);
41 m_productStoich.resizeCoeffs(m_kk, nRxn);
42 m_revProductStoich.resizeCoeffs(m_kk, nRxn);
43
44 m_rbuf.resize(nRxn);
45
46 // products are created for positive net rate of progress
47 m_stoichMatrix = m_productStoich.stoichCoeffs();
48 // reactants are destroyed for positive net rate of progress
49 m_stoichMatrix -= m_reactantStoich.stoichCoeffs();
50
51 m_ready = true;
52}

107std::pair<size_t, size_t> Kinetics::checkDuplicates(bool throw_err) const
108{
109 //! Map of (key indicating participating species) to reaction numbers
110 std::map<size_t, std::vector<size_t> > participants;
111 std::vector<std::map<int, double> > net_stoich;
112 std::unordered_set<size_t> unmatched_duplicates;
113 for (size_t i = 0; i < m_reactions.size(); i++) {
114 if (m_reactions[i]->duplicate) {
115 unmatched_duplicates.insert(i);
116 }
117 }
118
119 for (size_t i = 0; i < m_reactions.size(); i++) {
120 // Get data about this reaction
121 unsigned long int key = 0;
122 Reaction& R = *m_reactions[i];
123 net_stoich.emplace_back();
124 std::map<int, double>& net = net_stoich.back();
125 for (const auto& [name, stoich] : R.reactants) {
126 int k = static_cast<int>(kineticsSpeciesIndex(name));
127 key += k*(k+1);
128 net[-1 -k] -= stoich;
129 }
130 for (const auto& [name, stoich] : R.products) {
131 int k = static_cast<int>(kineticsSpeciesIndex(name));
132 key += k*(k+1);
133 net[1+k] += stoich;
134 }
135
136 // Compare this reaction to others with similar participants
137 vector<size_t>& related = participants[key];
138 for (size_t m : related) {
139 Reaction& other = *m_reactions[m];
140 if (R.duplicate && other.duplicate) {
141 // marked duplicates
142 unmatched_duplicates.erase(i);
143 unmatched_duplicates.erase(m);
144 continue;
145 } else if (R.type() != other.type()) {
146 continue; // different reaction types
147 } else if (R.rate() && other.rate()
148 && R.rate()->type() != other.rate()->type())
149 {
150 continue; // different rate parameterizations
151 }
152 doublereal c = checkDuplicateStoich(net_stoich[i], net_stoich[m]);
153 if (c == 0) {
154 continue; // stoichiometries differ (not by a multiple)
155 } else if (c < 0.0 && !R.reversible && !other.reversible) {
156 continue; // irreversible reactions in opposite directions
157 } else if (R.usesThirdBody() && other.usesThirdBody()) {
158 ThirdBody& tb1 = *(R.thirdBody());
159 ThirdBody& tb2 = *(other.thirdBody());
160 bool thirdBodyOk = true;
161 for (size_t k = 0; k < nTotalSpecies(); k++) {
162 string s = kineticsSpeciesName(k);
163 if (tb1.efficiency(s) * tb2.efficiency(s) != 0.0) {
164 // non-zero third body efficiencies for species `s` in
165 // both reactions
166 thirdBodyOk = false;
167 break;
168 }
169 }
170 if (thirdBodyOk) {
171 continue; // No overlap in third body efficiencies
172 }
173 }
174 if (throw_err) {
175 throw InputFileError("Kinetics::checkDuplicates",
176 R.input, other.input,
177 "Undeclared duplicate reactions detected:\n"
178 "Reaction {}: {}\nReaction {}: {}\n",
179 i+1, other.equation(), m+1, R.equation());
180 } else {
181 return {i,m};
182 }
183 }
184 participants[key].push_back(i);
185 }
186 if (unmatched_duplicates.size()) {
187 size_t i = *unmatched_duplicates.begin();
188 if (throw_err) {
189 throw InputFileError("Kinetics::checkDuplicates",
190 m_reactions[i]->input,
191 "No duplicate found for declared duplicate reaction number {}"
192 " ({})", i, m_reactions[i]->equation());
193 } else {
194 return {i, i};
195 }
196 }
197 return {npos, npos};
198}

200double Kinetics::checkDuplicateStoich(std::map<int, double>& r1,
201 std::map<int, double>& r2) const
202{
203 std::unordered_set<int> keys; // species keys (k+1 or -k-1)
204 for (auto& [speciesKey, stoich] : r1) {
205 keys.insert(speciesKey);
206 }
207 for (auto& [speciesKey, stoich] : r2) {
208 keys.insert(speciesKey);
209 }
210 int k1 = r1.begin()->first;
211 // check for duplicate written in the same direction
212 doublereal ratio = 0.0;
213 if (r1[k1] && r2[k1]) {
214 ratio = r2[k1]/r1[k1];
215 bool different = false;
216 for (int k : keys) {
217 if ((r1[k] && !r2[k]) ||
218 (!r1[k] && r2[k]) ||
219 (r1[k] && fabs(r2[k]/r1[k] - ratio) > 1.e-8)) {
220 different = true;
221 break;
222 }
223 }
224 if (!different) {
225 return ratio;
226 }
227 }
228
229 // check for duplicate written in the reverse direction
230 if (r1[k1] == 0.0 || r2[-k1] == 0.0) {
231 return 0.0;
232 }
233 ratio = r2[-k1]/r1[k1];
234 for (int k : keys) {
235 if ((r1[k] && !r2[-k]) ||
236 (!r1[k] && r2[-k]) ||
237 (r1[k] && fabs(r2[-k]/r1[k] - ratio) > 1.e-8)) {
238 return 0.0;
239 }
240 }
241 return ratio;
242}

594void Kinetics::addThermo(shared_ptr<ThermoPhase> thermo)
595{
596 // the phase with lowest dimensionality is assumed to be the
597 // phase/interface at which reactions take place
598 if (thermo->nDim() <= m_mindim) {
599 if (!m_thermo.empty()) {
600 warn_deprecated("Kinetics::addThermo", "The reacting (lowest dimensional) "
601 "phase should be added first. This warning will become an error after "
602 "Cantera 3.0.");
603 }
604 m_mindim = thermo->nDim();
605 m_rxnphase = nPhases();
606 }
607
608 // there should only be one surface phase
609 if (boost::algorithm::contains(thermo->type(), "surface")) {
610 m_surfphase = nPhases();
611 }
612 m_thermo.push_back(thermo.get());
613 m_sharedThermo.push_back(thermo);
614 m_phaseindex[m_thermo.back()->name()] = nPhases();
615 resizeSpecies();
616}

653{

664bool Kinetics::addReaction(shared_ptr<Reaction> r, bool resize)
665{
666 r->check();
667 r->validate(*this);
668
669 if (m_kk == 0) {
670 init();
671 }
672 resizeSpecies();
673
674 // Check validity of reaction within the context of the Kinetics object
675 if (!r->checkSpecies(*this)) {
676 // Do not add reaction
677 return false;
678 }
679
680 // For reactions created outside the context of a Kinetics object, the units
681 // of the rate coefficient can't be determined in advance. Do that here.
682 if (r->rate_units.factor() == 0) {
683 r->rate()->setRateUnits(r->calculateRateCoeffUnits(*this));
684 }
685
686 size_t irxn = nReactions(); // index of the new reaction
687
688 // indices of reactant and product species within this Kinetics object
689 std::vector<size_t> rk, pk;
690
691 // Reactant and product stoichiometric coefficients, such that rstoich[i] is
692 // the coefficient for species rk[i]
693 vector_fp rstoich, pstoich;
694
695 for (const auto& [name, stoich] : r->reactants) {
696 rk.push_back(kineticsSpeciesIndex(name));
697 rstoich.push_back(stoich);
698 }
699
700 for (const auto& [name, stoich] : r->products) {
701 pk.push_back(kineticsSpeciesIndex(name));
702 pstoich.push_back(stoich);
703 }
704
705 // The default order for each reactant is its stoichiometric coefficient,
706 // which can be overridden by entries in the Reaction.orders map. rorder[i]
707 // is the order for species rk[i].
708 vector_fp rorder = rstoich;
709 for (const auto& [name, order] : r->orders) {
710 size_t k = kineticsSpeciesIndex(name);
711 // Find the index of species k within rk
712 auto rloc = std::find(rk.begin(), rk.end(), k);
713 if (rloc != rk.end()) {
714 rorder[rloc - rk.begin()] = order;
715 } else {
716 // If the reaction order involves a non-reactant species, add an
717 // extra term to the reactants with zero stoichiometry so that the
718 // stoichiometry manager can be used to compute the global forward
719 // reaction rate.
720 rk.push_back(k);
721 rstoich.push_back(0.0);
722 rorder.push_back(order);
723 }
724 }
725
726 m_reactantStoich.add(irxn, rk, rorder, rstoich);
727 // product orders = product stoichiometric coefficients
728 m_productStoich.add(irxn, pk, pstoich, pstoich);
729 if (r->reversible) {
730 m_revProductStoich.add(irxn, pk, pstoich, pstoich);
731 }
732
733 m_reactions.push_back(r);
734 m_rfn.push_back(0.0);
735 m_delta_gibbs0.push_back(0.0);
736 m_rkcn.push_back(0.0);
737 m_ropf.push_back(0.0);
738 m_ropr.push_back(0.0);
739 m_ropnet.push_back(0.0);
740 m_perturb.push_back(1.0);
741 m_dH.push_back(0.0);
742
743 if (resize) {
744 resizeReactions();
745 } else {
746 m_ready = false;
747 }
748
749 return true;
750}

752void Kinetics::modifyReaction(size_t i, shared_ptr<Reaction> rNew)
753{
754 checkReactionIndex(i);
755 shared_ptr<Reaction>& rOld = m_reactions[i];
756 if (rNew->type() != rOld->type()) {
757 throw CanteraError("Kinetics::modifyReaction",
758 "Reaction types are different: {} != {}.",
759 rOld->type(), rNew->type());
760 }
761
762 if (rNew->rate()->type() != rOld->rate()->type()) {
763 throw CanteraError("Kinetics::modifyReaction",
764 "ReactionRate types are different: {} != {}.",
765 rOld->rate()->type(), rNew->rate()->type());
766 }
767
768 if (rNew->reactants != rOld->reactants) {
769 throw CanteraError("Kinetics::modifyReaction",
770 "Reactants are different: '{}' != '{}'.",
771 rOld->reactantString(), rNew->reactantString());
772 }
773
774 if (rNew->products != rOld->products) {
775 throw CanteraError("Kinetics::modifyReaction",
776 "Products are different: '{}' != '{}'.",
777 rOld->productString(), rNew->productString());
778 }
779 m_reactions[i] = rNew;
780 invalidateCache();
781}

795double Kinetics::reactionEnthalpy(const Composition& reactants, const Composition& products)
796{
797 warn_deprecated("Kinetics::reactionEnthalpy", "To be removed after Cantera 3.0");
798 vector_fp hk(nTotalSpecies());
799 for (size_t n = 0; n < nPhases(); n++) {
800 thermo(n).getPartialMolarEnthalpies(&hk[m_start[n]]);
801 }
802 double rxn_deltaH = 0;
803 for (const auto& [name, stoich] : products) {
804 size_t k = kineticsSpeciesIndex(name);
805 rxn_deltaH += hk[k] * stoich;
806 }
807 for (const auto& [name, stoich] : reactants) {
808 size_t k = kineticsSpeciesIndex(name);
809 rxn_deltaH -= hk[k] * stoich;
810 }
811 return rxn_deltaH;
812}
