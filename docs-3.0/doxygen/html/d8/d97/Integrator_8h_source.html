115 //! Set preconditioner used by the linear solver
116 /*!
117 * @param preconditioner preconditioner object used for the linear solver
118 */
119 virtual void setPreconditioner(shared_ptr<PreconditionerBase> preconditioner) {
120 m_preconditioner = preconditioner;
121 if (preconditioner->preconditionerSide() == "none") {
122 m_prec_side = PreconditionerSide::NO_PRECONDITION;
123 } else if (preconditioner->preconditionerSide() == "left") {
124 m_prec_side = PreconditionerSide::LEFT_PRECONDITION;
125 } else if (preconditioner->preconditionerSide() == "right") {
126 m_prec_side = PreconditionerSide::RIGHT_PRECONDITION;
127 } else if (preconditioner->preconditionerSide() == "both") {
128 m_prec_side = PreconditionerSide::BOTH_PRECONDITION;
129 } else {
130 throw CanteraError("Integrator::setPreconditioner",
131 "Invalid option '{}'", preconditioner->preconditionerSide());
132 }
133 }
134
135 //! Solve a linear system Ax=b where A is the preconditioner

138 * @param[in] rhs right hand side vector used in linear system

140 */
141 virtual void preconditionerSolve(size_t stateSize, double* rhs, double* output) {
142 m_preconditioner->solve(stateSize, rhs, output);
