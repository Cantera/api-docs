23void InterfaceData::update(double T, const vector_fp& values)
24{
25 warn_user("InterfaceData::update",
26 "This method does not update the site density.");
27 ReactionData::update(T);
28 sqrtT = sqrt(T);
29 if (coverages.size() == 0) {
30 coverages = values;
31 logCoverages.resize(values.size());
32 } else if (values.size() == coverages.size()) {
33 std::copy(values.begin(), values.end(), coverages.begin());
34 } else {
35 throw CanteraError("InterfaceData::update",
36 "Incompatible lengths of coverage arrays: received {} elements while "
37 "{} are required.", values.size(), coverages.size());
38 }
39 for (size_t n = 0; n < coverages.size(); n++) {
40 logCoverages[n] = std::log(std::max(coverages[n], Tiny));
41 }
42}

44bool InterfaceData::update(const ThermoPhase& phase, const Kinetics& kin)
45{
46 int mf = 0;
47 for (size_t n = 0; n < kin.nPhases(); n++) {
48 mf += kin.thermo(n).stateMFNumber();
49 }
50
51 double T = phase.temperature();
52 bool changed = false;
53 const auto& surf = dynamic_cast<const SurfPhase&>(
54 kin.thermo(kin.reactionPhaseIndex()));
55 double site_density = surf.siteDensity();
56 if (density != site_density) {
57 density = surf.siteDensity();
58 changed = true;
59 }
60 if (T != temperature) {
61 ReactionData::update(T);
62 sqrtT = sqrt(T);
63 changed = true;
64 }
65 if (changed || mf != m_state_mf_number) {
66 surf.getCoverages(coverages.data());
67 for (size_t n = 0; n < coverages.size(); n++) {
68 logCoverages[n] = std::log(std::max(coverages[n], Tiny));
69 }
70 for (size_t n = 0; n < kin.nPhases(); n++) {
71 size_t start = kin.kineticsSpeciesIndex(0, n);
72 const auto& ph = kin.thermo(n);
73 electricPotentials[n] = ph.electricPotential();
74 ph.getPartialMolarEnthalpies(partialMolarEnthalpies.data() + start);
75 ph.getStandardChemPotentials(standardChemPotentials.data() + start);
76 size_t nsp = ph.nSpecies();
77 for (size_t k = 0; k < nsp; k++) {
78 // only used for exchange current density formulation
79 standardConcentrations[k + start] = ph.standardConcentration(k);
80 }
81 }
82 m_state_mf_number = mf;
83 changed = true;
84 }
85 return changed;
86}

137void InterfaceRateBase::setCoverageDependencies(const AnyMap& dependencies,
138 const UnitSystem& units)
139{
140 m_cov.clear();
141 m_ac.clear();
142 m_ec.clear();
143 m_mc.clear();
144 m_lindep.clear();
145 for (const auto& [species, coeffs] : dependencies) {
146 double a, m;
147 vector_fp E(5, 0.0);
148 if (coeffs.is<AnyMap>()) {
149 auto& cov_map = coeffs.as<AnyMap>();
150 a = cov_map["a"].asDouble();
151 m = cov_map["m"].asDouble();
152 if (cov_map["E"].isScalar()) {
153 m_lindep.push_back(true);
154 E[1] = units.convertActivationEnergy(cov_map["E"], "K");
155 } else {
156 m_lindep.push_back(false);
157 auto& E_temp = cov_map["E"].asVector<AnyValue>(1, 4);
158 for (size_t i = 0; i < E_temp.size(); i++) {
159 E[i+1] = units.convertActivationEnergy(E_temp[i], "K");
160 }
161 }
162 } else {
163 auto& cov_vec = coeffs.asVector<AnyValue>(3);
164 a = cov_vec[0].asDouble();
165 m = cov_vec[1].asDouble();
166 if (cov_vec[2].isScalar()) {
167 m_lindep.push_back(true);
168 E[1] = units.convertActivationEnergy(cov_vec[2], "K");
169 } else {
170 m_lindep.push_back(false);
171 auto& E_temp = cov_vec[2].asVector<AnyValue>(1, 4);
172 for (size_t i = 0; i < E_temp.size(); i++) {
173 E[i+1] = units.convertActivationEnergy(E_temp[i], "K");
174 }
175 }
176 }
177 addCoverageDependence(species, a, m, E);
178 }

181void InterfaceRateBase::getCoverageDependencies(AnyMap& dependencies,
182 bool asVector) const
183{
184 for (size_t k = 0; k < m_cov.size(); k++) {
185 if (asVector) {
186 // this preserves the previous 'coverage_deps' units
187 warn_deprecated("InterfaceRateBase::getCoverageDependencies",
188 "To be changed after Cantera 3.0: second argument will be removed.");
189 vector_fp dep(3);
190 if (m_lindep[k]) {
191 dep[2] = m_ec[k][1];
192 } else {
193 throw NotImplementedError("InterfaceRateBase::getCoverageDependencies",
194 "Polynomial dependency not implemented for asVector.");
195 }
196 dep[0] = m_ac[k];
197 dep[1] = m_mc[k];
198 dependencies[m_cov[k]] = std::move(dep);
199 } else {
200 AnyMap dep;
201 dep["a"] = m_ac[k];
202 dep["m"] = m_mc[k];
203 if (m_lindep[k]) {
204 dep["E"].setQuantity(m_ec[k][1], "K", true);
205 } else {
206 vector<AnyValue> E_temp(4);
207 for (size_t i = 0; i < m_ec[k].size() - 1; i++) {
208 E_temp[i].setQuantity(m_ec[k][i+1], "K", true);
209 }
210 dep["E"] = E_temp;
211 }
212 dependencies[m_cov[k]] = std::move(dep);
213 }
214 }

247void InterfaceRateBase::updateFromStruct(const InterfaceData& shared_data) {
248 if (shared_data.ready) {
249 m_siteDensity = shared_data.density;
250 }
251
252 if (m_indices.size() != m_cov.size()) {
253 // object is not set up correctly (setSpecies needs to be run)
254 m_acov = NAN;
255 m_ecov = NAN;
256 m_mcov = NAN;
257 return;
258 }
259 m_acov = 0.0;
260 m_ecov = 0.0;
261 m_mcov = 0.0;
262 for (auto& [iCov, iKin] : m_indices) {
263 m_acov += m_ac[iCov] * shared_data.coverages[iKin];
264 if (m_lindep[iCov]) {
265 m_ecov += m_ec[iCov][1] * shared_data.coverages[iKin];
266 } else {
267 m_ecov += poly4(shared_data.coverages[iKin], m_ec[iCov].data());
268 }
269 m_mcov += m_mc[iCov] * shared_data.logCoverages[iKin];
270 }
271
272 // Update change in electrical potential energy
273 if (m_chargeTransfer) {
274 m_deltaPotential_RT = 0.;
275 for (const auto& [iPhase, netCharge] : m_netCharges) {
276 m_deltaPotential_RT +=
277 shared_data.electricPotentials[iPhase] * netCharge;
278 }
279 m_deltaPotential_RT /= GasConstant * shared_data.temperature;
280 }
281
282 // Update quantities used for exchange current density formulation
283 if (m_exchangeCurrentDensityFormulation) {
284 m_deltaGibbs0_RT = 0.;
285 m_prodStandardConcentrations = 1.;
286 for (const auto& [k, stoich] : m_stoichCoeffs) {
287 m_deltaGibbs0_RT +=
288 shared_data.standardChemPotentials[k] * stoich;
289 if (stoich > 0.) {
290 m_prodStandardConcentrations *=
291 shared_data.standardConcentrations[k];
292 }
293 }
294 m_deltaGibbs0_RT /= GasConstant * shared_data.temperature;
295 }
296}

298void InterfaceRateBase::setContext(const Reaction& rxn, const Kinetics& kin)
299{
300 setSpecies(kin.thermo().speciesNames());
301
302 m_chargeTransfer = rxn.usesElectrochemistry(kin);
303 if (!m_chargeTransfer) {
304 return;
305 }
306
307 m_stoichCoeffs.clear();
308 for (const auto& [name, stoich] : rxn.reactants) {
309 m_stoichCoeffs.emplace_back(kin.kineticsSpeciesIndex(name), -stoich);
310 }
311 for (const auto& [name, stoich] : rxn.products) {
312 m_stoichCoeffs.emplace_back(kin.kineticsSpeciesIndex(name), stoich);
313 }
314
315 m_netCharges.clear();
316 for (const auto& [k, stoich] : m_stoichCoeffs) {
317 size_t n = kin.speciesPhaseIndex(k);
318 size_t start = kin.kineticsSpeciesIndex(0, n);
319 double charge = kin.thermo(n).charge(k - start);
320 m_netCharges.emplace_back(n, Faraday * charge * stoich);
321 }
322}

353void StickingCoverage::setContext(const Reaction& rxn, const Kinetics& kin)
354{
355 // Ensure that site density is initialized
356 const ThermoPhase& phase = kin.thermo(kin.reactionPhaseIndex());
357 const auto& surf = dynamic_cast<const SurfPhase&>(phase);
358 m_siteDensity = surf.siteDensity();
359 if (!m_explicitMotzWise) {
360 m_motzWise = kin.thermo().input().getBool("Motz-Wise", false);
361 }
362
363 // Identify the interface phase
364 size_t iInterface = kin.reactionPhaseIndex();
365
366 string sticking_species = m_stickingSpecies;
367 if (sticking_species == "") {
368 // Identify the sticking species if not explicitly given
369 vector<string> gasSpecies;
370 vector<string> anySpecies;
371 for (const auto& [name, stoich] : rxn.reactants) {
372 size_t iPhase = kin.speciesPhaseIndex(kin.kineticsSpeciesIndex(name));
373 if (iPhase != iInterface) {
374 // Non-interface species. There should be exactly one of these
375 // (either in gas phase or other phase)
376 if (kin.thermo(iPhase).phaseOfMatter() == "gas") {
377 gasSpecies.push_back(name);
378 }
379 anySpecies.push_back(name);
380 }
381 }
382 if (gasSpecies.size() == 1) {
383 // single sticking species in gas phase
384 sticking_species = gasSpecies[0];
385 } else if (anySpecies.size() == 1) {
386 // single sticking species in any phase
387 sticking_species = anySpecies[0];
388 } else if (anySpecies.size() == 0) {
389 throw InputFileError("StickingCoverage::setContext",
390 rxn.input, "No non-interface species found "
391 "in sticking reaction: '{}'", rxn.equation());
392 } else {
393 throw InputFileError("StickingCoverage::setContext",
394 rxn.input, "Multiple non-interface species ({})\nfound in sticking "
395 "reaction: '{}'.\nSticking species must be explicitly specified.",
396 fmt::format("'{}'", fmt::join(anySpecies, "', '")), rxn.equation());
397 }
398 }
399 m_stickingSpecies = sticking_species;
400
401 double surface_order = 0.0;
402 double multiplier = 1.0;
403 // Adjust the A-factor
404 for (const auto& [name, stoich] : rxn.reactants) {
405 size_t iPhase = kin.speciesPhaseIndex(kin.kineticsSpeciesIndex(name));
406 const ThermoPhase& p = kin.thermo(iPhase);
407 size_t k = p.speciesIndex(name);
408 if (name == sticking_species) {
409 multiplier *= sqrt(GasConstant / (2 * Pi * p.molecularWeight(k)));
410 } else {
411 // Non-sticking species. Convert from coverages used in the
412 // sticking probability expression to the concentration units
413 // used in the mass action rate expression. For surface phases,
414 // the dependence on the site density is incorporated when the
415 // rate constant is evaluated, since we don't assume that the
416 // site density is known at this time.
417 double order = getValue(rxn.orders, name, stoich);
418 if (&p == &surf) {
419 multiplier *= pow(surf.size(k), order);
420 surface_order += order;
421 } else {
422 multiplier *= pow(p.standardConcentration(k), -order);
423 }
424 }
425 }
426 m_surfaceOrder = surface_order;
427 m_multiplier = multiplier;
428}
