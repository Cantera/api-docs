18/**
19 * Virtual base class for ODE/DAE right-hand-side function evaluators.
20 * Classes derived from FuncEval evaluate the right-hand-side function
21 * \f$ \vec{F}(t,\vec{y})\f$ in
22 * \f[
23 * \dot{\vec{y}} = \vec{F}(t,\vec{y}).
24 * \f]
25 * @ingroup odeGroup
26 */
27class FuncEval
28{
29public:
30 FuncEval() = default;
31 virtual ~FuncEval() = default;
32
33 /**
34 * Evaluate the right-hand-side ODE function. Called by the integrator.
35 * @param[in] t time.

37 * @param[out] ydot rate of change of solution vector, length neq()
38 * @param[in] p sensitivity parameter vector, length nparams()
39 */
40 virtual void eval(double t, double* y, double* ydot, double* p) {
41 throw NotImplementedError("FuncEval::eval");
42 }

52 virtual void evalDae(double t, double* y, double* ydot, double* p,

63 //! Evaluate the right-hand side using return code to indicate status.
64 /*!
65 * Errors are indicated using the return value, rather than by throwing
66 * exceptions. This method is used when calling from a C-based integrator
67 * such as CVODES. Exceptions may either be stored or printed, based on the
68 * setting of suppressErrors().
69 * @returns 0 for a successful evaluation; 1 after a potentially-
70 * recoverable error; -1 after an unrecoverable error.
71 */
72 int evalNoThrow(double t, double* y, double* ydot);

74 //! Evaluate the right-hand side using return code to indicate status.
