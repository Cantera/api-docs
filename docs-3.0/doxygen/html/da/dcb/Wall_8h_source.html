133//! Represents a wall between between two ReactorBase objects.
134/*!
135 * Walls can move (changing the volume of the adjacent reactors) and allow heat
136 * transfer between reactors.
137 */
138class Wall : public WallBase
139{
140public:
141 Wall() = default;
142
143 //! String indicating the wall model implemented. Usually
144 //! corresponds to the name of the derived class.
145 virtual std::string type() const {
146 return "Wall";
147 }
148
149 //! Wall velocity \f$ v(t) \f$ at current reactor network time.
150 //! @since New in Cantera 3.0.
151 double velocity() const;
152
153 //! Set the wall velocity to a specified function of time, \f$ v(t) \f$.
154 void setVelocity(Func1* f=0) {
155 if (f) {
156 m_vf = f;
157 }
158 }
159
160 //! Rate of volume change (m^3/s) for the adjacent reactors.
161 /*!
162 * The volume rate of change is given by
163 * \f[
164 * \dot V = K A (P_{left} - P_{right}) + F(t)
165 * \f]
166 * where *K* is the specified expansion rate coefficient, *A* is the wall
167 * area, and *F(t)* is a specified function of time. Positive values for
168 * `vdot` correspond to increases in the volume of reactor on left, and
169 * decreases in the volume of the reactor on the right.
170 * @deprecated Still used by traditional MATLAB toolbox; replaceable by
171 * expansionRate.
172 */
173 virtual double vdot(double t);
174
175 //! Rate of volume change (m^3/s) for the adjacent reactors.
176 /*!
177 * The volume rate of change is given by
178 * \f[
179 * \dot V = K A (P_{left} - P_{right}) + F(t)
180 * \f]
181 * where *K* is the specified expansion rate coefficient, *A* is the wall area,
182 * and and *F(t)* is a specified function evaluated at the current network time.
183 * Positive values for `expansionRate` correspond to increases in the volume of
184 * reactor on left, and decreases in the volume of the reactor on the right.
185 * @since New in Cantera 3.0.
186 */
187 virtual double expansionRate();
188
189 //! Heat flux function \f$ q_0(t) \f$ evaluated at current reactor network time.
190 //! @since New in Cantera 3.0.
191 double heatFlux() const;
192
193 //! Specify the heat flux function \f$ q_0(t) \f$.
194 void setHeatFlux(Func1* q) {
195 m_qf = q;
196 }
197
198 //! Heat flow rate through the wall (W).
199 /*!
200 * The heat flux is given by
201 * \f[
202 * Q = h A (T_{left} - T_{right}) + A G(t)
203 * \f]
204 * where *h* is the heat transfer coefficient, *A* is the wall area, and
205 * *G(t)* is a specified function of time. Positive values denote a flux
206 * from left to right.
207 * @deprecated Still used by traditional MATLAB toolbox; replaceable by heatRate.
208 */
209 virtual double Q(double t);
210
211 //! Heat flow rate through the wall (W).
212 /*!
213 * The heat flux is given by
214 * \f[
215 * Q = h A (T_{left} - T_{right}) + A G(t)
216 * \f]
217 * where *h* is the heat transfer coefficient, *A* is the wall area, and
218 * *G(t)* is a specified function of time evaluated at the current network
219 * time. Positive values denote a flux from left to right.
220 * @since New in Cantera 3.0.
221 */
222 virtual double heatRate();
223
224 void setThermalResistance(double Rth) {
225 m_rrth = 1.0/Rth;
226 }
227
228 //! Set the overall heat transfer coefficient [W/m^2/K].
229 void setHeatTransferCoeff(double U) {
230 m_rrth = U;
231 }
232
233 //! Get the overall heat transfer coefficient [W/m^2/K].
234 double getHeatTransferCoeff() const {
235 return m_rrth;
236 }
237
238 //! Set the emissivity.
239 void setEmissivity(double epsilon) {
240 if (epsilon > 1.0 || epsilon < 0.0) {
241 throw CanteraError("WallBase::setEmissivity",
242 "emissivity must be between 0.0 and 1.0");
243 }
244 m_emiss = epsilon;
245 }
246
247 //! Get the emissivity.
248 double getEmissivity() const {
249 return m_emiss;
250 }
251
252 //! Set the expansion rate coefficient.
253 void setExpansionRateCoeff(double k) {
254 m_k = k;
255 }
256
257 //! Get the expansion rate coefficient
258 double getExpansionRateCoeff() const {
259 return m_k;
260 }
261
262protected:
263
264 //! expansion rate coefficient
265 double m_k = 0.0;
266
267 //! heat transfer coefficient
268 double m_rrth = 0.0;
269
270 //! emissivity
271 double m_emiss = 0.0;
272
273 //! Velocity function
274 Func1* m_vf = nullptr;
275
276 //! Heat flux function
277 Func1* m_qf = nullptr;
278};
