34void CoverageDependentSurfPhase::addInterpolativeDependency(
35 const InterpolativeDependency& int_deps)
36{
37 if (int_deps.enthalpy_map.begin()->first != 0.0) {
38 throw CanteraError("CoverageDependentSurfPhase::addInterpolativeDependency",
39 "The first element of enthalpy-coverages array must be 0.0.");
40 }
41 if (int_deps.enthalpy_map.rbegin()->first != 1.0) {
42 throw CanteraError("CoverageDependentSurfPhase::addInterpolativeDependency",
43 "The last element of enthalpy-coverages array must be 1.0.");
44 }
45
46 if (int_deps.entropy_map.begin()->first != 0.0) {
47 throw CanteraError("CoverageDependentSurfPhase::addInterpolativeDependency",
48 "The first element of entropy-coverages array must be 0.0.");
49 }
50 if (int_deps.entropy_map.rbegin()->first != 1.0) {
51 throw CanteraError("CoverageDependentSurfPhase::addInterpolativeDependency",
52 "The last element of entropy-coverages array must be 1.0.");
53 }
54
55 m_InterpolativeDependency.push_back(int_deps);
56}

58void CoverageDependentSurfPhase::initThermo()
59{
60 SurfPhase::initThermo();
61 if (m_input.hasKey("reference-state-coverage")) {
62 m_theta_ref = m_input["reference-state-coverage"].as<double>();
63 if (m_theta_ref <= 0.0 || m_theta_ref > 1.0) {
64 throw InputFileError("CoverageDependentSurfPhase::initThermo",
65 m_input, "Reference state coverage must be greater than 0.0 and \
66 less than or equal to 1.0.");
67 }
68 }
69 for (auto& item : m_species) {
70 // Read enthalpy and entropy dependencies from species 'input' information
71 // (i.e. as specified in a YAML input file) for both self- and cross-
72 // interactions.
73 if (item.second->input.hasKey("coverage-dependencies")) {
74 auto& cov_map = item.second->input["coverage-dependencies"];
75 for (const auto& item2 : cov_map) {
76 size_t k = speciesIndex(item.first);
77 size_t j = speciesIndex(item2.first);
78 if (k == npos) {
79 throw InputFileError("CoverageDependentSurfPhase::initThermo",
80 item.second->input, "Unknown species '{}'.", item.first);
81 }
82 if (j == npos) {
83 throw InputFileError("CoverageDependentSurfPhase::initThermo",
84 item.second->input, "Unknown species '{}'.", item2.first);
85 }
86 auto& dep_map = item2.second.as<AnyMap>();
87 // For linear model and polynomial model
88 if (dep_map["model"] == "linear" || dep_map["model"] == "polynomial") {
89 PolynomialDependency poly_deps(k, j, dep_map);
90 m_PolynomialDependency.push_back(poly_deps);
91 // For piecewise-linear model and interpolative model
92 } else if (dep_map["model"] == "piecewise-linear" ||
93 dep_map["model"] == "interpolative") {
94 InterpolativeDependency int_deps(k, j, dep_map,
95 item.second->input);
96 addInterpolativeDependency(int_deps);
97 } else {
98 throw InputFileError("CoverageDependentSurfPhase::initThermo",
99 item.second->input, "Unrecognized coverage dependency model. \
100 Model must be 'linear', 'piecewise-linear', 'polynomial', \
101 or 'interpolative'.");
102 }
103 // For coverage-dependent heat capacity parameters, if present
104 if (dep_map.hasKey("heat-capacity-a")) {
105 HeatCapacityDependency cpcov_deps(k, j);
106 cpcov_deps.coeff_a = dep_map.convert("heat-capacity-a", "J/kmol/K");
107 cpcov_deps.coeff_b = dep_map.convert("heat-capacity-b", "J/kmol/K");
108
109 m_HeatCapacityDependency.push_back(cpcov_deps);
110 }
111 }
112 }
113 }

139void CoverageDependentSurfPhase::getSpeciesParameters(const std::string& name,
140 AnyMap& speciesNode) const
141{
142 SurfPhase::getSpeciesParameters(name, speciesNode);
143 size_t k = speciesIndex(name);
144 // Get linear and polynomial model parameters from PolynomialDependency vector
145 for (auto& item : m_PolynomialDependency) {
146 if (item.k == k) {
147 if (item.isLinear) {
148 auto& covdepNode =
149 speciesNode["coverage-dependencies"][speciesName(item.j)]
150 .getMapWhere("model", "linear", true);
151 covdepNode["enthalpy"].setQuantity(item.enthalpy_coeffs[1], "J/kmol");
152 covdepNode["entropy"].setQuantity(item.entropy_coeffs[1], "J/kmol/K");
153 } else {
154 auto& covdepNode =
155 speciesNode["coverage-dependencies"][speciesName(item.j)]
156 .getMapWhere("model", "polynomial", true);
157 vector_fp hvec (
158 item.enthalpy_coeffs.begin() + 1, item.enthalpy_coeffs.end());
159 covdepNode["enthalpy-coefficients"].setQuantity(hvec, "J/kmol");
160 vector_fp svec (
161 item.entropy_coeffs.begin() + 1, item.entropy_coeffs.end());
162 covdepNode["entropy-coefficients"].setQuantity(svec, "J/kmol/K");
163 }
164 }
165 }
166 // Get piecewise-linear model parameters from InterpolativeDependency vector
167 for (auto& item : m_InterpolativeDependency) {
168 if (item.k == k) {
169 if (item.isPiecewise) {
170 auto& covdepNode =
171 speciesNode["coverage-dependencies"][speciesName(item.j)]
172 .getMapWhere("model", "piecewise-linear", true);
173 vector_fp hcovs, enthalpies, scovs, entropies;
174 for (const auto& hmap : item.enthalpy_map) {
175 hcovs.push_back(hmap.first);
176 enthalpies.push_back(hmap.second);
177 }
178 for (const auto& smap : item.entropy_map) {
179 scovs.push_back(smap.first);
180 entropies.push_back(smap.second);
181 }
182 covdepNode["enthalpy-change"] = hcovs[1];
183 covdepNode["enthalpy-low"].setQuantity(
184 (enthalpies[1] - enthalpies[0]) / (hcovs[1] - hcovs[0]), "J/kmol");
185 covdepNode["enthalpy-high"].setQuantity(
186 (enthalpies[2] - enthalpies[1]) / (hcovs[2] - hcovs[1]), "J/kmol");
187 covdepNode["entropy-change"] = scovs[1];
188 covdepNode["entropy-low"].setQuantity(
189 (entropies[1] - entropies[0]) / (scovs[1] - scovs[0]), "J/kmol/K");
190 covdepNode["entropy-high"].setQuantity(
191 (entropies[2] - entropies[1]) / (scovs[2] - scovs[1]), "J/kmol/K");
192 } else {
193 auto& covdepNode =
194 speciesNode["coverage-dependencies"][speciesName(item.j)]
195 .getMapWhere("model", "interpolative", true);
196 vector_fp hcovs, enthalpies, scovs, entropies;
197 for (const auto& hmap : item.enthalpy_map) {
198 hcovs.push_back(hmap.first);
199 enthalpies.push_back(hmap.second);
200 }
201 for (const auto& smap : item.entropy_map) {
202 scovs.push_back(smap.first);
203 entropies.push_back(smap.second);
204 }
205 covdepNode["enthalpy-coverages"] = std::move(hcovs);
206 covdepNode["enthalpies"].setQuantity(enthalpies, "J/kmol");
207 covdepNode["entropy-coverages"] = std::move(scovs);
208 covdepNode["entropies"].setQuantity(entropies, "J/kmol/K");
209 }
210 }
211 }
212 // Get heat capacity model parameters from HeatCapacityDependency vector
213 for (auto& item : m_HeatCapacityDependency) {
214 if (item.k == k) {
215 auto& covdepNode =
216 speciesNode["coverage-dependencies"][speciesName(item.j)]
217 .getMapWhere("heat-capacity-a", "", true);
218 covdepNode["heat-capacity-a"].setQuantity(item.coeff_a, "J/kmol/K");
219 covdepNode["heat-capacity-b"].setQuantity(item.coeff_b, "J/kmol/K");
220 }
221 }
222}

340double CoverageDependentSurfPhase::entropy_mole() const
341{
342 _updateTotalThermo();
343 double entropy = 0.0;
344 for (size_t k = 0; k < m_kk; k++) {
345 entropy += moleFraction(k) * (m_entropy[k] -
346 GasConstant * log(std::max(m_cov[k], SmallNumber) / m_theta_ref));
347 }
348 return entropy;

357void CoverageDependentSurfPhase::_updateCovDepThermo() const
358{
359 int stateNumnow = stateMFNumber();
360 double tnow = temperature();
361 if (m_stateNumlast != stateNumnow || m_tlast != tnow) {
362 for (size_t k = 0; k < m_kk; k++) {
363 m_h_cov[k] = 0.0;
364 m_s_cov[k] = 0.0;
365 m_cp_cov[k] = 0.0;
366 }
367 getCoverages(m_cov.data());
368
369 // For linear and polynomial model
370 for (auto& item : m_PolynomialDependency) {
371 m_h_cov[item.k] += poly4(m_cov[item.j], item.enthalpy_coeffs.data());
372 m_s_cov[item.k] += poly4(m_cov[item.j], item.entropy_coeffs.data());
373 }
374
375 // For piecewise-linear and interpolative model
376 for (auto& item : m_InterpolativeDependency) {
377 auto h_iter = item.enthalpy_map.upper_bound(m_cov[item.j]);
378 auto s_iter = item.entropy_map.upper_bound(m_cov[item.j]);
379 AssertThrowMsg(h_iter != item.enthalpy_map.end(),
380 "CoverageDependentSurfPhase::_updateCovDepThermo",
381 "Coverage out of range: {}", m_cov[item.j]);
382 AssertThrowMsg(h_iter != item.enthalpy_map.begin(),
383 "CoverageDependentSurfPhase::_updateCovDepThermo",
384 "Coverage out of range: {}", m_cov[item.j]);
385 AssertThrowMsg(s_iter != item.entropy_map.end(),
386 "CoverageDependentSurfPhase::_updateCovDepThermo",
387 "Coverage out of range: {}", m_cov[item.j]);
388 AssertThrowMsg(s_iter != item.entropy_map.begin(),
389 "CoverageDependentSurfPhase::_updateCovDepThermo",
390 "Coverage out of range: {}", m_cov[item.j]);
391
392 double highHcov = h_iter->first;
393 double highH = h_iter->second;
394 double lowHcov = (--h_iter)->first;
395 double lowH = h_iter->second;
396
397 double highScov = s_iter->first;
398 double highS = s_iter->second;
399 double lowScov = (--s_iter)->first;
400 double lowS = s_iter->second;
401
402 m_h_cov[item.k] += (highH - lowH) / (highHcov - lowHcov)
403 * (m_cov[item.j] - lowHcov) + lowH;
404
405 m_s_cov[item.k] += (highS - lowS) / (highScov - lowScov)
406 * (m_cov[item.j] - lowScov) + lowS;
407 }
408
409 // For coverage-dependent heat capacity
410 for (auto& item : m_HeatCapacityDependency) {
411 double a = item.coeff_a;
412 double b = item.coeff_b;
413 m_cp_cov[item.k] += (a * log(tnow) + b) * m_cov[item.j] * m_cov[item.j];
414 double int_cp_tnow = tnow * (a * log(tnow) - a + b);
415 double int_cp_298 = 298.15 * (a * log(298.15) - a + b);
416 m_h_cov[item.k] += (int_cp_tnow - int_cp_298) * m_cov[item.j]
417 * m_cov[item.j];
418 double int_cp_T_tnow = log(tnow) * (a * log(tnow) + 2 * b);
419 double int_cp_T_298 = log(298.15) * (a * log(298.15) + 2 * b);
420 m_s_cov[item.k] += 0.5 * (int_cp_T_tnow - int_cp_T_298) * m_cov[item.j]
421 * m_cov[item.j];
422 }
423
424 for (size_t k = 0; k < m_kk; k++) {
425 m_mu_cov[k] = m_h_cov[k] - tnow * m_s_cov[k];
426 }
427 m_stateNumlast = stateNumnow;
428 }
429}

431void CoverageDependentSurfPhase::_updateTotalThermo() const
432{
433 _updateCovDepThermo();
434 SurfPhase::_updateThermo();
435
436 for (size_t k = 0; k < m_kk; k++) {
437 m_enthalpy[k] = m_h0[k] + m_h_cov[k];
438 m_entropy[k] = m_s0[k] + m_s_cov[k];
439 m_heatcapacity[k] = m_cp0[k] + m_cp_cov[k];
440 m_chempot[k] = m_mu0[k] + m_mu_cov[k];
441 }
442}
