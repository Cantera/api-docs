24/**
25 * The base class for boundaries between one-dimensional spatial domains. The
26 * boundary may have its own internal variables, such as surface species
27 * coverages.
28 *
29 * The boundary types are an inlet, an outlet, a symmetry plane, and a surface.
30 *
31 * The public methods are all virtual, and the base class implementations throw
32 * exceptions.
33 * @ingroup onedim
34 */
35class Boundary1D : public Domain1D
36{
37public:
38 Boundary1D();
39
40 virtual void init() {
41 _init(1);
42 }
43
44 virtual string type() const {
45 return "boundary";
46 }
47
48 virtual bool isConnector() {
49 return true;
50 }
51
52 //! Set the temperature.
53 virtual void setTemperature(double t) {
54 m_temp = t;
55 }
56
57 //! Temperature [K].
58 virtual double temperature() {
59 return m_temp;
60 }
61
62 virtual size_t nSpecies() {
63 return 0;
64 }
65
66 //! Set the mole fractions by specifying a std::string.
67 virtual void setMoleFractions(const std::string& xin) {
68 throw NotImplementedError("Boundary1D::setMoleFractions");
69 }
70
71 //! Set the mole fractions by specifying an array.
72 virtual void setMoleFractions(const double* xin) {
73 throw NotImplementedError("Boundary1D::setMoleFractions");
74 }
75
76 //! Mass fraction of species k.
77 virtual double massFraction(size_t k) {
78 throw NotImplementedError("Boundary1D::massFraction");
79 }
80
81 //! Set the total mass flow rate.
82 virtual void setMdot(double mdot) {
83 m_mdot = mdot;
84 }
85
86 //! Set tangential velocity gradient [1/s] at this boundary.
87 virtual void setSpreadRate(double V0) {
88 throw NotImplementedError("Boundary1D::setSpreadRate");
89 }
90
91 //! Tangential velocity gradient [1/s] at this boundary.
92 virtual double spreadRate() {
93 throw NotImplementedError("Boundary1D::spreadRate");
94 }
95
96 //! The total mass flow rate [kg/m2/s].
97 virtual double mdot() {
98 return m_mdot;
99 }
100
101 virtual void setupGrid(size_t n, const double* z) {}
102
103protected:
104 void _init(size_t n);
105
106 StFlow* m_flow_left = nullptr;
107 StFlow* m_flow_right = nullptr;
108 size_t m_ilr = 0;
109 size_t m_left_nv = 0;
110 size_t m_right_nv = 0;
111 size_t m_left_loc = 0;
112 size_t m_right_loc = 0;
113 size_t m_left_points = 0;
114 size_t m_left_nsp = 0;
115 size_t m_right_nsp = 0;
116 size_t m_sp_left = 0;
117 size_t m_sp_right = 0;
118 size_t m_start_left = 0;
119 size_t m_start_right = 0;
120 ThermoPhase* m_phase_left = nullptr;
121 ThermoPhase* m_phase_right = nullptr;
122 double m_temp = 0.0;
123 double m_mdot = 0.0;
124};
