21void Boundary1D::_init(size_t n)
22{
23 if (m_index == npos) {
24 throw CanteraError("Boundary1D::_init",
25 "install in container before calling init.");
26 }
27
28 // A boundary object contains only one grid point
29 resize(n,1);
30
31 m_left_nsp = 0;
32 m_right_nsp = 0;
33
34 // check for left and right flow objects
35 if (m_index > 0) {
36 Domain1D& r = container().domain(m_index-1);
37 if (!r.isConnector()) { // multi-point domain
38 m_left_nv = r.nComponents();
39 if (m_left_nv > c_offset_Y) {
40 m_left_nsp = m_left_nv - c_offset_Y;
41 } else {
42 m_left_nsp = 0;
43 }
44 m_left_loc = container().start(m_index-1);
45 m_left_points = r.nPoints();
46 m_flow_left = dynamic_cast<StFlow*>(&r);
47 if (m_flow_left != nullptr) {
48 m_phase_left = &m_flow_left->phase();
49 }
50 } else {
51 throw CanteraError("Boundary1D::_init",
52 "Boundary domains can only be connected on the left to flow "
53 "domains, not '{}' domains.", r.type());
54 }
55 }
56
57 // if this is not the last domain, see what is connected on the right
58 if (m_index + 1 < container().nDomains()) {
59 Domain1D& r = container().domain(m_index+1);
60 if (!r.isConnector()) { // multi-point domain
61 m_right_nv = r.nComponents();
62 if (m_right_nv > c_offset_Y) {
63 m_right_nsp = m_right_nv - c_offset_Y;
64 } else {
65 m_right_nsp = 0;
66 }
67 m_right_loc = container().start(m_index+1);
68 m_flow_right = dynamic_cast<StFlow*>(&r);
69 if (m_flow_right != nullptr) {
70 m_phase_right = &m_flow_right->phase();
71 }
72 } else {
73 throw CanteraError("Boundary1D::_init",
74 "Boundary domains can only be connected on the right to flow "
75 "domains, not '{}' domains.", r.type());
76 }
77 }
78}

138void Inlet1D::init()
139{
140 _init(0);
141
142 // if a flow domain is present on the left, then this must be a right inlet.
143 // Note that an inlet object can only be a terminal object - it cannot have
144 // flows on both the left and right
145 if (m_flow_left) {
146 m_ilr = RightInlet;
147 m_flow = m_flow_left;
148 } else if (m_flow_right) {
149 m_ilr = LeftInlet;
150 m_flow = m_flow_right;
151 } else {
152 throw CanteraError("Inlet1D::init","no flow!");
153 }
154
155 // components = u, V, T, lambda, + mass fractions
156 m_nsp = m_flow->phase().nSpecies();
157 m_yin.resize(m_nsp, 0.0);
158 if (m_xstr != "") {
159 setMoleFractions(m_xstr);
160 } else {
161 m_yin[0] = 1.0;
162 }
163}

165void Inlet1D::eval(size_t jg, double* xg, double* rg,
166 integer* diagg, double rdt)
167{
168 if (jg != npos && (jg + 2 < firstPoint() || jg > lastPoint() + 2)) {
169 return;
170 }
171
172 if (m_ilr == LeftInlet) {
173 // Array elements corresponding to the first point of the flow domain
174 double* xb = xg + m_flow->loc();
175 double* rb = rg + m_flow->loc();
176
177 // The first flow residual is for u. This, however, is not modified by
178 // the inlet, since this is set within the flow domain from the
179 // continuity equation.
180
181 // spreading rate. The flow domain sets this to V(0),
182 // so for finite spreading rate subtract m_V0.
183 rb[c_offset_V] -= m_V0;
184
185 if (m_flow->doEnergy(0)) {
186 // The third flow residual is for T, where it is set to T(0). Subtract
187 // the local temperature to hold the flow T to the inlet T.
188 rb[c_offset_T] -= m_temp;
189 }
190
191 if (m_flow->fixed_mdot()) {
192 // The flow domain sets this to -rho*u. Add mdot to specify the mass
193 // flow rate.
194 rb[c_offset_L] += m_mdot;
195 } else {
196 // if the flow is a freely-propagating flame, mdot is not specified.
197 // Set mdot equal to rho*u, and also set lambda to zero.
198 m_mdot = m_flow->density(0)*xb[0];
199 rb[c_offset_L] = xb[c_offset_L];
200 }
201
202 // add the convective term to the species residual equations
203 for (size_t k = 0; k < m_nsp; k++) {
204 if (k != m_flow_right->leftExcessSpecies()) {
205 rb[c_offset_Y+k] += m_mdot*m_yin[k];
206 }
207 }
208
209 } else {
210 // right inlet
211 // Array elements corresponding to the last point in the flow domain
212 double* rb = rg + loc() - m_flow->nComponents();
213 rb[c_offset_V] -= m_V0;
214 if (m_flow->doEnergy(m_flow->nPoints() - 1)) {
215 rb[c_offset_T] -= m_temp; // T
216 }
217 rb[c_offset_U] += m_mdot; // u
218 for (size_t k = 0; k < m_nsp; k++) {
219 if (k != m_flow_left->rightExcessSpecies()) {
220 rb[c_offset_Y+k] += m_mdot * m_yin[k];
221 }
222 }
223 }

285void Symm1D::eval(size_t jg, double* xg, double* rg, integer* diagg,
286 double rdt)
287{
288 if (jg != npos && (jg + 2< firstPoint() || jg > lastPoint() + 2)) {
289 return;
290 }
291
292 // start of local part of global arrays
293 double* x = xg + loc();
294 double* r = rg + loc();
295 integer* diag = diagg + loc();
296
297 if (m_flow_right) {
298 size_t nc = m_flow_right->nComponents();
299 double* xb = x;
300 double* rb = r;
301 int* db = diag;
302 db[c_offset_V] = 0;
303 db[c_offset_T] = 0;
304 rb[c_offset_V] = xb[c_offset_V] - xb[c_offset_V + nc]; // zero dV/dz
305 if (m_flow_right->doEnergy(0)) {
306 rb[c_offset_T] = xb[c_offset_T] - xb[c_offset_T + nc]; // zero dT/dz
307 }
308 }
309
310 if (m_flow_left) {
311 size_t nc = m_flow_left->nComponents();
312 double* xb = x - nc;
313 double* rb = r - nc;
314 int* db = diag - nc;
315 db[c_offset_V] = 0;
316 db[c_offset_T] = 0;
317 rb[c_offset_V] = xb[c_offset_V] - xb[c_offset_V - nc]; // zero dV/dz
318 if (m_flow_left->doEnergy(m_flow_left->nPoints() - 1)) {
319 rb[c_offset_T] = xb[c_offset_T] - xb[c_offset_T - nc]; // zero dT/dz
320 }
321 }

382 size_t nc = m_flow_left->nComponents();
383 double* xb = x - nc;
384 double* rb = r - nc;
385 int* db = diag - nc;
386
387 // zero Lambda
388 if (m_flow_left->fixed_mdot()) {
389 rb[c_offset_U] = xb[c_offset_L];
390 }
391
392 if (m_flow_left->doEnergy(m_flow_left->nPoints()-1)) {
393 rb[c_offset_T] = xb[c_offset_T] - xb[c_offset_T - nc]; // zero T gradient
394 }
395 size_t kSkip = c_offset_Y + m_flow_left->rightExcessSpecies();
396 for (size_t k = c_offset_Y; k < nc; k++) {
397 if (k != kSkip) {
398 rb[k] = xb[k] - xb[k - nc]; // zero mass fraction gradient
399 db[k] = 0;
400 }
401 }
402 }

678void ReactingSurf1D::eval(size_t jg, double* xg, double* rg,
679 integer* diagg, double rdt)
680{
681 if (jg != npos && (jg + 2 < firstPoint() || jg > lastPoint() + 2)) {
682 return;
683 }
684
685 // start of local part of global arrays
686 double* x = xg + loc();
687 double* r = rg + loc();
688 integer* diag = diagg + loc();
689
690 // set the coverages
691 double sum = 0.0;
692 for (size_t k = 0; k < m_nsp; k++) {
693 m_work[k] = x[k];
694 sum += x[k];
695 }
696 m_sphase->setTemperature(m_temp);
697 m_sphase->setCoveragesNoNorm(m_work.data());
698
699 // set the left gas state to the adjacent point
700
701 size_t leftloc = 0, rightloc = 0;
702 size_t pnt = 0;
703
704 if (m_flow_left) {
705 leftloc = m_flow_left->loc();
706 pnt = m_flow_left->nPoints() - 1;
707 m_flow_left->setGas(xg + leftloc, pnt);
708 }
709
710 if (m_flow_right) {
711 rightloc = m_flow_right->loc();
712 m_flow_right->setGas(xg + rightloc, 0);
713 }
714
715 m_kin->getNetProductionRates(m_work.data());
716 double rs0 = 1.0/m_sphase->siteDensity();
717 size_t ioffset = m_kin->kineticsSpeciesIndex(0, m_surfindex);
718
719 if (m_enabled) {
720 for (size_t k = 0; k < m_nsp; k++) {
721 r[k] = m_work[k + ioffset] * m_sphase->size(k) * rs0;
722 r[k] -= rdt*(x[k] - prevSoln(k,0));
723 diag[k] = 1;
724 }
725 r[0] = 1.0 - sum;
726 diag[0] = 0;
727 } else {
728 for (size_t k = 0; k < m_nsp; k++) {
729 r[k] = x[k] - m_fixed_cov[k];
730 diag[k] = 0;
731 }
732 }
733
734 if (m_flow_right) {
735 double* rb = r + m_nsp;
736 double* xb = x + m_nsp;
737 rb[c_offset_T] = xb[c_offset_T] - m_temp; // specified T
738 }
739 if (m_flow_left) {
740 size_t nc = m_flow_left->nComponents();
741 const vector_fp& mwleft = m_phase_left->molecularWeights();
742 double* rb = r - nc;
743 double* xb = x - nc;
744 rb[c_offset_T] = xb[c_offset_T] - m_temp; // specified T
745 size_t nSkip = m_flow_left->rightExcessSpecies();
746 size_t l_offset = 0;
747 ThermoPhase* left_thermo = &m_flow_left->phase();
748 for (size_t nth = 0; nth < m_kin->nPhases(); nth++) {
749 if (&m_kin->thermo(nth) == left_thermo) {
750 l_offset = m_kin->kineticsSpeciesIndex(0, nth);
751 break;
752 }
753 }
754 for (size_t nl = 0; nl < m_left_nsp; nl++) {
755 if (nl != nSkip) {
756 rb[c_offset_Y+nl] += m_work[nl + l_offset]*mwleft[nl];
757 }
758 }
759 }
760}

762shared_ptr<SolutionArray> ReactingSurf1D::asArray(const double* soln) const
763{
764 AnyMap meta = Boundary1D::getMeta();
765 meta["temperature"] = m_temp;
766 meta["phase"]["name"] = m_sphase->name();
767 AnyValue source = m_sphase->input().getMetadata("filename");
768 meta["phase"]["source"] = source.empty() ? "<unknown>" : source.asString();
769
770 // set state of surface phase
771 m_sphase->setState_TP(m_temp, m_sphase->pressure());
772 m_sphase->setCoverages(soln);
773 vector_fp data(m_sphase->stateSize());
774 m_sphase->saveState(data.size(), &data[0]);
775
776 auto arr = SolutionArray::create(m_solution, 1, meta);
777 arr->setState(0, data);
778 return arr;
779}
