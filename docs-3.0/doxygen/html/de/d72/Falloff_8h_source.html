65 /**
66  * Base class for falloff rate calculators.
67  * Each instance of a subclass of FalloffRate calculates the falloff reaction rate
68  * based on specific implementations of the falloff function.
69  *
70  * The falloff function @f$ F(P_r, T) @f$ is implemented by FalloffRate specializations,
71  * and is defined so that the rate coefficient is
72  * @f[
73  * k = k_\infty \frac{P_r}{1 + P_r} F(P_r,T)
74  * @f]
75  *
76  * Here @f$ P_r @f$ is the reduced pressure, defined by
77  * @f[
78  * P_r = \frac{k_0 [M]}{k_\infty}.
79  * @f]
80  * @ingroup falloffGroup
81  */
82 class FalloffRate : public ReactionRate
83 {
84 public:
85  FalloffRate() = default;
86 
87  FalloffRate(const AnyMap& node, const UnitStack& rate_units={});
88 
89  /**
90  * Initialize. Must be called before any other method is invoked.
91  *
92  * @param c Vector of coefficients of the parameterization. The number and
93  * meaning of these coefficients is subclass-dependent.
94  *
95  * @deprecated To be removed after %Cantera 3.0; superseded by setFalloffCoeffs()
96  */
97  void init(const vector<double>& c);
98 
99  /**
100  * Set coefficients of the falloff parameterization.
101  *
102  * @param c Vector of coefficients of the parameterization. The number and
103  * meaning of these coefficients is subclass-dependent.
104  */
105  virtual void setFalloffCoeffs(const vector<double>& c);
106 
107  /**
108  * Retrieve coefficients of the falloff parameterization.
109  *
110  * @param c Vector of coefficients of the parameterization. The number and
111  * meaning of these coefficients is subclass-dependent.
112  */
113  virtual void getFalloffCoeffs(vector<double>& c) const;
114 
115  /**
116  * Update the temperature-dependent portions of the falloff function, if
117  * any, and store them in the 'work' array. If not overloaded, the default
118  * behavior is to do nothing.
119  * @param T Temperature [K].
120  * @param work storage space for intermediate results.
121  */
122  virtual void updateTemp(double T, double* work) const {}
123 
124  /**
125  * The falloff function.
126  *
127  * @param pr reduced pressure (dimensionless).
128  * @param work array of size workSize() containing cached
129  * temperature-dependent intermediate results from a prior call
130  * to updateTemp.
131  * @returns the value of the falloff function @f$ F @f$ defined above
132  */
133  virtual double F(double pr, const double* work) const {
134  return 1.0;
135  }
136 
137  //! Evaluate falloff function at current conditions
138  double evalF(double T, double conc3b) {
139  updateTemp(T, m_work.data());
140  double logT = std::log(T);
141  double recipT = 1. / T;
142  m_rc_low = m_lowRate.evalRate(logT, recipT);
143  m_rc_high = m_highRate.evalRate(logT, recipT);
144  double pr = conc3b * m_rc_low / (m_rc_high + SmallNumber);
145  return F(pr, m_work.data());
146  }
147 
148  //! The size of the work array required.
149  /**
150  * @deprecated To be removed after %Cantera 3.0; unused.
151  */
152  virtual size_t workSize() const {
153  warn_deprecated("FalloffRate::workSize",
154  "To be removed after Cantera 3.0; unused.");
155  return 0;
156  }
157 
158  virtual const string type() const override {
159  if (m_chemicallyActivated) {
160  return "chemically-activated";
161  }
162  return "falloff";
163  }
164 
165  //! Returns the number of parameters used by this parameterization. The
166  //! values of these parameters can be obtained from getParameters().
167  virtual size_t nParameters() const {
168  return 0;
169  }
170 
171  virtual void setParameters(
172  const AnyMap& node, const UnitStack& rate_units) override;
173 
174  //! Get the values of the parameters for this object. *params* must be an
175  //! array of at least nParameters() elements.
176  /**
177  * @deprecated To be removed after %Cantera 3.0; superseded by getFalloffCoeffs()
178  */
179  virtual void getParameters(double* params) const {
180  warn_deprecated("FalloffRate::getParameters",
181  "To be removed after Cantera 3.0; superseded by getFalloffCoeffs.");
182  }
183 
184  virtual void getParameters(AnyMap& node) const override;
185 
186  //! Evaluate reaction rate
187  //! @param shared_data data shared by all reactions of a given type
188  double evalFromStruct(const FalloffData& shared_data) {
189  updateTemp(shared_data.temperature, m_work.data());
190  m_rc_low = m_lowRate.evalRate(shared_data.logT, shared_data.recipT);
191  m_rc_high = m_highRate.evalRate(shared_data.logT, shared_data.recipT);
192  double thirdBodyConcentration;
193  if (shared_data.ready) {
194  thirdBodyConcentration = shared_data.conc_3b[m_rate_index];
195  } else {
196  thirdBodyConcentration = shared_data.conc_3b[0];
197  }
198  double pr = thirdBodyConcentration * m_rc_low / (m_rc_high + SmallNumber);
199 
200  // Apply falloff function
201  if (m_chemicallyActivated) {
202  // 1 / (1 + Pr) * F
203  pr = F(pr, m_work.data()) / (1.0 + pr);
204  return pr * m_rc_low;
205  }
206 
207  // Pr / (1 + Pr) * F
208  pr *= F(pr, m_work.data()) / (1.0 + pr);
209  return pr * m_rc_high;
210  }
211 
212  virtual void check(const string& equation) override;
213  virtual void validate(const string& equation, const Kinetics& kin) override;
214 
215  //! Get flag indicating whether negative A values are permitted

217  return m_negativeA_ok;

221  void setAllowNegativePreExponentialFactor(bool value) {
222  m_negativeA_ok = value;
223  }
224 
225  //! Get flag indicating whether reaction is chemically activated
226  bool chemicallyActivated() const {
227  return m_chemicallyActivated;
228  }
229 
230  //! Set flag indicating whether reaction is chemically activated
231  void setChemicallyActivated(bool activated) {
232  m_chemicallyActivated = activated;
233  }
234

236  ArrheniusRate& lowRate() {
237  return m_lowRate;
238  }
239 
240  //! Set reaction rate in the low-pressure limit
241  void setLowRate(const ArrheniusRate& low);
242 
243  //! Get reaction rate in the high-pressure limit
244  ArrheniusRate& highRate() {
245  return m_highRate;
246  }
247 
248  //! Set reaction rate in the high-pressure limit
249  void setHighRate(const ArrheniusRate& high);
250 
251 protected:
252  ArrheniusRate m_lowRate; //!< The reaction rate in the low-pressure limit
253  ArrheniusRate m_highRate; //!< The reaction rate in the high-pressure limit
254 
255  //! Flag labeling reaction as chemically activated
256  bool m_chemicallyActivated = false;
257  //! Flag indicating whether negative A values are permitted
258  bool m_negativeA_ok = false;
259 
260  double m_rc_low = NAN; //!< Evaluated reaction rate in the low-pressure limit
261  double m_rc_high = NAN; //!< Evaluated reaction rate in the high-pressure limit
262  vector<double> m_work; //!< Work vector
263 };

292 //! The 3- or 4-parameter Troe falloff parameterization.
293 /*!
294  * The falloff function defines the value of @f$ F @f$ in the following
295  * rate expression @cite gilbert1983
296  *
297  * @f[
298  * k = k_{\infty} \left( \frac{P_r}{1 + P_r} \right) F(T, P_r)
299  * @f]
300  * where
301  * @f[
302  * P_r = \frac{k_0 [M]}{k_{\infty}}
303  * @f]
304  *
305  * This parameterization is defined by
306  *
307  * @f[
308  * \log_{10} F(T, P_r) = \frac{\log_{10} F_{cent}(T)}{1 + f_1^2}
309  * @f]
310  * where
311  * @f[
312  * F_{cent}(T) = (1 - A)\exp\left(\frac{-T}{T_3}\right)
313  * + A \exp\left(\frac{-T}{T_1}\right) + \exp\left(\frac{-T_2}{T}\right)
314  * @f]
315  *
316  * @f[
317  * f_1 = \frac{\log_{10} P_r + C}{N - 0.14 (\log_{10} P_r + C)}
318  * @f]
319  *
320  * @f[
321  * C = -0.4 - 0.67 \log_{10} F_{cent}
322  * @f]
323  *
324  * @f[
325  * N = 0.75 - 1.27 \log_{10} F_{cent}
326  * @f]
327  *
328  * - If @f$ T_3 @f$ is zero, then the corresponding term is set to zero.
329  * - If @f$ T_1 @f$ is zero, then the corresponding term is set to zero.
330  * - If @f$ T_2 @f$ is zero, then the corresponding term is set to zero.
331  *
332  * @ingroup falloffGroup
333  */
334 class TroeRate final : public FalloffRate
335 {
336 public:
337  //! Constructor
338  TroeRate() : m_a(NAN), m_rt3(0.0), m_rt1(0.0), m_t2(0.0) {
339  m_work.resize(1);
340  }
341 
342  TroeRate(const AnyMap& node, const UnitStack& rate_units={});
343  TroeRate(const ArrheniusRate& low, const ArrheniusRate& high,
344  const vector<double>& c);
345 
346  unique_ptr<MultiRateBase> newMultiRate() const override {
347  return make_unique<MultiRate<TroeRate, FalloffData>>();
348  }
349 
350  //! Set coefficients used by parameterization
351  /*!

355  virtual void setFalloffCoeffs(const vector<double>& c) override;

406 //! The SRI falloff function
407 /*!
408  * This falloff function is based on the one originally due to Stewart et al.
409  * @cite stewart1989, which required three parameters @f$ a @f$, @f$ b @f$, and
410  * @f$ c @f$. Kee et al. @cite kee1989 generalized this slightly by adding two more
411  * parameters @f$ d @f$ and @f$ e @f$. (The original form corresponds to @f$ d = 1 @f$
412  * and @f$ e = 0 @f$.) In keeping with the nomenclature of Kee et al. @cite kee1989,
413  * the rate is referred to as the *SRI falloff function*.
414  *
415  * The falloff function defines the value of @f$ F @f$ in the following
416  * rate expression
417  * @f[
418  * k = k_{\infty} \left( \frac{P_r}{1 + P_r} \right) F
419  * @f]
420  * where
421  * @f[
422  * P_r = \frac{k_0 [M]}{k_{\infty}}
423  * @f]
424  *
425  * @f[
426  * F(T, P_r) = {\left[ a \; \exp\left(\frac{-b}{T}\right)
427  * + \exp\left(\frac{-T}{c}\right)\right]}^n \; d \; T^e
428  * @f]
429  * where
430  * @f[
431  * n = \frac{1.0}{1.0 + (\log_{10} P_r)^2}
432  * @f]
433  *
434  * @f$ c @f$ is required to be greater than or equal to zero. If it is zero, then the
435  * corresponding term is set to zero. @f$ d @f$ is required to be greater than zero.
436  *
437  * @ingroup falloffGroup
438  */
439 class SriRate final : public FalloffRate
440 {
441 public:
442  //! Constructor
443  SriRate() : m_a(NAN), m_b(-1.0), m_c(-1.0), m_d(-1.0), m_e(-1.0) {
444  m_work.resize(2);
445  }
446 
447  SriRate(const AnyMap& node, const UnitStack& rate_units={});
448

450  : SriRate()
451  {
452  m_lowRate = low;
453  m_highRate = high;
454  setFalloffCoeffs(c);
455  }
456

458  return make_unique<MultiRate<SriRate, FalloffData>>();
459  }

466  */

521 //! The 1- or 2-parameter Tsang falloff parameterization.
522 /*!
523  * The Tsang falloff model is adapted from that of Troe.
524  * It provides a constant or linear in temperature value for @f$ F_{cent} @f$:
525  * @f[ F_{cent} = A + B*T @f]
526  *
527  * The value of @f$ F_{cent} @f$ is then applied to Troe's model for the
528  * determination of the value of @f$ F(T, P_r) @f$:
529  * @f[ \log_{10} F(T, P_r) = \frac{\log_{10} F_{cent}(T)}{1 + f_1^2} @f]
530  * where
531  * @f[ f_1 = \frac{\log_{10} P_r + C}{N - 0.14 (\log_{10} P_r + C)} @f]
532  *
533  * @f[ C = -0.4 - 0.67 \log_{10} F_{cent} @f]
534  *
535  * @f[ N = 0.75 - 1.27 \log_{10} F_{cent} @f]
536  *
537  * References:
538  * * Example of reaction database developed by Tsang utilizing this format
539  * @cite tsang1991
540  * * Example of Chemkin implementation of Tsang format (supplemental materials)
541  * @cite lucassen2011
542  *
543  * @ingroup falloffGroup
544  */
