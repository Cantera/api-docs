304void Phase::setMoleFractions(const double* const x)
305{
306 // Use m_y as a temporary work vector for the non-negative mole fractions
307 double norm = 0.0;
308 // sum is calculated below as the unnormalized molecular weight
309 double sum = 0;
310 for (size_t k = 0; k < m_kk; k++) {
311 double xk = std::max(x[k], 0.0); // Ignore negative mole fractions
312 m_y[k] = xk;
313 norm += xk;
314 sum += m_molwts[k] * xk;
315 }
316
317 // Set m_ym to the normalized mole fractions divided by the normalized mean
318 // molecular weight:
319 // m_ym_k = X_k / (sum_k X_k M_k)
320 const double invSum = 1.0/sum;
321 for (size_t k=0; k < m_kk; k++) {
322 m_ym[k] = m_y[k]*invSum;
323 }
324
325 // Now set m_y to the normalized mass fractions:
326 // m_y = X_k M_k / (sum_k X_k M_k)
327 for (size_t k=0; k < m_kk; k++) {
328 m_y[k] = m_ym[k] * m_molwts[k];
329 }
330
331 // Calculate the normalized molecular weight
332 m_mmw = sum/norm;
333 compositionChanged();
334}

356void Phase::setMassFractions(const double* const y)
357{
358 for (size_t k = 0; k < m_kk; k++) {
359 m_y[k] = std::max(y[k], 0.0); // Ignore negative mass fractions
360 }
361 double norm = accumulate(m_y.begin(), m_y.end(), 0.0);
362 scale(m_y.begin(), m_y.end(), m_y.begin(), 1.0/norm);
363
364 transform(m_y.begin(), m_y.end(), m_rmolwts.begin(),
365 m_ym.begin(), multiplies<double>());
366 m_mmw = 1.0 / accumulate(m_ym.begin(), m_ym.end(), 0.0);
367 compositionChanged();
368}

600void Phase::setConcentrations(const double* const conc)
601{
602 assertCompressible("setConcentrations");
603
604 // Use m_y as temporary storage for non-negative concentrations
605 double sum = 0.0, norm = 0.0;
606 for (size_t k = 0; k != m_kk; ++k) {
607 double ck = std::max(conc[k], 0.0); // Ignore negative concentrations
608 m_y[k] = ck;
609 sum += ck * m_molwts[k];
610 norm += ck;
611 }
612 m_mmw = sum/norm;
613 setDensity(sum);
614 double rsum = 1.0/sum;
615 for (size_t k = 0; k != m_kk; ++k) {
616 m_ym[k] = m_y[k] * rsum;
617 m_y[k] = m_ym[k] * m_molwts[k]; // m_y is now the mass fraction
618 }
619 compositionChanged();
620}

641void Phase::setMolesNoTruncate(const double* const N)
642{
643 // get total moles
644 copy(N, N + m_kk, m_ym.begin());
645 double totalMoles = accumulate(m_ym.begin(), m_ym.end(), 0.0);
646 // get total mass
647 copy(N, N + m_kk, m_y.begin());
648 transform(m_y.begin(), m_y.end(), m_molwts.begin(), m_y.begin(), multiplies<double>());
649 double totalMass = accumulate(m_y.begin(), m_y.end(), 0.0);
650 // mean molecular weight
651 m_mmw = totalMass/totalMoles;
652 // mass fractions
653 scale(m_y.begin(), m_y.end(), m_y.begin(), 1/totalMass);
654 // moles fractions/m_mmw
655 scale(m_ym.begin(), m_ym.end(), m_ym.begin(), 1/(m_mmw * totalMoles));
656 // composition has changed
657 compositionChanged();
658}

660doublereal Phase::elementalMassFraction(const size_t m) const
661{
662 checkElementIndex(m);
663 doublereal Z_m = 0.0;
664 for (size_t k = 0; k != m_kk; ++k) {
665 Z_m += nAtoms(k, m) * atomicWeight(m) / molecularWeight(k)
666 * massFraction(k);
667 }
668 return Z_m;
669}

671doublereal Phase::elementalMoleFraction(const size_t m) const
672{
673 checkElementIndex(m);
674 double denom = 0;
675 for (size_t k = 0; k < m_kk; k++) {
676 double atoms = 0;
677 for (size_t j = 0; j < nElements(); j++) {
678 atoms += nAtoms(k, j);
679 }
680 denom += atoms * moleFraction(k);
681 }
682 doublereal numerator = 0.0;
683 for (size_t k = 0; k != m_kk; ++k) {
684 numerator += nAtoms(k, m) * moleFraction(k);
685 }
686 return numerator / denom;
687}

756size_t Phase::addElement(const std::string& symbol, doublereal weight,
757 int atomic_number, doublereal entropy298,
758 int elem_type)
759{
760 // Look up the atomic weight if not given
761 if (weight == 0.0) {
762 try {
763 weight = getElementWeight(symbol);
764 } catch (CanteraError&) {
765 // assume this is just a custom element with zero atomic weight
766 }
767 } else if (weight == -12345.0) {
768 weight = getElementWeight(symbol);
769 }
770
771 // Try to look up the standard entropy if not given. Fail silently.
772 if (entropy298 == ENTROPY298_UNKNOWN) {
773 try {
774 const static AnyMap db = AnyMap::fromYamlFile(
775 "element-standard-entropies.yaml");
776 const AnyMap& elem = db["elements"].getMapWhere("symbol", symbol);
777 entropy298 = elem.convert("entropy298", "J/kmol/K", ENTROPY298_UNKNOWN);
778 } catch (CanteraError&) {
779 }
780 }
781
782 // Check for duplicates
783 auto iter = find(m_elementNames.begin(), m_elementNames.end(), symbol);
784 if (iter != m_elementNames.end()) {
785 size_t m = iter - m_elementNames.begin();
786 if (m_atomicWeights[m] != weight) {
787 throw CanteraError("Phase::addElement",
788 "Duplicate elements ({}) have different weights", symbol);
789 } else {
790 // Ignore attempt to add duplicate element with the same weight
791 return m;
792 }
793 }
794
795 // Add the new element
796 m_atomicWeights.push_back(weight);
797 m_elementNames.push_back(symbol);
798 m_atomicNumbers.push_back(atomic_number);
799 m_entropy298.push_back(entropy298);
800 if (symbol == "E") {
801 m_elem_type.push_back(CT_ELEM_TYPE_ELECTRONCHARGE);
802 } else {
803 m_elem_type.push_back(elem_type);
804 }
805 m_mm++;
806
807 // Update species compositions
808 if (m_kk) {
809 vector_fp old(m_speciesComp);
810 m_speciesComp.resize(m_kk*m_mm, 0.0);
811 for (size_t k = 0; k < m_kk; k++) {
812 size_t m_old = m_mm - 1;
813 for (size_t m = 0; m < m_old; m++) {
814 m_speciesComp[k * m_mm + m] = old[k * (m_old) + m];
815 }
816 m_speciesComp[k * (m_mm) + (m_mm-1)] = 0.0;
817 }
818 }
819
820 return m_mm-1;
821}

823bool Phase::addSpecies(shared_ptr<Species> spec) {
824 if (m_species.find(spec->name) != m_species.end()) {
825 throw CanteraError("Phase::addSpecies",
826 "Phase '{}' already contains a species named '{}'.",
827 m_name, spec->name);
828 }
829
830 vector_fp comp(nElements());
831 for (const auto& [eName, stoich] : spec->composition) {
832 size_t m = elementIndex(eName);
833 if (m == npos) { // Element doesn't exist in this phase
834 switch (m_undefinedElementBehavior) {
835 case UndefElement::ignore:
836 return false;
837
838 case UndefElement::add:
839 addElement(eName);
840 comp.resize(nElements());
841 m = elementIndex(eName);
842 break;
843
844 case UndefElement::error:
845 default:
846 throw CanteraError("Phase::addSpecies",
847 "Species '{}' contains an undefined element '{}'.",
848 spec->name, eName);
849 }
850 }
851 comp[m] = stoich;
852 }
853
854 size_t ne = nElements();
855 const vector_fp& aw = atomicWeights();
856 if (spec->charge != 0.0) {
857 size_t eindex = elementIndex("E");
858 if (eindex != npos) {
859 doublereal ecomp = comp[eindex];
860 if (fabs(spec->charge + ecomp) > 0.001) {
861 if (ecomp != 0.0) {
862 throw CanteraError("Phase::addSpecies",
863 "Input charge and element E compositions differ "
864 "for species " + spec->name);
865 } else {
866 // Just fix up the element E composition based on the input
867 // species charge
868 comp[eindex] = -spec->charge;
869 }
870 }
871 } else {
872 addElement("E", 0.000545, 0, 0.0, CT_ELEM_TYPE_ELECTRONCHARGE);
873 ne = nElements();
874 eindex = elementIndex("E");
875 comp.resize(ne);
876 comp[ne - 1] = - spec->charge;
877 }
878 }
879
880 double wt = 0.0;
881 for (size_t m = 0; m < ne; m++) {
882 wt += comp[m] * aw[m];
883 }
884
885 // Some surface phases may define species representing empty sites
886 // that have zero molecular weight. Give them a very small molecular
887 // weight to avoid dividing by zero.
888 wt = std::max(wt, Tiny);
889
890 spec->setMolecularWeight(wt);
891
892 m_molwts.push_back(wt);
893 m_rmolwts.push_back(1.0/wt);
894
895 for (size_t m = 0; m < ne; m++) {
896 m_speciesComp.push_back(comp[m]);
897 }
898
899 m_speciesNames.push_back(spec->name);
900 m_species[spec->name] = spec;
901 m_speciesIndices[spec->name] = m_kk;
902 m_speciesCharge.push_back(spec->charge);
903
904 std::string nLower = toLowerCopy(spec->name);
905 if (m_speciesLower.find(nLower) == m_speciesLower.end()) {
906 m_speciesLower[nLower] = m_kk;
907 } else {
908 m_speciesLower[nLower] = npos;
909 }
910 m_kk++;
911
912 // Ensure that the Phase has a valid mass fraction vector that sums to
913 // one. We will assume that species 0 has a mass fraction of 1.0 and mass
914 // fraction of all other species is 0.0.
915 if (m_kk == 1) {
916 m_y.push_back(1.0);
917 m_ym.push_back(m_rmolwts[0]);
918 m_mmw = 1.0 / m_ym[0];
919 } else {
920 m_y.push_back(0.0);
921 m_ym.push_back(0.0);
922 }
923 invalidateCache();
924 return true;
925}

1046{

1060void Phase::moleFractionsToMassFractions(const double* X, double* Y) const
1061{

1065 "no input composition given");
1066 }

1068 for (size_t k = 0; k != m_kk; ++k) {
1069 Y[k] = X[k]*m_molwts[k]*rmmw;
1070 }
