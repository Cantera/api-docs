68 warn_deprecated("Wall::Q",
69 "To be removed after Cantera 3.0; replaceable by 'heatRate'.");
70 double q1 = (m_area * m_rrth) *
71 (m_left->temperature() - m_right->temperature());
72 if (m_emiss > 0.0) {
