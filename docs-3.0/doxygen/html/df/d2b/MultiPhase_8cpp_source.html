39void MultiPhase::addPhase(ThermoPhase* p, doublereal moles)
40{
41 if (m_init) {
42 throw CanteraError("MultiPhase::addPhase",
43 "phases cannot be added after init() has been called.");
44 }
45
46 if (!p->compatibleWithMultiPhase()) {
47 throw CanteraError("MultiPhase::addPhase", "Phase '{}'' is not "
48 "compatible with MultiPhase equilibrium solver", p->name());
49 }
50
51 // save the pointer to the phase object
52 m_phase.push_back(p);
53
54 // store its number of moles
55 m_moles.push_back(moles);
56 m_temp_OK.push_back(true);
57
58 // update the total number of species
59 m_nsp += p->nSpecies();
60
61 // determine if this phase has new elements for each new element, add an
62 // entry in the map from names to index number + 1:
63
64 // iterate over the elements in this phase
65 for (size_t m = 0; m < p->nElements(); m++) {
66 string ename = p->elementName(m);
67
68 // if no entry is found for this element name, then it is a new element.
69 // In this case, add the name to the list of names, increment the
70 // element count, and add an entry to the name->(index+1) map.
71 if (m_enamemap.find(ename) == m_enamemap.end()) {
72 m_enamemap[ename] = m_nel + 1;
73 m_enames.push_back(ename);
74 m_atomicNumber.push_back(p->atomicNumber(m));
75
76 // Element 'E' (or 'e') is special. Note its location.
77 if (ename == "E" || ename == "e") {
78 m_eloc = m_nel;
79 }
80
81 m_nel++;
82 }
83 }
84
85 // If the mixture temperature hasn't been set, then set the temperature and
86 // pressure to the values for the phase being added. There is no good way to
87 // do this. However, this will be overridden later.
88 if (m_temp == 298.15 && p->temperature() > 2.0E-3) {
89 m_temp = p->temperature();
90 m_press = p->pressure();
91 }
92
93 // If this is a solution phase, update the minimum and maximum mixture
94 // temperatures. Stoichiometric phases are excluded, since a mixture may
95 // define multiple stoichiometric phases, each of which has thermo data
96 // valid only over a limited range. For example, a mixture might be defined
97 // to contain a phase representing water ice and one representing liquid
98 // water, only one of which should be present if the mixture represents an
99 // equilibrium state.
100 if (p->nSpecies() > 1) {
101 m_Tmin = std::max(p->minTemp(), m_Tmin);
102 m_Tmax = std::min(p->maxTemp(), m_Tmax);
103 }
104}

106void MultiPhase::init()
107{
108 if (m_init) {
109 return;
110 }
111
112 // allocate space for the atomic composition matrix
113 m_atoms.resize(m_nel, m_nsp, 0.0);
114 m_moleFractions.resize(m_nsp, 0.0);
115 m_elemAbundances.resize(m_nel, 0.0);
116
117 // iterate over the elements
118 // -> fill in m_atoms(m,k), m_snames(k), m_spphase(k), m_spstart(ip)
119 for (size_t m = 0; m < m_nel; m++) {
120 size_t k = 0;
121 // iterate over the phases
122 for (size_t ip = 0; ip < nPhases(); ip++) {
123 ThermoPhase* p = m_phase[ip];
124 size_t nsp = p->nSpecies();
125 size_t mlocal = p->elementIndex(m_enames[m]);
126 for (size_t kp = 0; kp < nsp; kp++) {
127 if (mlocal != npos) {
128 m_atoms(m, k) = p->nAtoms(kp, mlocal);
129 }
130 if (m == 0) {
131 m_snames.push_back(p->speciesName(kp));
132 if (kp == 0) {
133 m_spstart.push_back(m_spphase.size());
134 }
135 m_spphase.push_back(ip);
136 }
137 k++;
138 }
139 }
140 }
141
142 // set the initial composition within each phase to the
143 // mole fractions stored in the phase objects
144 m_init = true;
145 uploadMoleFractionsFromPhases();
146 updatePhases();
147}

180doublereal MultiPhase::elementMoles(size_t m) const
181{
182 doublereal sum = 0.0;
183 for (size_t i = 0; i < nPhases(); i++) {
184 double phasesum = 0.0;
185 size_t nsp = m_phase[i]->nSpecies();
186 for (size_t ik = 0; ik < nsp; ik++) {
187 size_t k = speciesIndex(ik, i);
188 phasesum += m_atoms(m,k)*m_moleFractions[k];
189 }
190 sum += phasesum * m_moles[i];
191 }
192 return sum;
193}

220doublereal MultiPhase::phaseCharge(size_t p) const
221{
222 doublereal phasesum = 0.0;
223 size_t nsp = m_phase[p]->nSpecies();
224 for (size_t ik = 0; ik < nsp; ik++) {
225 size_t k = speciesIndex(ik, p);
226 phasesum += m_phase[p]->charge(ik)*m_moleFractions[k];
227 }
228 return Faraday*phasesum*m_moles[p];
229}

245 // iterate over the phases

360void MultiPhase::getMoles(doublereal* molNum) const
361{
362 // First copy in the mole fractions
363 copy(m_moleFractions.begin(), m_moleFractions.end(), molNum);
364 doublereal* dtmp = molNum;
365 for (size_t ip = 0; ip < nPhases(); ip++) {
366 doublereal phasemoles = m_moles[ip];
367 ThermoPhase* p = m_phase[ip];
368 size_t nsp = p->nSpecies();
369 for (size_t ik = 0; ik < nsp; ik++) {
370 *(dtmp++) *= phasemoles;
371 }
372 }

375void MultiPhase::setMoles(const doublereal* n)
376{
377 if (!m_init) {
378 init();
379 }
380 size_t loc = 0;
381 size_t k = 0;
382 for (size_t ip = 0; ip < nPhases(); ip++) {
383 ThermoPhase* p = m_phase[ip];
384 size_t nsp = p->nSpecies();
385 double phasemoles = 0.0;
386 for (size_t ik = 0; ik < nsp; ik++) {
387 phasemoles += n[k];
388 k++;
389 }
390 m_moles[ip] = phasemoles;
391 if (nsp > 1) {
392 if (phasemoles > 0.0) {
393 p->setState_TPX(m_temp, m_press, n + loc);
394 p->getMoleFractions(&m_moleFractions[loc]);
395 } else {
396 p->getMoleFractions(&m_moleFractions[loc]);
397 }
398 } else {
399 m_moleFractions[loc] = 1.0;
400 }
401 loc += nsp;
402 }

440void MultiPhase::calcElemAbundances() const
441{
442 size_t loc = 0;
443 doublereal spMoles;
444 for (size_t eGlobal = 0; eGlobal < m_nel; eGlobal++) {
445 m_elemAbundances[eGlobal] = 0.0;
446 }
447 for (size_t ip = 0; ip < nPhases(); ip++) {
448 ThermoPhase* p = m_phase[ip];
449 size_t nspPhase = p->nSpecies();
450 doublereal phasemoles = m_moles[ip];
451 for (size_t ik = 0; ik < nspPhase; ik++) {
452 size_t kGlobal = loc + ik;
453 spMoles = m_moleFractions[kGlobal] * phasemoles;
454 for (size_t eGlobal = 0; eGlobal < m_nel; eGlobal++) {
455 m_elemAbundances[eGlobal] += m_atoms(eGlobal, kGlobal) * spMoles;
456 }
457 }
458 loc += nspPhase;
459 }

472double MultiPhase::equilibrate_MultiPhaseEquil(int XY, doublereal err,
473 int maxsteps, int maxiter,
474 int loglevel)
475{
476 bool strt = false;
477 doublereal dta = 0.0;
478 if (!m_init) {
479 init();
480 }
481
482 if (XY == TP) {
483 // create an equilibrium manager
484 MultiPhaseEquil e(this);
485 return e.equilibrate(XY, err, maxsteps, loglevel);
486 } else if (XY == HP) {
487 double h0 = enthalpy();
488 double Tlow = 0.5*m_Tmin; // lower bound on T
489 double Thigh = 2.0*m_Tmax; // upper bound on T
490 doublereal Hlow = Undef, Hhigh = Undef;
491 for (int n = 0; n < maxiter; n++) {
492 // if 'strt' is false, the current composition will be used as
493 // the starting estimate; otherwise it will be estimated
494 MultiPhaseEquil e(this, strt);
495 // start with a loose error tolerance, but tighten it as we get
496 // close to the final temperature
497
498 try {
499 e.equilibrate(TP, err, maxsteps, loglevel);
500 double hnow = enthalpy();
501 // the equilibrium enthalpy monotonically increases with T;
502 // if the current value is below the target, the we know the
503 // current temperature is too low. Set
504 if (hnow < h0) {
505 if (m_temp > Tlow) {
506 Tlow = m_temp;
507 Hlow = hnow;
508 }
509 } else {
510 // the current enthalpy is greater than the target;
511 // therefore the current temperature is too high.
512 if (m_temp < Thigh) {
513 Thigh = m_temp;
514 Hhigh = hnow;
515 }
516 }
517 double dt;
518 if (Hlow != Undef && Hhigh != Undef) {
519 double cpb = (Hhigh - Hlow)/(Thigh - Tlow);
520 dt = (h0 - hnow)/cpb;
521 dta = fabs(dt);
522 double dtmax = 0.5*fabs(Thigh - Tlow);
523 if (dta > dtmax) {
524 dt *= dtmax/dta;
525 }
526 } else {
527 double tnew = sqrt(Tlow*Thigh);
528 dt = tnew - m_temp;
529 }
530
531 double herr = fabs((h0 - hnow)/h0);
532
533 if (herr < err) {
534 return err;
535 }
536 double tnew = m_temp + dt;
537 if (tnew < 0.0) {
538 tnew = 0.5*m_temp;
539 }
540 setTemperature(tnew);
541
542 // if the size of Delta T is not too large, use
543 // the current composition as the starting estimate
544 if (dta < 100.0) {
545 strt = false;
546 }
547
548 } catch (CanteraError&) {
549 if (!strt) {
550 strt = true;
551 } else {
552 double tnew = 0.5*(m_temp + Thigh);
553 if (fabs(tnew - m_temp) < 1.0) {
554 tnew = m_temp + 1.0;
555 }
556 setTemperature(tnew);
557 }
558 }
559 }
560 throw CanteraError("MultiPhase::equilibrate_MultiPhaseEquil",
561 "No convergence for T");
562 } else if (XY == SP) {
563 double s0 = entropy();
564 double Tlow = 1.0; // lower bound on T
565 double Thigh = 1.0e6; // upper bound on T
566 for (int n = 0; n < maxiter; n++) {
567 MultiPhaseEquil e(this, strt);
568
569 try {
570 e.equilibrate(TP, err, maxsteps, loglevel);
571 double snow = entropy();
572 if (snow < s0) {
573 Tlow = std::max(Tlow, m_temp);
574 } else {
575 Thigh = std::min(Thigh, m_temp);
576 }
577 double dt = (s0 - snow)*m_temp/cp();
578 double dtmax = 0.5*fabs(Thigh - Tlow);
579 dtmax = (dtmax > 500.0 ? 500.0 : dtmax);
580 dta = fabs(dt);
581 if (dta > dtmax) {
582 dt *= dtmax/dta;
583 }
584 if (dta < 1.0e-4) {
585 return err;
586 }
587 double tnew = m_temp + dt;
588 setTemperature(tnew);
589
590 // if the size of Delta T is not too large, use
591 // the current composition as the starting estimate
592 if (dta < 100.0) {
593 strt = false;
594 }
595 } catch (CanteraError&) {
596 if (!strt) {
597 strt = true;
598 } else {
599 double tnew = 0.5*(m_temp + Thigh);
600 setTemperature(tnew);
601 }
602 }
603 }
604 throw CanteraError("MultiPhase::equilibrate_MultiPhaseEquil",
605 "No convergence for T");
606 } else if (XY == TV) {
607 doublereal v0 = volume();
608 bool start = true;
609 for (int n = 0; n < maxiter; n++) {
610 double pnow = pressure();
611 MultiPhaseEquil e(this, start);
612 start = false;
613
614 e.equilibrate(TP, err, maxsteps, loglevel);
615 double vnow = volume();
616 double verr = fabs((v0 - vnow)/v0);
617
618 if (verr < err) {
619 return err;
620 }
621 // find dV/dP
622 setPressure(pnow*1.01);
623 double dVdP = (volume() - vnow)/(0.01*pnow);
624 setPressure(pnow + 0.5*(v0 - vnow)/dVdP);
625 }
626 } else {
627 throw CanteraError("MultiPhase::equilibrate_MultiPhaseEquil",
628 "unknown option");
629 }
630 return -1.0;
631}

633void MultiPhase::equilibrate(const std::string& XY, const std::string& solver,
634 double rtol, int max_steps, int max_iter,
635 int estimate_equil, int log_level)
636{
637 // Save the initial state so that it can be restored in case one of the
638 // solvers fails
639 vector_fp initial_moleFractions = m_moleFractions;
640 vector_fp initial_moles = m_moles;
641 double initial_T = m_temp;
642 double initial_P = m_press;
643 int ixy = _equilflag(XY.c_str());
644 if (solver == "auto" || solver == "vcs") {
645 try {
646 debuglog("Trying VCS equilibrium solver\n", log_level);
647 vcs_MultiPhaseEquil eqsolve(this, log_level-1);
648 int ret = eqsolve.equilibrate(ixy, estimate_equil, log_level-1,
649 rtol, max_steps);
650 if (ret) {
651 throw CanteraError("MultiPhase::equilibrate",
652 "VCS solver failed. Return code: {}", ret);
653 }
654 debuglog("VCS solver succeeded\n", log_level);
655 return;
656 } catch (std::exception& err) {
657 debuglog("VCS solver failed.\n", log_level);
658 debuglog(err.what(), log_level);
659 m_moleFractions = initial_moleFractions;
660 m_moles = initial_moles;
661 m_temp = initial_T;
662 m_press = initial_P;
663 updatePhases();
664 if (solver == "auto") {
665 } else {
666 throw;
667 }
668 }
669 }
670
671 if (solver == "auto" || solver == "gibbs") {
672 try {
673 debuglog("Trying MultiPhaseEquil (Gibbs) equilibrium solver\n",
674 log_level);
675 equilibrate_MultiPhaseEquil(ixy, rtol, max_steps, max_iter,
676 log_level-1);
677 debuglog("MultiPhaseEquil solver succeeded\n", log_level);
678 return;
679 } catch (std::exception& err) {
680 debuglog("MultiPhaseEquil solver failed.\n", log_level);
681 debuglog(err.what(), log_level);
682 m_moleFractions = initial_moleFractions;
683 m_moles = initial_moles;
684 m_temp = initial_T;
685 m_press = initial_P;
686 updatePhases();
687 throw;
688 }
689 }
690
691 if (solver != "auto") {
692 throw CanteraError("MultiPhase::equilibrate",
693 "Invalid solver specified: '" + solver + "'");
694 }
695}
