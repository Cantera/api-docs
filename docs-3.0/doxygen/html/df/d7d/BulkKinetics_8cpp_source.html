26 bool BulkKinetics::addReaction(shared_ptr<Reaction> r, bool resize)
27 {
28  bool added = Kinetics::addReaction(r, resize);
29  if (!added) {
30  // undeclared species, etc.
31  return false;
32  }
33  double dn = 0.0;
34  for (const auto& [name, stoich] : r->products) {
35  dn += stoich;
36  }
37  for (const auto& [name, stoich] : r->reactants) {
38  dn -= stoich;
39  }
40 
41  m_dn.push_back(dn);
42 
43  if (r->reversible) {
44  m_revindex.push_back(nReactions()-1);
45  } else {
46  m_irrev.push_back(nReactions()-1);
47  }
48 
49  shared_ptr<ReactionRate> rate = r->rate();
50  string rtype = rate->subType();
51  if (rtype == "") {
52  rtype = rate->type();
53  }
54 
55  // If necessary, add new MultiRate evaluator
56  if (m_bulk_types.find(rtype) == m_bulk_types.end()) {
57  m_bulk_types[rtype] = m_bulk_rates.size();
58  m_bulk_rates.push_back(rate->newMultiRate());
59  m_bulk_rates.back()->resize(m_kk, nReactions(), nPhases());
60  }
61 
62  // Set index of rate to number of reaction within kinetics
63  rate->setRateIndex(nReactions() - 1);
64  rate->setContext(*r, *this);
65 
66  // Add reaction rate to evaluator
67  size_t index = m_bulk_types[rtype];
68  m_bulk_rates[index]->add(nReactions() - 1, *rate);
69 
70  // Add reaction to third-body evaluator
71  if (r->thirdBody() != nullptr) {
72  addThirdBody(r);
73  }
74 
75  m_concm.push_back(NAN);
76  m_ready = resize;
77  return true;
78 }

80 void BulkKinetics::addThirdBody(shared_ptr<Reaction> r)
81 {
82  map<size_t, double> efficiencies;
83  for (const auto& [name, efficiency] : r->thirdBody()->efficiencies) {
84  size_t k = kineticsSpeciesIndex(name);
85  if (k != npos) {
86  efficiencies[k] = efficiency;
87  } else if (!m_skipUndeclaredThirdBodies) {
88  throw CanteraError("BulkKinetics::addThirdBody", "Found third-body"
89  " efficiency for undefined species '{}' while adding reaction '{}'",
90  name, r->equation());
91  }
92  }
93  m_multi_concm.install(nReactions() - 1, efficiencies,
94  r->thirdBody()->default_efficiency,
95  r->thirdBody()->mass_action);
96 }

98 void BulkKinetics::modifyReaction(size_t i, shared_ptr<Reaction> rNew)
99 {
100  // operations common to all reaction types
101  Kinetics::modifyReaction(i, rNew);
102 
103  shared_ptr<ReactionRate> rate = rNew->rate();
104  string rtype = rate->subType();
105  if (rtype == "") {
106  rtype = rate->type();
107  }
108 
109  // Ensure that MultiRate evaluator is available
110  if (m_bulk_types.find(rtype) == m_bulk_types.end()) {
111  throw CanteraError("BulkKinetics::modifyReaction",
112  "Evaluator not available for type '{}'.", rtype);
113  }
114 
115  // Replace reaction rate to evaluator
116  size_t index = m_bulk_types[rtype];
117  rate->setRateIndex(i);
118  rate->setContext(*rNew, *this);
119  m_bulk_rates[index]->replace(i, *rate);
120  invalidateCache();
121 }

173 void BulkKinetics::getEquilibriumConstants(double* kc)
174 {
175  updateROP();
176 
177  vector<double>& delta_gibbs0 = m_rbuf0;
178  fill(delta_gibbs0.begin(), delta_gibbs0.end(), 0.0);
179 
180  // compute Delta G^0 for all reactions
181  getReactionDelta(m_grt.data(), delta_gibbs0.data());
182 
183  double rrt = 1.0 / thermo().RT();
184  double logStandConc = log(thermo().standardConcentration());
185  for (size_t i = 0; i < nReactions(); i++) {
186  kc[i] = exp(-delta_gibbs0[i] * rrt + m_dn[i] * logStandConc);
187  }
188 }

190 void BulkKinetics::getRevRateConstants(double* krev, bool doIrreversible)
191 {
192  // go get the forward rate constants. -> note, we don't really care about
193  // speed or redundancy in these informational routines.
194  getFwdRateConstants(krev);
195 
196  if (doIrreversible) {
197  getEquilibriumConstants(m_rbuf0.data());
198  for (size_t i = 0; i < nReactions(); i++) {
199  krev[i] /= m_rbuf0[i];
200  }
201  } else {
202  // m_rkcn[] is zero for irreversible reactions
203  for (size_t i = 0; i < nReactions(); i++) {
204  krev[i] *= m_rkcn[i];
205  }
206  }
207 }

462 void BulkKinetics::updateROP()
463 {
464  static const int cacheId = m_cache.getId();
465  CachedScalar last = m_cache.getScalar(cacheId);
466  double T = thermo().temperature();
467  double rho = thermo().density();
468  int statenum = thermo().stateMFNumber();
469 
470  if (last.state1 != T || last.state2 != rho) {
471  // Update properties that are independent of the composition
472  thermo().getStandardChemPotentials(m_grt.data());
473  fill(m_delta_gibbs0.begin(), m_delta_gibbs0.end(), 0.0);
474  double logStandConc = log(thermo().standardConcentration());
475 
476  // compute Delta G^0 for all reversible reactions
477  getRevReactionDelta(m_grt.data(), m_delta_gibbs0.data());
478 
479  double rrt = 1.0 / thermo().RT();
480  for (size_t i = 0; i < m_revindex.size(); i++) {
481  size_t irxn = m_revindex[i];
482  m_rkcn[irxn] = std::min(
483  exp(m_delta_gibbs0[irxn] * rrt - m_dn[irxn] * logStandConc), BigNumber);
484  }
485 
486  for (size_t i = 0; i != m_irrev.size(); ++i) {
487  m_rkcn[ m_irrev[i] ] = 0.0;
488  }
489  }
490 
491  if (!last.validate(T, rho, statenum)) {
492  // Update terms dependent on species concentrations and temperature
493  thermo().getActivityConcentrations(m_act_conc.data());
494  thermo().getConcentrations(m_phys_conc.data());
495  double ctot = thermo().molarDensity();
496 
497  // Third-body objects interacting with MultiRate evaluator
498  m_multi_concm.update(m_phys_conc, ctot, m_concm.data());
499 
500  // loop over MultiRate evaluators for each reaction type
501  for (auto& rates : m_bulk_rates) {
502  bool changed = rates->update(thermo(), *this);
503  if (changed) {
504  rates->getRateConstants(m_kf0.data());
505  }
506  }
507  m_ROP_ok = false;
508  }
509 
510  if (m_ROP_ok) {
511  // rates of progress are up-to-date only if both the thermodynamic state
512  // and m_perturb are unchanged
513  return;
514  }
515 
516  // Scale the forward rate coefficient by the perturbation factor
517  for (size_t i = 0; i < nReactions(); ++i) {
518  m_rfn[i] = m_kf0[i] * m_perturb[i];
519  }
520 
521  copy(m_rfn.begin(), m_rfn.end(), m_ropf.data());
522  processThirdBodies(m_ropf.data());
523  copy(m_ropf.begin(), m_ropf.end(), m_ropr.begin());
524 
525  // multiply ropf by concentration products
526  m_reactantStoich.multiply(m_act_conc.data(), m_ropf.data());
527 
528  // for reversible reactions, multiply ropr by concentration products
529  applyEquilibriumConstants(m_ropr.data());
530  m_revProductStoich.multiply(m_act_conc.data(), m_ropr.data());
531  for (size_t j = 0; j != nReactions(); ++j) {
532  m_ropnet[j] = m_ropf[j] - m_ropr[j];
533  }
534 
535  for (size_t i = 0; i < m_rfn.size(); i++) {
536  AssertFinite(m_rfn[i], "BulkKinetics::updateROP",
537  "m_rfn[{}] is not finite.", i);
538  AssertFinite(m_ropf[i], "BulkKinetics::updateROP",
539  "m_ropf[{}] is not finite.", i);
540  AssertFinite(m_ropr[i], "BulkKinetics::updateROP",
541  "m_ropr[{}] is not finite.", i);
542  }
543  m_ROP_ok = true;
544 }

569 void BulkKinetics::applyEquilibriumConstants_ddT(double* drkcn)
570 {
571  double T = thermo().temperature();
572  double P = thermo().pressure();
573  double rrt = 1. / thermo().RT();
574 
575  vector<double>& grt = m_sbuf0;
576  vector<double>& delta_gibbs0 = m_rbuf1;
577  fill(delta_gibbs0.begin(), delta_gibbs0.end(), 0.0);
578 
579  // compute perturbed Delta G^0 for all reversible reactions
580  thermo().saveState(m_state);
581  thermo().setState_TP(T * (1. + m_jac_rtol_delta), P);
582  thermo().getStandardChemPotentials(grt.data());
583  getRevReactionDelta(grt.data(), delta_gibbs0.data());
584 
585  // apply scaling for derivative of inverse equilibrium constant
586  double Tinv = 1. / T;
587  double rrt_dTinv = rrt * Tinv / m_jac_rtol_delta;
588  double rrtt = rrt * Tinv;
589  for (size_t i = 0; i < m_revindex.size(); i++) {
590  size_t irxn = m_revindex[i];
591  double factor = delta_gibbs0[irxn] - m_delta_gibbs0[irxn];
592  factor *= rrt_dTinv;
593  factor += m_dn[irxn] * Tinv - m_delta_gibbs0[irxn] * rrtt;
594  drkcn[irxn] *= factor;
595  }
596 
597  for (size_t i = 0; i < m_irrev.size(); ++i) {
598  drkcn[m_irrev[i]] = 0.0;
599  }
600 
601  thermo().restoreState(m_state);
602 }

622 void BulkKinetics::process_ddC(StoichManagerN& stoich, const vector<double>& in,
623  double* drop, bool mass_action)
624 {
625  Eigen::Map<Eigen::VectorXd> out(drop, nReactions());
626  out.setZero();
627  double ctot_inv = 1. / thermo().molarDensity();
628 
629  // derivatives due to concentrations in law of mass action
630  if (mass_action) {
631  stoich.scale(in.data(), out.data(), ctot_inv);
632  }
633  if (m_jac_skip_third_bodies || m_multi_concm.empty()) {
634  return;
635  }
636 
637  // derivatives due to third-body colliders in law of mass action
638  Eigen::Map<Eigen::VectorXd> outM(m_rbuf1.data(), nReactions());
639  if (mass_action) {
640  outM.fill(0.);
641  m_multi_concm.scale(in.data(), outM.data(), ctot_inv);
642  out += outM;
643  }
644 
645  // derivatives due to reaction rates depending on third-body colliders
646  if (!m_jac_skip_falloff) {
647  m_multi_concm.scaleM(in.data(), outM.data(), m_concm.data(), ctot_inv);
648  for (auto& rates : m_bulk_rates) {
649  // processing step assigns zeros to entries not dependent on M
650  rates->processRateConstants_ddM(
651  outM.data(), m_rfn.data(), m_jac_rtol_delta);
652  }
653  out += outM;
654  }

657 Eigen::SparseMatrix<double> BulkKinetics::calculateCompositionDerivatives(
658  StoichManagerN& stoich, const vector<double>& in, bool ddX)
659 {
660  Eigen::SparseMatrix<double> out;
661  vector<double>& scaled = m_rbuf1;
662  vector<double>& outV = m_rbuf2;
663 
664  // convert from concentration to mole fraction output
665  copy(in.begin(), in.end(), scaled.begin());
666  if (ddX) {
667  double ctot = thermo().molarDensity();
668  for (size_t i = 0; i < nReactions(); ++i) {
669  scaled[i] *= ctot;
670  }
671  }
672 
673  // derivatives handled by StoichManagerN
674  copy(scaled.begin(), scaled.end(), outV.begin());
675  processThirdBodies(outV.data());
676  out = stoich.derivatives(m_act_conc.data(), outV.data());
677  if (m_jac_skip_third_bodies || m_multi_concm.empty()) {
678  return out;
679  }
680 
681  // derivatives due to law of mass action
682  copy(scaled.begin(), scaled.end(), outV.begin());
683  stoich.multiply(m_act_conc.data(), outV.data());
684 
685  // derivatives due to reaction rates depending on third-body colliders
686  if (!m_jac_skip_falloff) {
687  for (auto& rates : m_bulk_rates) {
688  // processing step does not modify entries not dependent on M
689  rates->processRateConstants_ddM(
690  outV.data(), m_rfn.data(), m_jac_rtol_delta, false);
691  }
692  }
693 
694  // derivatives handled by ThirdBodyCalc
695  out += m_multi_concm.derivatives(outV.data());
696  return out;
697 }
