38 void ConstCpPoly::updateProperties(const double* tt,
39  double* cp_R,
40  double* h_RT,
41  double* s_R) const
42 {
43  double t = *tt;
44  double logt = log(t);
45  double rt = 1.0/t;
46  *cp_R = m_cp0_R;

49 }
