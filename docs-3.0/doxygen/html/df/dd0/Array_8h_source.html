17 //! A class for 2D arrays stored in column-major (Fortran-compatible) form.
18 /*!
19  * In this form, the data entry for an n row, m col matrix is
20  *
21  * index = i + (n-1) * j
22  *
23  * where
24  *
25  * J(i,j) = data_start + index
26  * i = row
27  * j = column
28  *
29  * @ingroup matrices
30  */
31 class Array2D
32 {

165  //! Returns a changeable reference to position in the matrix
166  /*!
167  * Returns a reference to the matrix's (i,j) element. This may be used as an

171  * @param j The column index
172  * @returns a changeable reference to the matrix entry
173  */
174  double& value(size_t i, size_t j) {

184  */
185  double value(size_t i, size_t j) const {
