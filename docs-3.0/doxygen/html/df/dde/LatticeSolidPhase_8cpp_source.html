26 double LatticeSolidPhase::minTemp(size_t k) const
27 {
28  if (k != npos) {
29  for (size_t n = 0; n < m_lattice.size(); n++) {
30  if (lkstart_[n+1] < k) {
31  return m_lattice[n]->minTemp(k-lkstart_[n]);
32  }
33  }
34  }
35  double mm = 1.0E300;
36  for (size_t n = 0; n < m_lattice.size(); n++) {
37  double ml = m_lattice[n]->minTemp();
38  mm = std::min(mm, ml);
39  }
40  return mm;
41 }

181 void LatticeSolidPhase::getMoleFractions(double* const x) const
182 {
183  size_t strt = 0;
184  // the ifdef block should be the way we calculate this.!!!!!
185  Phase::getMoleFractions(x);
186  for (size_t n = 0; n < m_lattice.size(); n++) {
187  size_t nsp = m_lattice[n]->nSpecies();
188  double sum = 0.0;
189  for (size_t k = 0; k < nsp; k++) {
190  sum += (x + strt)[k];
191  }
192  for (size_t k = 0; k < nsp; k++) {
193  (x + strt)[k] /= sum;
194  }
195 
196  // At this point we can check against the mole fraction vector of the
197  // underlying LatticePhase objects and get the same answer.
198  m_lattice[n]->getMoleFractions(&m_x[strt]);
199  for (size_t k = 0; k < nsp; k++) {
200  if (fabs((x + strt)[k] - m_x[strt+k]) > 1.0E-14) {
201  throw CanteraError("LatticeSolidPhase::getMoleFractions",
202  "internal error");
203  }
204  }
205  strt += nsp;
206  }
207 }

351 void LatticeSolidPhase::addLattice(shared_ptr<ThermoPhase> lattice)
352 {
353  m_lattice.push_back(lattice);
354  if (lkstart_.empty()) {
355  lkstart_.push_back(0);
356  }
357  lkstart_.push_back(lkstart_.back() + lattice->nSpecies());
358 
359  if (theta_.size() == 0) {
360  theta_.push_back(1.0);
361  } else {
362  theta_.push_back(0.0);
363  }
364 
365  for (size_t k = 0; k < lattice->nSpecies(); k++) {
366  ThermoPhase::addSpecies(lattice->species(k));
367  vector<double> constArr(lattice->nElements());
368  const vector<double>& aws = lattice->atomicWeights();
369  for (size_t es = 0; es < lattice->nElements(); es++) {
370  addElement(lattice->elementName(es), aws[es], lattice->atomicNumber(es),
371  lattice->entropyElement298(es), lattice->elementType(es));
372  }
373  m_x.push_back(lattice->moleFraction(k));
374  tmpV_.push_back(0.0);
375  }
376 }

378 void LatticeSolidPhase::setLatticeStoichiometry(const Composition& comp)
379 {
380  for (size_t i = 0; i < m_lattice.size(); i++) {
381  theta_[i] = getValue(comp, m_lattice[i]->name(), 0.0);
382  }
383  // Add in the lattice stoichiometry constraint
384  for (size_t i = 1; i < m_lattice.size(); i++) {
385  string econ = fmt::format("LC_{}_{}", i, name());
386  size_t m = addElement(econ, 0.0, 0, 0.0, CT_ELEM_TYPE_LATTICERATIO);
387  size_t mm = nElements();
388  for (size_t k = 0; k < m_lattice[0]->nSpecies(); k++) {
389  m_speciesComp[k * mm + m] = -theta_[0];
390  }
391  for (size_t k = 0; k < m_lattice[i]->nSpecies(); k++) {
392  size_t ks = lkstart_[i] + k;
393  m_speciesComp[ks * mm + m] = theta_[i];
394  }
395  }
